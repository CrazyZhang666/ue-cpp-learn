# 第2章案例：背包系统（纯C++控制台练习）

> **难度**：⭐⭐（入门级）
> **预计时间**：1小时
> **目标**：用纯C++（不涉及UE）实现一个简单的背包系统，练习本章学的所有知识。

---

## 案例目标

实现一个文字版背包系统，支持：

1. 添加物品（限制堆叠数量）
2. 删除物品
3. 查找物品
4. 显示背包中所有物品
5. 计算背包总重量

> **为什么是纯C++？** 在学习UE C++之前，先脱离引擎练习纯C++的变量、控制流、函数、类、容器等，把基础打牢。这个练习用VS2022的控制台项目或在线编译器都能跑。

## 第二步：完整代码

```cpp
// InventoryDemo.cpp
// 文字版背包系统 — 第2章综合练习

#include <iostream>
#include <vector>    // C++标准库的动态数组（等同于UE的TArray）
#include <string>    // C++标准库的字符串（等同于UE的FString）
#include <algorithm> // 查找等算法

using namespace std;  // 在UE中不要用这个，但普通C++练习中可以

// ============================================================
// 第一部分：定义物品结构体
// ============================================================

// enum = 枚举，给一组相关常量起名字
// 这个枚举定义了物品的稀有度等级
enum class EItemRarity  // "E"前缀是UE习惯，表示Enum
{
    Common,      // 0 - 普通（白色）
    Uncommon,    // 1 - 不凡（绿色）
    Rare,        // 2 - 稀有（蓝色）
    Epic,        // 3 - 史诗（紫色）
    Legendary    // 4 - 传说（橙色）
};

// 把稀有度转成中文文字的函数
string RarityToString(EItemRarity Rarity)
{
    // switch 用法练习
    switch (Rarity)
    {
        case EItemRarity::Common:    return "普通";
        case EItemRarity::Uncommon:  return "不凡";
        case EItemRarity::Rare:      return "稀有";
        case EItemRarity::Epic:      return "史诗";
        case EItemRarity::Legendary: return "传说";
        default:                     return "未知";
    }
}

// struct = 结构体，用来组合多个相关数据
// 每个"物品"由以下属性组成
struct FItem  // "F"前缀是UE习惯，表示Struct
{
    string Name;            // 物品名称
    string Description;     // 物品描述
    EItemRarity Rarity;     // 稀有度（使用上面定义的枚举）
    float Weight;           // 单件重量（kg）
    int MaxStack;           // 最大堆叠数量
    int CurrentStack;       // 当前堆叠数量

    // 构造函数：创建物品时自动调用
    // 参数后面的 ": 成员(值)" 叫"初始化列表"，比在函数体内赋值更高效
    FItem(string InName, string InDesc, EItemRarity InRarity,
           float InWeight, int InMaxStack)
        : Name(InName)               // Name = InName（初始化列表赋值）
        , Description(InDesc)
        , Rarity(InRarity)
        , Weight(InWeight)
        , MaxStack(InMaxStack)
        , CurrentStack(1)            // 默认堆叠数量为1
    {
        // 构造函数体：可以在这里做额外的初始化
        // 上面已经用初始化列表完成了成员初始化，这里不需要再写
    }

    // 成员函数：获取这个物品堆的总重量
    float GetTotalWeight() const  // const = 这个函数不会修改任何成员变量
    {
        return Weight * CurrentStack;
    }

    // 成员函数：是否还能继续堆叠
    bool CanStack() const
    {
        return CurrentStack < MaxStack;  // 当前数量 < 最大数量就可以继续堆
    }

    // 成员函数：添加堆叠（返回实际添加的数量）
    int AddToStack(int Amount)
    {
        int CanAdd = MaxStack - CurrentStack;  // 还能加几个
        int ActualAdd = (Amount < CanAdd) ? Amount : CanAdd;  // 取较小值（三元运算符练习）
        //                ─────┬─────    ──┬──    ──┬──
        //                 条件          true值   false值
        CurrentStack += ActualAdd;
        return ActualAdd;  // 返回实际加了多少个
    }
};


// ============================================================
// 第二部分：背包类
// ============================================================

class CInventory  // "C"前缀不是UE标准，这里表示纯C++类
{
private:  // 私有成员：外部不能直接访问
    vector<FItem> Items;       // 物品数组（等同于UE的TArray<FItem>）
    float MaxWeight;           // 背包最大负重

public:  // 公开成员：外部可以访问
    // 构造函数
    CInventory(float InMaxWeight)
        : MaxWeight(InMaxWeight)
    {
        // 空背包，不需要额外初始化
    }

    // ----- 添加物品 -----
    // 返回true表示添加成功，false表示失败（超重或其它原因）
    bool AddItem(const FItem& NewItem)
    //          ─────┬─────
    //          const引用：不拷贝，且保证不修改原对象
    {
        // 1. 先检查是否会超重
        float CurrentWeight = GetTotalWeight();
        float NewItemWeight = NewItem.GetTotalWeight();

        if (CurrentWeight + NewItemWeight > MaxWeight)
        {
            cout << "【失败】背包已满！当前负重: " << CurrentWeight
                 << "kg, 最大负重: " << MaxWeight << "kg" << endl;
            return false;  // 添加失败
        }

        // 2. 检查是否已有同种物品可以堆叠
        // 遍历背包中所有物品（引用遍历：可以修改元素）
        for (FItem& ExistingItem : Items)  // 注意这里有 &，所以是引用
        {
            if (ExistingItem.Name == NewItem.Name)  // 找到同种物品
            {
                if (ExistingItem.CanStack())
                {
                    int Added = ExistingItem.AddToStack(NewItem.CurrentStack);
                    cout << "【成功】" << NewItem.Name << " 堆叠+" << Added
                         << "，当前堆叠: " << ExistingItem.CurrentStack
                         << "/" << ExistingItem.MaxStack << endl;
                    return true;
                }
                else
                {
                    cout << "【失败】" << NewItem.Name << " 已达最大堆叠数（"
                         << ExistingItem.MaxStack << "）" << endl;
                    return false;
                }
            }
        }

        // 3. 没有同种物品可堆叠，作为新物品添加到背包
        Items.push_back(NewItem);  // push_back = TArray::Add
        cout << "【成功】获得了 " << NewItem.Name
             << "（" << RarityToString(NewItem.Rarity) << "）" << endl;
        return true;
    }

    // ----- 删除物品 -----
    bool RemoveItem(const string& ItemName, int Amount = 1)
    //                                  ──────┬─────
    //                                  默认参数：不指定数量时删1个
    {
        for (int i = 0; i < Items.size(); i++)
        {
            if (Items[i].Name == ItemName)
            {
                if (Items[i].CurrentStack <= Amount)
                {
                    // 要删除的数量>=当前堆叠，直接删掉这一整堆
                    cout << "【删除】" << Items[i].Name
                         << " x" << Items[i].CurrentStack << " 被移除" << endl;
                    Items.erase(Items.begin() + i);  // 删除第i个元素（等同于RemoveAt(i)）
                    //           ──────┬──────
                    //          begin()返回第0个元素的迭代器，+i就是第i个
                }
                else
                {
                    // 只减少堆叠数量
                    Items[i].CurrentStack -= Amount;
                    cout << "【删除】" << Items[i].Name << " -" << Amount
                         << "，剩余: " << Items[i].CurrentStack << endl;
                }
                return true;  // 删除成功
            }
        }

        cout << "【失败】背包中没有 " << ItemName << endl;
        return false;
    }

    // ----- 查找物品 -----
    // 返回指向找到物品的指针，找不到返回nullptr
    FItem* FindItem(const string& ItemName)
    {
        for (FItem& Item : Items)
        {
            if (Item.Name == ItemName)
            {
                return &Item;  // 返回这个物品的地址（指针练习！）
            }
        }
        return nullptr;  // 没找到，返回空指针
    }

    // ----- 计算总重量 -----
    float GetTotalWeight() const
    {
        float Total = 0.0f;
        for (const FItem& Item : Items)
        {
            Total += Item.GetTotalWeight();  // 累加每堆物品的重量
        }
        return Total;
    }

    // ----- 获取物品数量 -----
    int GetItemCount() const
    {
        return Items.size();  // 返回数组大小（堆的数量，不是物品总数）
    }

    // ----- 显示背包内容 -----
    void Display() const
    {
        cout << endl;
        cout << "╔══════════════════════════════════════╗" << endl;
        cout << "║        背 包 物 品 列 表              ║" << endl;
        cout << "╠══════════════════════════════════════╣" << endl;

        if (Items.empty())
        {
            cout << "║  （背包空空如也）                     ║" << endl;
        }
        else
        {
            for (int i = 0; i < Items.size(); i++)
            {
                const FItem& Item = Items[i];

                // 输出格式：[序号] 名称 x堆叠数 (稀有度) - 重量kg
                cout << "║ [" << (i + 1) << "] "
                     << Item.Name
                     << " x" << Item.CurrentStack
                     << " (" << RarityToString(Item.Rarity) << ")"
                     << " - " << Item.GetTotalWeight() << "kg";

                // 补齐空格使边框对齐（纯装饰）
                int LineLen = 8 + Item.Name.length()
                            + to_string(Item.CurrentStack).length()
                            + RarityToString(Item.Rarity).length()
                            + to_string((int)Item.GetTotalWeight()).length();
                for (int s = LineLen; s < 30; s++) cout << " ";
                cout << "║" << endl;
            }
        }

        cout << "╠══════════════════════════════════════╣" << endl;
        cout << "║ 总负重: " << GetTotalWeight() << "kg / "
             << MaxWeight << "kg";
        // 补齐空格
        int len = to_string((int)GetTotalWeight()).length()
                + to_string((int)MaxWeight).length();
        for (int s = len; s < 12; s++) cout << " ";
        cout << "║" << endl;
        cout << "╚══════════════════════════════════════╝" << endl;
    }
};


// ============================================================
// 第三部分：主函数 — 测试背包系统
// ============================================================

int main()
{
    cout << "===== UE5 C++ 教学 — 第2章背包系统案例 =====" << endl;
    cout << endl;

    // ----- 3.1 创建背包对象 -----
    CInventory Backpack(100.0f);  // 最大负重100kg
    cout << "创建了一个背包（最大负重100kg）" << endl;

    // ----- 3.2 创建一些物品 -----
    FItem HealthPotion("生命药水", "恢复50点生命值", EItemRarity::Common, 0.5f, 10);
    //                ─────┬─────  ───────┬───────  ──────────┬──────────  ──┬── ──┬─  ─┬─
    //                  名称          描述              稀有度=普通     重量0.5  堆叠上限  当前数量=1

    FItem ManaPotion("法力药水", "恢复30点法力值", EItemRarity::Common, 0.3f, 10);
    FItem IronSword("铁剑", "一把普通的铁剑", EItemRarity::Uncommon, 5.0f, 1);
    //                                                                     ──┬──  ─┬─
    //                                                                  重量5kg  最大堆叠1（武器不能堆叠）

    FItem DragonScale("龙鳞", "传说巨龙的鳞片", EItemRarity::Legendary, 2.0f, 99);
    FItem MagicRing("魔法戒指", "蕴含魔力的戒指", EItemRarity::Rare, 0.1f, 1);
    FItem HeavyArmor("重甲", "沉重的板甲", EItemRarity::Epic, 45.0f, 1);

    // ----- 3.3 测试添加物品 -----
    cout << endl << "=== 测试1：添加物品 ===" << endl;
    Backpack.AddItem(HealthPotion);    // 添加生命药水
    Backpack.AddItem(HealthPotion);    // 再添加一瓶（应该堆叠到2瓶）
    Backpack.AddItem(HealthPotion);    // 第三瓶（堆叠到3瓶）
    Backpack.AddItem(ManaPotion);      // 法力药水
    Backpack.AddItem(IronSword);       // 铁剑
    Backpack.AddItem(DragonScale);     // 龙鳞
    Backpack.AddItem(MagicRing);       // 魔法戒指

    Backpack.Display();  // 显示当前背包

    // ----- 3.4 测试查找物品 -----
    cout << endl << "=== 测试2：查找物品 ===" << endl;

    FItem* Found = Backpack.FindItem("龙鳞");  // 用指针接收查找结果
    if (Found != nullptr)  // 空指针检查（重要！）
    {
        cout << "找到了！物品：" << Found->Name  // -> 访问指针成员
             << "，稀有度：" << RarityToString(Found->Rarity)
             << "，数量：" << Found->CurrentStack << endl;
    }
    else
    {
        cout << "没找到龙鳞（？不应该啊）" << endl;
    }

    // 测试找不存在的物品
    FItem* NotFound = Backpack.FindItem("神剑");
    if (NotFound == nullptr)
    {
        cout << "找不到神剑（正常，我们没添加过）" << endl;
    }

    // ----- 3.5 测试超重保护 -----
    cout << endl << "=== 测试3：超重保护 ===" << endl;
    cout << "尝试添加重甲（45kg）... ";
    bool bResult = Backpack.AddItem(HeavyArmor);
    if (!bResult)
    {
        cout << "超重保护生效！重甲无法加入背包" << endl;
    }

    // ----- 3.6 测试删除物品 -----
    cout << endl << "=== 测试4：删除物品 ===" << endl;
    Backpack.RemoveItem("生命药水", 2);  // 删2瓶（还剩1瓶）
    Backpack.RemoveItem("铁剑");          // 默认删1个（铁剑最大堆叠=1，直接移除）
    Backpack.RemoveItem("不存在的物品");   // 测试删除不存在的物品

    // ----- 3.7 最终显示 -----
    cout << endl << "=== 最终背包状态 ===" << endl;
    Backpack.Display();

    // ----- 3.8 统计信息 -----
    cout << endl << "=== 统计 ===" << endl;
    cout << "背包中共有 " << Backpack.GetItemCount() << " 种不同的物品" << endl;
    cout << "总负重: " << Backpack.GetTotalWeight() << "kg" << endl;

    // ----- 3.9 测试while循环：一直添加直到背包满 -----
    cout << endl << "=== 测试5：填满背包 ===" << endl;

    FItem Log("原木", "普通的木材", EItemRarity::Common, 8.0f, 5);

    int SuccessCount = 0;
    while (Backpack.AddItem(Log))  // 只要还能添加就继续
    {
        SuccessCount++;
    }
    cout << "成功添加了 " << SuccessCount << " 组原木后背包满了" << endl;

    Backpack.Display();

    cout << endl << "===== 测试结束 =====" << endl;
    return 0;
}
```
