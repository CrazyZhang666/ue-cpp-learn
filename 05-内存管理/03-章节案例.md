# 第5章案例：计时器管理系统

> **难度**：⭐⭐⭐（中级实战级）
> **预计时间**：60分钟
> **目标**：综合练习TSharedPtr/TUniquePtr/TWeakPtr的完整用法，同时巩固UPROPERTY和GC知识。实现一个可在UE中实际使用的计时器管理器。

---

## 案例目标

创建一个完整的计时器管理系统：

- `FTimerData` 结构体：存储一个计时器的所有信息
- `FTimerManager` 类：管理多个计时器的生命周期
- 使用 **TSharedPtr** 共享计时器数据
- 使用 **TUniquePtr** 管理管理器内部的唯一数据
- 使用 **TWeakPtr** 防止循环引用
- 在 **Actor** 的Tick中驱动计时器
- 演示智能指针的**引用计数行为**
- 正确处理 **GC**（UPROPERTY保护）

## 第一步：创建计时器数据结构体

创建 `TimerData.h`：

```cpp
// TimerData.h
#pragma once

#include "CoreMinimal.h"

// 前置声明：我们会在另一个文件中定义FTimerManager
// 前置声明让编译器知道这个名字存在，但不需要完整定义
// 这样就可以用 TWeakPtr<FTimerManager> 而无需包含完整头文件
class FTimerManager;

// ===== 计时器状态的枚举 =====
// 虽然不是必需，但可以让状态管理更清晰
enum class ETimerState : uint8
{
    Running,     // 正常运行中
    Paused,      // 暂停中（保持剩余时间不变）
    Completed,   // 已完成（时间到了）
    Cancelled    // 被取消（手动停止）
};

// ===== 计时器数据：单个计时器的所有信息 =====
// 这是一个纯C++结构体，不需要USTRUCT
// 原因：不需要编辑器显示、蓝图访问、网络同步
// 它是纯逻辑数据，用智能指针管理最合适
struct FTimerData
{
    // ──────── 标识信息 ────────
    // 计时器的唯一名称（用于查找和管理）
    FName TimerName;

    // 计时器当前状态
    ETimerState State;

    // ──────── 时间数据 ────────
    // 计时器的总时长（秒），例如 5.0f = 5秒倒计时
    float TotalDuration;

    // 剩余时间（秒）。每次Tick时减少这个值
    // 当 RemainingTime <= 0 时，计时器完成
    float RemainingTime;

    // 已经过的时间（秒）。从0开始累加
    float ElapsedTime;

    // ──────── 行为控制 ────────
    // 是否暂停。暂停时RemainingTime不减少
    bool bIsPaused;

    // 是否循环。循环计时器完成后会重置并重新开始
    bool bIsLooping;

    // 是否已完成。用于查询计时器是否到期
    bool bIsCompleted;

    // ──────── 循环相关 ────────
    // 当前已循环的次数（仅对循环计时器有意义）
    int32 CurrentLoopCount;

    // 最大循环次数。设为 -1 表示无限循环
    int32 MaxLoopCount;

    // ──────── 所有者引用 ────────
    // ⚠️ 关键设计：使用 TWeakPtr 而非 TSharedPtr！
    // 如果这里用 TSharedPtr<FTimerManager> → 循环引用：
    //   FTimerManager → TSharedPtr → FTimerData
    //        ↑                            │
    //        └────── TSharedPtr ──────────┘
    //   循环引用 → 内存泄漏！
    //
    // 用 TWeakPtr 打破循环引用：
    //   FTimerManager → TSharedPtr → FTimerData
    //        ↑                            │
    //        └────── TWeakPtr ───────────┘ (不增加引用计数)
    TWeakPtr<FTimerManager> Owner;

    // ──────── 构造函数 ────────
    FTimerData()
        : TimerName(NAME_None)
        , State(ETimerState::Running)
        , TotalDuration(0.0f)
        , RemainingTime(0.0f)
        , ElapsedTime(0.0f)
        , bIsPaused(false)
        , bIsLooping(false)
        , bIsCompleted(false)
        , CurrentLoopCount(0)
        , MaxLoopCount(-1)  // -1 表示无限循环
    {
    }

    // ──────── 便捷函数 ────────

    // 获取完成进度（0.0 ~ 1.0），用于UI进度条
    float GetProgress() const
    {
        // 避免除以0
        if (TotalDuration <= 0.0f)
        {
            return 1.0f;  // 没有时长 = 立即完成
        }

        // 用已过时间除以总时长
        // ElapsedTime / TotalDuration = 完成百分比
        return FMath::Clamp(ElapsedTime / TotalDuration, 0.0f, 1.0f);
        //                           FMath::Clamp = 限制范围在0~1之间
    }

    // 获取剩余时间百分比（1.0 ~ 0.0）
    float GetRemainingPercent() const
    {
        return 1.0f - GetProgress();
    }

    // 获取格式化的时间字符串（MM:SS / HH:MM:SS）
    FString GetFormattedTime() const
    {
        // 将剩余的秒数转换为 分:秒 格式
        int32 TotalSeconds = FMath::CeilToInt(RemainingTime);  // 向上取整
        int32 Minutes = TotalSeconds / 60;                     // 分钟数
        int32 Seconds = TotalSeconds % 60;                     // 秒数

        if (Minutes > 0)
        {
            return FString::Printf(TEXT("%d:%02d"), Minutes, Seconds);
            // %d = 整数，%02d = 补零到2位的整数（0→00, 5→05, 12→12）
        }
        else
        {
            return FString::Printf(TEXT("%d秒"), Seconds);
        }
    }

    // 重置计时器（回到初始状态）
    void Reset()
    {
        RemainingTime = TotalDuration;  // 剩余时间恢复为原始总时长
        ElapsedTime = 0.0f;             // 经过时间归零
        bIsCompleted = false;           // 未完成
        bIsPaused = false;              // 取消暂停（默认开始运行）
        State = ETimerState::Running;   // 状态设为运行中
        CurrentLoopCount = 0;           // 循环计数归零
    }

    // 打印调试信息
    void PrintDebugInfo() const
    {
        UE_LOG(LogTemp, Warning, TEXT("═══ 计时器: %s ═══"), *TimerName.ToString());
        UE_LOG(LogTemp, Warning, TEXT("  状态: %s"), *UEnumToString(State));
        UE_LOG(LogTemp, Warning, TEXT("  总时长: %.1f秒"), TotalDuration);
        UE_LOG(LogTemp, Warning, TEXT("  剩余: %.1f秒 (%.0f%%)"), RemainingTime,
               GetRemainingPercent() * 100.0f);
        UE_LOG(LogTemp, Warning, TEXT("  已过: %.1f秒"), ElapsedTime);
        UE_LOG(LogTemp, Warning, TEXT("  进度: %s"), *GetFormattedTime());
        UE_LOG(LogTemp, Warning, TEXT("  暂停: %s"), bIsPaused ? TEXT("是") : TEXT("否"));
        UE_LOG(LogTemp, Warning, TEXT("  循环: %s (%d/%d)"),
               bIsLooping ? TEXT("是") : TEXT("否"), CurrentLoopCount,
               MaxLoopCount);
    }

private:
    // 辅助函数：枚举转字符串
    static FString UEnumToString(ETimerState InState)
    {
        switch (InState)
        {
            case ETimerState::Running:   return TEXT("运行中");
            case ETimerState::Paused:    return TEXT("已暂停");
            case ETimerState::Completed: return TEXT("已完成");
            case ETimerState::Cancelled: return TEXT("已取消");
            default:                     return TEXT("未知");
        }
    }
};
```

### 逐行理解关键点

```cpp
// ===== TWeakPtr<FTimerManager> Owner 的设计原因 =====
//
// 计时器持有Owner的指针是为了：
// 1. 完成时能通知管理者
// 2. 知道谁在管理自己
//
// 但为什么用TWeakPtr？
// ❌ 如果：Owner 是 TSharedPtr<FTimerManager>
//    → FTimerManager持有TSharedPtr<FTimerData>（通过TArray）
//    → FTimerData持有TSharedPtr<FTimerManager>
//    → 互相持有 → 引用计数永不归零 → 内存泄漏！
//
// ✅ 正确：Owner 用 TWeakPtr<FTimerManager>
//    → FTimerManager持有TSharedPtr<FTimerData>
//    → FTimerData持有TWeakPtr<FTimerManager>（弱引用不增加计数）
//    → Manager取消时，弱引用自动失效
//    → 没有循环引用 → 内存正常释放
```

## 第二步：创建计时器管理器

创建 `TimerManager.h`：

```cpp
// TimerManager.h
#pragma once

#include "CoreMinimal.h"
#include "TimerData.h"  // 使用我们的FTimerData

// ===== 计时器管理器：管理所有计时器的生命周期 =====
//
// 这是一个纯C++类（非UObject），所以：
// - 不需要UCLASS标记
// - 不需要GENERATED_BODY
// - 使用TSharedPtr管理自身和内部数据
// - 使用TUniquePtr管理独有内部状态
//
// 设计思路：
// - FTimerManager 本身由调用者通过 TSharedPtr 持有
// - 内部计时器列表用 TArray<TSharedPtr<FTimerData>>
//   （多个地方可能需要引用同一个计时器的数据）
// - 内部独有数据（如全局设置）用 TUniquePtr

class FTimerManager
{
public:
    // ──────── 构造函数 ────────
    FTimerManager()
        : bIsInitialized(false)
    {
        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 构造函数被调用"));
        // 此时还没初始化，InternalData尚未创建
    }

    // ──────── 初始化 ────────
    void Initialize()
    {
        if (bIsInitialized)
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 已经初始化过了，跳过"));
            return;
        }

        // 创建内部独有的数据结构（TUniquePtr 独占所有权）
        // InternalData 只属于这个 Manager，其他人不能共享
        InternalData = MakeUnique<FManagerInternalData>();

        bIsInitialized = true;

        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 初始化完成"));
    }

    // ──────── 关闭（清理所有计时器）────────
    void Shutdown()
    {
        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 开始关闭..."));

        // 清理所有计时器
        // Reset() 会减少每个TSharedPtr的引用计数
        // 如果某个FTimerData没有其他地方持有它 → 自动删除
        int32 RemovedCount = Timers.Num();  // 记录清除了多少个
        Timers.Empty();  // 清空数组，释放所有TSharedPtr

        // 销毁内部数据
        InternalData.Reset();  // TUniquePtr.Reset() = 立即销毁

        bIsInitialized = false;

        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 关闭完成，清除了 %d 个计时器"),
               RemovedCount);
    }

    // ──────── 添加计时器 ────────
    // 返回 TSharedPtr<FTimerData> 让调用者可以持有引用
    // 例如：调用者想在UI上显示这个计时器的进度
    TSharedPtr<FTimerData> AddTimer(
        FName TimerName,      // 计时器名称（用于查找）
        float Duration,       // 总时长（秒）
        bool bLoop = false,   // 是否循环（默认不循环）
        int32 MaxLoops = -1   // 最大循环次数（默认无限）
    )
    {
        if (!bIsInitialized)
        {
            UE_LOG(LogTemp, Error, TEXT("[TimerManager] 未初始化！请先调用 Initialize()"));
            return nullptr;  // 未初始化，返回空
        }

        // 检查同名计时器是否已存在
        if (FindTimerIndex(TimerName) != INDEX_NONE)
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 计时器 '%s' 已存在！"),
                   *TimerName.ToString());
            return nullptr;  // 同名计时器已存在，返回空
        }

        // ===== 创建新的计时器数据 =====
        // MakeShared = 在堆上分配 FTimerData 并返回 TSharedPtr
        TSharedPtr<FTimerData> NewTimer = MakeShared<FTimerData>();

        // 填充数据
        NewTimer->TimerName = TimerName;                     // 名称
        NewTimer->TotalDuration = Duration;                  // 总时长
        NewTimer->RemainingTime = Duration;                  // 初始剩余时间 = 总时长
        NewTimer->ElapsedTime = 0.0f;                        // 已经过时间为0
        NewTimer->bIsLooping = bLoop;                        // 是否循环
        NewTimer->MaxLoopCount = MaxLoops;                   // 最大循环次数
        NewTimer->bIsCompleted = false;                      // 未完成
        NewTimer->bIsPaused = false;                         // 未暂停
        NewTimer->State = ETimerState::Running;              // 运行中
        NewTimer->CurrentLoopCount = 0;                      // 循环次数为0

        // ═══════════════════════════════════════════
        // 关键：设置Owner弱引用
        // 必须传入一个 TSharedPtr<FTimerManager> 来创建 TWeakPtr
        // 但FTimerData的Owner是TWeakPtr，FTimerManager没办法直接传this
        // 解决方案：在外部设置Owner（见后面的Actor代码）
        // 这里先不设置Owner，让调用者通过 SetOwnerForTimer 设置
        // ═══════════════════════════════════════════

        // 添加到管理列表
        Timers.Add(NewTimer);
        // Add之后：NewTimer的引用计数变为2（NewTimer局部变量 + Timers数组）
        // 函数返回后NewTimer局部变量销毁 → 引用计数变为1（只有Timers数组）

        UE_LOG(LogTemp, Warning,
               TEXT("[TimerManager] 添加计时器: '%s', 时长=%.1f秒, 循环=%s, 当前总数=%d"),
               *TimerName.ToString(), Duration,
               bLoop ? TEXT("是") : TEXT("否"),
               Timers.Num());

        UE_LOG(LogTemp, Warning,
               TEXT("[TimerManager] 新计时器引用计数 = %d"),
               NewTimer.GetSharedReferenceCount());
        // 这会显示 2（NewTimer局部变量 + Timers数组各持有1个引用）
        // 返回后NewTimer销毁，引用计数变为1

        return NewTimer;  // 返回TSharedPtr的拷贝，调用者获得一个引用
    }

    // 为计时器设置所有者弱引用
    // 这需要在FTimerManager外部调用，因为TSharedPtr的this在内部无法直接获取
    void SetOwnerForTimer(TSharedPtr<FTimerData> Timer,
                          TSharedPtr<FTimerManager> ManagerSharedPtr)
    {
        if (!Timer.IsValid())
        {
            return;
        }

        // 从TSharedPtr创建TWeakPtr（不增加引用计数）
        Timer->Owner = ManagerSharedPtr;

        UE_LOG(LogTemp, Warning,
               TEXT("[TimerManager] 设置计时器 '%s' 的Owner弱引用"),
               *Timer->TimerName.ToString());
    }

    // ──────── 停止（取消）计时器 ────────
    bool CancelTimer(FName TimerName)
    {
        int32 Index = FindTimerIndex(TimerName);
        if (Index == INDEX_NONE)
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 找不到计时器 '%s'，无法取消"),
                   *TimerName.ToString());
            return false;
        }

        // 标记为已取消
        TSharedPtr<FTimerData> Timer = Timers[Index];
        Timer->State = ETimerState::Cancelled;
        Timer->bIsCompleted = true;  // 标记为完成，让查询者知道

        // 从数组中移除
        // RemoveAt 会释放数组对该TSharedPtr的引用
        // 如果有其他地方还在持有这个计时器的TSharedPtr，对象不会被删除
        Timers.RemoveAt(Index);

        UE_LOG(LogTemp, Warning,
               TEXT("[TimerManager] 取消计时器: '%s', 剩余引用计数=%d"),
               *TimerName.ToString(),
               Timer.GetSharedReferenceCount());
        // 引用计数 > 0 说明还有其他持有者

        return true;
    }

    // ──────── 暂停计时器 ────────
    bool PauseTimer(FName TimerName)
    {
        TSharedPtr<FTimerData> Timer = FindTimer(TimerName);
        if (!Timer.IsValid())
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 找不到计时器 '%s'，无法暂停"),
                   *TimerName.ToString());
            return false;
        }

        if (Timer->bIsPaused)
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 计时器 '%s' 已经是暂停状态"),
                   *TimerName.ToString());
            return false;  // 已经暂停了
        }

        Timer->bIsPaused = true;
        Timer->State = ETimerState::Paused;

        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 暂停计时器: '%s' (剩余 %.1f 秒)"),
               *TimerName.ToString(), Timer->RemainingTime);

        return true;
    }

    // ──────── 恢复计时器 ────────
    bool ResumeTimer(FName TimerName)
    {
        TSharedPtr<FTimerData> Timer = FindTimer(TimerName);
        if (!Timer.IsValid())
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 找不到计时器 '%s'，无法恢复"),
                   *TimerName.ToString());
            return false;
        }

        if (!Timer->bIsPaused)
        {
            UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 计时器 '%s' 没有在暂停状态"),
                   *TimerName.ToString());
            return false;  // 没有暂停
        }

        Timer->bIsPaused = false;
        Timer->State = ETimerState::Running;

        UE_LOG(LogTemp, Warning, TEXT("[TimerManager] 恢复计时器: '%s'"),
               *TimerName.ToString());

        return true;
    }

    // ──────── 查询计时器 ────────
    // 返回 TSharedPtr<FTimerData>
    // 调用者可以持有这个引用，不会因为Manager内部操作而失效
    TSharedPtr<FTimerData> FindTimer(FName TimerName) const
    {
        for (const TSharedPtr<FTimerData>& Timer : Timers)
        {
            if (Timer.IsValid() && Timer->TimerName == TimerName)
            {
                return Timer;  // 返回拷贝，引用计数+1
            }
        }
        return nullptr;  // 未找到
    }

    // 检查计时器是否存在且还在运行
    bool IsTimerActive(FName TimerName) const
    {
        TSharedPtr<FTimerData> Timer = FindTimer(TimerName);
        // IsValid：检查TSharedPtr不为空
        // 检查计时器没有完成且没有被取消
        return Timer.IsValid()
               && !Timer->bIsCompleted
               && Timer->State != ETimerState::Cancelled;
    }

    // ──────── 获取所有计时器 ────────
    // 返回const引用（不增加引用计数，不拷贝数组）
    const TArray<TSharedPtr<FTimerData>>& GetAllTimers() const
    {
        return Timers;
    }

    // ──────── 每帧更新（核心逻辑）────────
    // 这个方法需要在某处被每帧调用（在Actor::Tick中）
    void Tick(float DeltaTime)
    {
        // DeltaTime = 上一帧到这一帧经过的时间（秒）
        // 例如：60 FPS → DeltaTime ≈ 0.016s
        //       30 FPS → DeltaTime ≈ 0.033s

        if (!bIsInitialized)
        {
            return;  // 未初始化，不Tick
        }

        // ═══════════════════════════════════════════
        // ⚠️ 重要：安全的遍历方式
        // 我们不能在遍历过程中修改数组
        // 所以遍历期间只收集"需要处理的计时器"
        // 遍历完后统一处理
        // ═══════════════════════════════════════════

        // 收集需要处理的计时器（不能直接在遍历中修改数组）
        TArray<FName> CompletedTimers;  // 需要回收的计时器名称列表

        for (TSharedPtr<FTimerData>& Timer : Timers)
        {
            // ⚠️ 安全检查：确保计时器数据有效
            if (!Timer.IsValid())
            {
                continue;  // 跳过无效的计时器
            }

            // 跳过暂停的计时器（不减少剩余时间）
            if (Timer->bIsPaused)
            {
                continue;  // 暂停状态：不更新
            }

            // 跳过已完成的计时器（不需要再更新）
            if (Timer->bIsCompleted)
            {
                continue;
            }

            // ──── 更新计时器时间 ────
            // 减少剩余时间
            Timer->RemainingTime -= DeltaTime;
            // 增加已经过时间
            Timer->ElapsedTime += DeltaTime;

            // ──── 检查是否时间到了 ────
            if (Timer->RemainingTime <= 0.0f)
            {
                // 时间到了！

                if (Timer->bIsLooping)
                {
                    // ──── 循环计时器：重置并继续 ────
                    Timer->CurrentLoopCount++;  // 循环次数+1

                    // 检查是否达到了最大循环次数
                    if (Timer->MaxLoopCount > 0  // 大于0表示有限循环
                        && Timer->CurrentLoopCount >= Timer->MaxLoopCount)
                    {
                        // 达到最大循环次数，完成
                        Timer->bIsCompleted = true;
                        Timer->State = ETimerState::Completed;
                        CompletedTimers.Add(Timer->TimerName);

                        UE_LOG(LogTemp, Warning,
                               TEXT("[TimerManager] 循环计时器 '%s' 完成（循环了 %d 次）"),
                               *Timer->TimerName.ToString(),
                               Timer->CurrentLoopCount);
                    }
                    else
                    {
                        // 重置剩余时间，继续下一次循环
                        // 注意：RemainingTime是负数（多扣了一些时间）
                        // 把它加回TotalDuration来"重置"，保留超出部分
                        Timer->RemainingTime += Timer->TotalDuration;

                        UE_LOG(LogTemp, Warning,
                               TEXT("[TimerManager] 循环计时器 '%s' 完成第 %d 次循环"),
                               *Timer->TimerName.ToString(),
                               Timer->CurrentLoopCount);
                    }
                }
                else
                {
                    // ──── 不循环的计时器：直接完成 ────
                    Timer->bIsCompleted = true;
                    Timer->State = ETimerState::Completed;
                    CompletedTimers.Add(Timer->TimerName);

                    UE_LOG(LogTemp, Warning,
                           TEXT("[TimerManager] 计时器 '%s' 时间到！"),
                           *Timer->TimerName.ToString());
                }
            }
        }

        // ──── 清理已完成的计时器 ────
        for (const FName& TimerName : CompletedTimers)
        {
            // 从数组中移除（释放Manager的引用）
            // 注意：如果外部还持有这个计时器的TSharedPtr，对象不会被删除
            RemoveTimerFromArray(TimerName);
        }

        // ──── 更新InternalData的总运行时间 ────
        if (InternalData.IsValid())
        {
            InternalData->TotalTickTime += DeltaTime;
        }
    }

    // ──────── 打印所有计时器状态 ────────
    void PrintAllTimers() const
    {
        UE_LOG(LogTemp, Warning, TEXT("══════════════════════════════"));
        UE_LOG(LogTemp, Warning, TEXT("  计时器管理器状态报告"));
        UE_LOG(LogTemp, Warning, TEXT("  活跃计时器数: %d"), Timers.Num());
        UE_LOG(LogTemp, Warning, TEXT("══════════════════════════════"));

        if (Timers.Num() == 0)
        {
            UE_LOG(LogTemp, Warning, TEXT("  (无活跃计时器)"));
            return;
        }

        for (const TSharedPtr<FTimerData>& Timer : Timers)
        {
            if (Timer.IsValid())
            {
                Timer->PrintDebugInfo();
                // 打印引用计数（了解还有谁持有这个计时器）
                UE_LOG(LogTemp, Warning, TEXT("  引用计数: %d"),
                       Timer.GetSharedReferenceCount());
            }
        }

        UE_LOG(LogTemp, Warning, TEXT("══════════════════════════════"));
    }

private:
    // ──────── 内部数据结构 ────────
    // TUniquePtr：独占所有权，只有FTimerManager能访问
    struct FManagerInternalData
    {
        // 管理器从初始化到现在总共运行的时间
        float TotalTickTime;

        // 总共创建过的计时器数量
        int32 TotalTimersCreated;

        FManagerInternalData()
            : TotalTickTime(0.0f)
            , TotalTimersCreated(0)
        {
        }
    };

    // ═══════════════════════════════════════
    // 成员变量
    // ═══════════════════════════════════════

    // 是否已初始化
    bool bIsInitialized;

    // 内部独有数据（运行时状态，不对外共享）
    // TUniquePtr = 独占所有权
    // 只有FTimerManager能访问和修改
    TUniquePtr<FManagerInternalData> InternalData;

    // 计时器列表
    // TSharedPtr = 共享所有权
    // Manager持有引用，外部也可以持有引用
    // 只要还有任何TSharedPtr指向某个FTimerData，它就不会被删除
    TArray<TSharedPtr<FTimerData>> Timers;

    // ──────── 内部辅助函数 ────────

    // 找到计时器在数组中的索引
    // INDEX_NONE = -1，表示未找到
    int32 FindTimerIndex(FName TimerName) const
    {
        for (int32 i = 0; i < Timers.Num(); i++)
        {
            if (Timers[i].IsValid() && Timers[i]->TimerName == TimerName)
            {
                return i;  // 返回索引
            }
        }
        return INDEX_NONE;  // 未找到 = -1
    }

    // 从数组中移除计时器（不检查状态）
    void RemoveTimerFromArray(FName TimerName)
    {
        int32 Index = FindTimerIndex(TimerName);
        if (Index != INDEX_NONE)
        {
            // RemoveAt：从数组中移除元素
            // 这会释放数组中该位置的TSharedPtr → 引用计数-1
            Timers.RemoveAt(Index);
        }
    }
};
```

```cpp
// MyTimerActor.cpp
#include "MyTimerActor.h"

// ===== 构造函数 =====
AMyTimerActor::AMyTimerActor()
{
    // 开启Tick——我们需要每帧驱动计时器
    PrimaryActorTick.bCanEverTick = true;
    // 设置Tick间隔为0（每帧都Tick，不跳帧）
    PrimaryActorTick.TickInterval = 0.0f;

    // 初始化演示变量
    DemoAccumulatedTime = 0.0f;

    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] 构造函数"));
}

// ===== BeginPlay：游戏开始时调用 =====
void AMyTimerActor::BeginPlay()
{
    Super::BeginPlay();

    UE_LOG(LogTemp, Warning, TEXT("════════════════════════════════"));
    UE_LOG(LogTemp, Warning, TEXT("  [MyTimerActor] BeginPlay 开始"));
    UE_LOG(LogTemp, Warning, TEXT("════════════════════════════════"));

    // ═══════════════════════════════════════
    // 步骤1：创建TSharedPtr管理计时器管理器
    // ═══════════════════════════════════════
    // MakeShared<FTimerManager>() = 在堆上创建FTimerManager对象
    // 返回TSharedPtr，引用计数初始为1（只有TimerManager持有）
    TimerManager = MakeShared<FTimerManager>();

    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] 创建TimerManager后引用计数 = %d"),
           TimerManager.GetSharedReferenceCount());
    // 输出：引用计数 = 1（只有Actor持有）

    // ═══════════════════════════════════════
    // 步骤2：初始化管理器
    // ═══════════════════════════════════════
    TimerManager->Initialize();

    // ═══════════════════════════════════════
    // 步骤3：创建几个演示计时器
    // ═══════════════════════════════════════

    // 演示1：普通计时器（3秒后到期，不循环）
    UE_LOG(LogTemp, Warning, TEXT(""));
    UE_LOG(LogTemp, Warning, TEXT("--- 创建演示计时器 ---"));

    TSharedPtr<FTimerData> Timer1 = TimerManager->AddTimer(
        TEXT("Skill_Cooldown"),  // 名称：技能冷却
        3.0f,                    // 3秒后到期
        false                    // 不循环
    );
    // 设置Owner弱引用（用于计时器完成时的通知等功能）
    TimerManager->SetOwnerForTimer(Timer1, TimerManager);

    // 演示2：循环计时器（每隔2秒触发，总共3次）
    TSharedPtr<FTimerData> Timer2 = TimerManager->AddTimer(
        TEXT("Health_Regen"),    // 名称：生命恢复
        2.0f,                    // 2秒一个周期
        true,                    // 循环
        3                        // 最多循环3次
    );
    TimerManager->SetOwnerForTimer(Timer2, TimerManager);

    // 演示3：无限循环计时器（每5秒触发，无限循环）
    TSharedPtr<FTimerData> Timer3 = TimerManager->AddTimer(
        TEXT("Periodic_Check"),  // 名称：定期检查
        5.0f,                    // 5秒一个周期
        true,                    // 循环
        -1                       // -1 = 无限循环
    );
    TimerManager->SetOwnerForTimer(Timer3, TimerManager);

    // ═══════════════════════════════════════
    // 演示：引用计数测试
    // ═══════════════════════════════════════
    // 保存一个外部引用（ExternalTimerRef持有Timer1）
    ExternalTimerRef = Timer1;
    // 现在Timer1的引用计数：
    //   1（Timers数组） + 1（ExternalTimerRef） = 2
    //   （局部变量Timer1在返回后已销毁）

    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] 外部持有Timer1后引用计数 = %d"),
           ExternalTimerRef.GetSharedReferenceCount());

    // ═══════════════════════════════════════
    // 演示：暂停计时器
    // ═══════════════════════════════════════
    TimerManager->PauseTimer(TEXT("Periodic_Check"));
    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] 暂停了 Periodic_Check 计时器"));

    // 打印所有计时器的初始状态
    TimerManager->PrintAllTimers();

    UE_LOG(LogTemp, Warning, TEXT("════════════════════════════════"));
}

// ===== EndPlay：游戏结束或Actor销毁时调用 =====
void AMyTimerActor::EndPlay(const EEndPlayReason::Type EndPlayReason)
{
    // ═══════════════════════════════════════
    // 演示：释放外部引用
    // 即使TimerManager被销毁，ExternalTimerRef仍持有Timer1
    // ═══════════════════════════════════════
    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] EndPlay - ExternalTimerRef引用计数 = %d"),
           ExternalTimerRef.IsValid()
               ? ExternalTimerRef.GetSharedReferenceCount()
               : 0);

    // 重置外部引用（释放对Timer1的持有）
    ExternalTimerRef.Reset();  // 引用计数-1

    // 关闭计时器管理器
    if (TimerManager.IsValid())
    {
        TimerManager->Shutdown();
    }

    // 释放管理器（最后一步）
    TimerManager.Reset();  // 引用计数-1 → 如果为0，FTimerManager被删除

    Super::EndPlay(EndPlayReason);

    UE_LOG(LogTemp, Warning, TEXT("[MyTimerActor] EndPlay 完成"));
}

// ===== Tick：每帧调用 =====
void AMyTimerActor::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);

    // ═══════════════════════════════════════
    // 核心：驱动计时器管理器
    // 这会让所有活跃的计时器前进 DeltaTime 秒
    // ═══════════════════════════════════════
    if (TimerManager.IsValid())
    {
        TimerManager->Tick(DeltaTime);
    }

    // ═══════════════════════════════════════
    // 演示：每3秒打印一次所有计时器状态
    // ═══════════════════════════════════════
    DemoAccumulatedTime += DeltaTime;
    if (DemoAccumulatedTime >= 3.0f)
    {
        DemoAccumulatedTime = 0.0f;  // 重置累计时间
        if (TimerManager.IsValid())
        {
            UE_LOG(LogTemp, Warning, TEXT("--- [每3秒报告] ---"));
            TimerManager->PrintAllTimers();

            // 特别检查外部引用是否还有效
            if (ExternalTimerRef.IsValid())
            {
                UE_LOG(LogTemp, Warning,
                       TEXT("[MyTimerActor] ExternalTimerRef仍持有Timer1, 引用计数=%d"),
                       ExternalTimerRef.GetSharedReferenceCount());
            }
            else
            {
                UE_LOG(LogTemp, Warning,
                       TEXT("[MyTimerActor] ExternalTimerRef已失效"));
            }
        }
    }
}

bool AMyTimerActor::AddNewTimer(FName TimerName, float Duration, bool bLoop)
{
    if (!TimerManager.IsValid())

        TimerManager->SetOwnerForTimer(NewTimer, TimerManager);
        return true;
    }
    return false;
}

    if (!TimerManager.IsValid())

        return false;

    if (!TimerManager.IsValid())
    {
        return false;
```

## 本章知识点回顾

| 概念 | 在案例中的位置 |
|------|-------------|
| `TSharedPtr` | `FTimerManager` 自身管理、`TArray<TSharedPtr<FTimerData>>` 计时器列表 |
| `TUniquePtr` | `FManagerInternalData` 内部独有数据 |
| `TWeakPtr` | `FTimerData::Owner` 弱引用打破循环引用 |
| `MakeShared` | 创建 `FTimerData` 对象 |
| `MakeUnique` | 创建 `FManagerInternalData` 对象 |
| 引用计数 | 通过 `GetSharedReferenceCount()` 观察 |
| `TWeakPtr::Pin()` | 从弱引用获取临时强引用（在回调等场景中使用） |
| 循环引用破坏 | Owner用TWeakPtr而非TSharedPtr |
| `UPROPERTY` | Actor中不需要UPROPERTY保护TimerManager（非UObject用智能指针） |
