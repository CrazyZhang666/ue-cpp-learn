### 5.3.2 技能栏组件实现

创建文件 `Source/YourProject/Private/GAS/SkillBar/RPGSkillBarComponent.cpp`：

```cpp
// ====================================================================
// 文件：Source/YourProject/Private/GAS/SkillBar/RPGSkillBarComponent.cpp
// 描述：技能栏组件实现
// ====================================================================

#include "GAS/SkillBar/RPGSkillBarComponent.h"
#include "AbilitySystemComponent.h"
#include "GAS/Ability/RPGGameplayAbility.h"

URPGSkillBarComponent::URPGSkillBarComponent()
{
    PrimaryComponentTick.bCanEverTick = false;

    // ================================================================
    // 初始化4个技能槽位的默认InputTag
    // ================================================================
    // 在编辑器中可覆写这些默认值

    DefaultInputTags.Add(FGameplayTag::RequestGameplayTag("Input.Skill.Slot1"));
    DefaultInputTags.Add(FGameplayTag::RequestGameplayTag("Input.Skill.Slot2"));
    DefaultInputTags.Add(FGameplayTag::RequestGameplayTag("Input.Skill.Slot3"));
    DefaultInputTags.Add(FGameplayTag::RequestGameplayTag("Input.Skill.Slot4"));

    // 初始化4个空槽位
    SkillSlots.SetNum(4);
    for (int32 i = 0; i < 4; ++i)
    {
        SkillSlots[i].SlotIndex = i;
        SkillSlots[i].InputTag = DefaultInputTags[i];
        SkillSlots[i].bIsUnlocked = true;
    }
}

void URPGSkillBarComponent::InitializeSkillBar(UAbilitySystemComponent* InASC)
{
    ASC = InASC;

    if (!IsValid(ASC))
    {
        UE_LOG(LogTemp, Error, TEXT("SkillBarComponent: Invalid ASC"));
        return;
    }

    // ================================================================
    // 授予所有槽位中已配置的技能到ASC
    // ================================================================

    for (int32 i = 0; i < SkillSlots.Num(); ++i)
    {
        FRPGSkillSlot& Slot = SkillSlots[i];

        if (Slot.IsAvailable())
        {
            // 授予技能
            FGameplayAbilitySpec AbilitySpec(Slot.AbilityClass, 1, -1, GetOwner());
            Slot.AbilityHandle = ASC->GiveAbility(AbilitySpec);

            UE_LOG(LogTemp, Log, TEXT("SkillBar: Slot %d — Granted %s"),
                i, *GetNameSafe(Slot.AbilityClass));
        }
    }
}

void URPGSkillBarComponent::AssignSkillToSlot(
    int32 SlotIndex, TSubclassOf<UGameplayAbility> AbilityClass)
{
    // ================================================================
    // 参数验证
    // ================================================================
    if (SlotIndex < 0 || SlotIndex >= SkillSlots.Num())
    {
        UE_LOG(LogTemp, Error, TEXT("SkillBar: Invalid slot index %d"), SlotIndex);
        return;
    }

    if (!AbilityClass)
    {
        UE_LOG(LogTemp, Warning, TEXT("SkillBar: Cannot assign null ability to slot %d"), SlotIndex);
        return;
    }

    // ================================================================
    // 先移除旧技能
    // ================================================================
    RemoveSkillFromSlot(SlotIndex);

    // ================================================================
    // 授予新技能到ASC
    // ================================================================
    FRPGSkillSlot& Slot = SkillSlots[SlotIndex];
    Slot.AbilityClass = AbilityClass;

    if (IsValid(ASC))
    {
        FGameplayAbilitySpec AbilitySpec(AbilityClass, 1, -1, GetOwner());
        Slot.AbilityHandle = ASC->GiveAbility(AbilitySpec);
    }

    // ================================================================
    // 通知UI更新
    // ================================================================
    OnSkillSlotChanged.Broadcast(SlotIndex, Slot);

    UE_LOG(LogTemp, Log, TEXT("SkillBar: Assigned %s to slot %d"),
        *AbilityClass->GetName(), SlotIndex);
}

void URPGSkillBarComponent::RemoveSkillFromSlot(int32 SlotIndex)
{
    if (SlotIndex < 0 || SlotIndex >= SkillSlots.Num())
    {
        return;
    }

    FRPGSkillSlot& Slot = SkillSlots[SlotIndex];

    // 移除旧技能
    if (Slot.AbilityHandle.IsValid() && IsValid(ASC))
    {
        ASC->ClearAbility(Slot.AbilityHandle);
    }

    // 清空槽位
    Slot.AbilityClass = nullptr;
    Slot.AbilityHandle = FGameplayAbilitySpecHandle();

    OnSkillSlotChanged.Broadcast(SlotIndex, Slot);
}

const FRPGSkillSlot* URPGSkillBarComponent::GetSlotData(int32 SlotIndex) const
{
    if (SlotIndex >= 0 && SlotIndex < SkillSlots.Num())
    {
        return &SkillSlots[SlotIndex];
    }
    return nullptr;
}

float URPGSkillBarComponent::GetSlotCooldownRemaining(int32 SlotIndex) const
{
    // 查询指定槽位的冷却剩余时间
    // 通过ASC的活跃GE查询

    const FRPGSkillSlot* Slot = GetSlotData(SlotIndex);
    if (!Slot || !Slot->IsAvailable() || !IsValid(ASC))
    {
        return 0.0f;
    }

    // 获取技能的默认对象
    UGameplayAbility* AbilityCDO = Slot->AbilityClass->GetDefaultObject<UGameplayAbility>();
    URPGGameplayAbility* RPGAbility = Cast<URPGGameplayAbility>(AbilityCDO);

    if (!RPGAbility || RPGAbility->CooldownTags.IsEmpty())
    {
        return 0.0f;  // 技能没有冷却
    }

    // 查询冷却标签
    if (ASC->HasAnyMatchingGameplayTags(RPGAbility->CooldownTags))
    {
        // 获取具体的剩余时间
        FGameplayEffectQuery Query = FGameplayEffectQuery::MakeQuery_MatchAnyOwningTags(
            RPGAbility->CooldownTags);

        TArray<FActiveGameplayEffectHandle> ActiveEffects;
        ASC->GetActiveEffects(Query, ActiveEffects);

        float MaxRemaining = 0.0f;
        for (const FActiveGameplayEffectHandle& Handle : ActiveEffects)
        {
            float Remaining = 0.0f, Duration = 0.0f;
            if (ASC->GetGameplayEffectRemainingDuration(Handle, Remaining, Duration))
            {
                MaxRemaining = FMath::Max(MaxRemaining, Remaining);
            }
        }
        return MaxRemaining;
    }

    return 0.0f;  // 不在冷却中
}

float URPGSkillBarComponent::GetSlotCooldownProgress(int32 SlotIndex) const
{
    const FRPGSkillSlot* Slot = GetSlotData(SlotIndex);
    if (!Slot || !Slot->IsAvailable())
    {
        return 0.0f;
    }

    UGameplayAbility* AbilityCDO = Slot->AbilityClass->GetDefaultObject<UGameplayAbility>();
    URPGGameplayAbility* RPGAbility = Cast<URPGGameplayAbility>(AbilityCDO);

    if (!RPGAbility)
    {
        return 0.0f;
    }

    float Total = RPGAbility->GetCooldownTotalDuration();
    float Remaining = GetSlotCooldownRemaining(SlotIndex);

    if (Total <= 0.0f)
    {
        return 0.0f;
    }

    return Remaining / Total;
}

bool URPGSkillBarComponent::CanActivateSlot(int32 SlotIndex) const
{
    const FRPGSkillSlot* Slot = GetSlotData(SlotIndex);
    if (!Slot || !Slot->IsAvailable() || !IsValid(ASC))
    {
        return false;
    }

    // 获取技能CDO并检查阻止标签
    UGameplayAbility* AbilityCDO = Slot->AbilityClass->GetDefaultObject<UGameplayAbility>();
    URPGGameplayAbility* RPGAbility = Cast<URPGGameplayAbility>(AbilityCDO);

    if (!RPGAbility)
    {
        return false;
    }

    // 检查阻止标签
    if (ASC->HasAnyMatchingGameplayTags(RPGAbility->ActivationBlockedTags))
    {
        return false;  // 被状态阻止（死亡/眩晕等）
    }

    // 检查冷却
    if (GetSlotCooldownRemaining(SlotIndex) > 0.0f)
    {
        return false;  // 冷却中
    }

    return true;  // ✅ 可以激活
}
```

## 5.4 连击系统

### 5.4.1 连击系统设计

连击系统的工作流程：

```
玩家按下轻攻击（Light）
  → 激活 GA_Combo_Light1（连击第一段）
  → 技能内部设置一个"可连击窗口"（添加 GameplayTag）
  → 在窗口内再次按下轻攻击
      → 激活 GA_Combo_Light2（连击第二段）
      → 继续设置连击窗口
      → 窗口内按下重攻击
          → 激活 GA_Combo_Finisher（终结技，更高伤害！）
  → 如果窗口过期（0.5秒内没有输入）
      → 连击中断，回到第一段
```

```cpp
// ====================================================================
// 文件：Source/YourProject/Private/GAS/Ability/GA_ComboAttack.cpp
// 描述：连击攻击技能实现
// ====================================================================

#include "GAS/Ability/GA_ComboAttack.h"
#include "Abilities/Tasks/AbilityTask_PlayMontageAndWait.h"
#include "Abilities/Tasks/AbilityTask_WaitGameplayEvent.h"

UGA_ComboAttack::UGA_ComboAttack()
{
    // 默认配置
    ComboWindowDuration = 0.6f;
    BaseSkillMultiplier = 0.8f;
    bIsFinisher = false;
    FinisherDamageBonus = 0.5f;
    AttackAngle = 90.0f;
    AttackRadius = 200.0f;

    // 标签配置
    AbilityTypeTags.AddTag(FGameplayTag::RequestGameplayTag("Ability.Type.Attack"));
    AbilityTypeTags.AddTag(FGameplayTag::RequestGameplayTag("Ability.Type.Combo"));

    ActivationOwnedTags.AddTag(FGameplayTag::RequestGameplayTag("Ability.State.Attacking"));
    ActivationOwnedTags.AddTag(FGameplayTag::RequestGameplayTag("Ability.State.Combo"));

    ActivationBlockedTags.AddTag(FGameplayTag::RequestGameplayTag("State.Dead"));
    ActivationBlockedTags.AddTag(FGameplayTag::RequestGameplayTag("State.Stunned"));

    ComboWindowTag = FGameplayTag::RequestGameplayTag("Combat.Combo.Window");
    NextComboInputTag = FGameplayTag::RequestGameplayTag("Input.Attack.Light");

    NetExecutionPolicy = EGameplayAbilityNetExecutionPolicy::LocalPredicted;

void UGA_ComboAttack::ActivateAbility(
    const FGameplayAbilitySpecHandle Handle,
    const FGameplayAbilityActorInfo* ActorInfo,
    const FGameplayAbilityActivationInfo ActivationInfo,
    const FGameplayEventData* TriggerEventData)
{
    Super::ActivateAbility(Handle, ActorInfo, ActivationInfo, TriggerEventData);

    // ================================================================
    // 连击特定：检查是否在连击窗口内
    // ================================================================
    // 如果是从连击窗口触发的，技能可以有额外加成

    bool bInComboWindow = false;
    if (UAbilitySystemComponent* SourceASC = GetSourceASC())
    {
        // 检查角色是否拥有通用的连击窗口标签
        bInComboWindow = SourceASC->HasMatchingGameplayTag(
            FGameplayTag::RequestGameplayTag("Combat.Combo.Window"));

        if (bInComboWindow)
        {
            // 在连击窗口内激活 → 这是连击的一段！
            UE_LOG(LogTemp, Log, TEXT("ComboAttack: Activated within combo window!"));
        }
    }

    // --- 提交冷却和消耗 ---
    if (!CommitAbility(Handle, ActorInfo, ActivationInfo))
    {
        EndAbility(Handle, ActorInfo, ActivationInfo, true, true);
        return;
    }

    // --- 播放攻击动画 ---
    if (AbilityMontage)
    {
        MontageTask = UAbilityTask_PlayMontageAndWait::CreatePlayMontageAndWaitProxy(
            this, NAME_None, AbilityMontage, MontagePlayRate, MontageStartSection);
        MontageTask->OnCompleted.AddDynamic(this, &UGA_ComboAttack::OnMontageCompleted);
        MontageTask->ReadyForActivation();
    }

    // --- 等待动画伤害检测事件 ---
    WaitDamageEventTask = UAbilityTask_WaitGameplayEvent::WaitGameplayEvent(
        this,
        FGameplayTag::RequestGameplayTag("Event.Melee.DamageCheck"),
        nullptr, false, false);
    WaitDamageEventTask->EventReceived.AddDynamic(
        this, &UGA_ComboAttack::OnDamageCheckEvent);
    WaitDamageEventTask->ReadyForActivation();

    // --- 打开连击窗口（如果这不是终结技）---
    if (!bIsFinisher)
    {
        OpenComboWindow();
    }
}

void UGA_ComboAttack::OpenComboWindow()
{
    UAbilitySystemComponent* SourceASC = GetSourceASC();
    if (!IsValid(SourceASC))
    {
        return;
    }

    // ================================================================
    // 添加连击窗口标签
    // ================================================================
    // 添加两个标签：
    // 1. 通用连击窗口标签（Combat.Combo.Window）
    // 2. 本段特定的连击标签（Combat.Combo.Light1 / Light2）
    //    用于识别当前连击进度

    // 通用窗口标签
    SourceASC->AddLooseGameplayTag(
        FGameplayTag::RequestGameplayTag("Combat.Combo.Window"));

    // 本段特定标签（如 Combat.Combo.Light1）
    if (ComboWindowTag.IsValid())
    {
        SourceASC->AddLooseGameplayTag(ComboWindowTag);
    }

    // ================================================================
    // 设置连击窗口过期计时器
    // ================================================================
    // 如果玩家在ComboWindowDuration秒内没有输入下一段，连击中断

    GetWorld()->GetTimerManager().SetTimer(
        ComboWindowTimerHandle,
        this,
        &UGA_ComboAttack::OnComboWindowExpired,
        ComboWindowDuration,
        false  // 不循环，只执行一次
    );

    UE_LOG(LogTemp, Log, TEXT("ComboAttack: Combo window opened — %s for %.2fs"),
        *ComboWindowTag.ToString(), ComboWindowDuration);
}

void UGA_ComboAttack::OnComboWindowExpired()
{
    // ================================================================
    // 连击窗口过期 → 移除所有连击标签
    // ================================================================
    UAbilitySystemComponent* SourceASC = GetSourceASC();
    if (!IsValid(SourceASC))
    {
        return;
    }

    // 移除通用窗口标签
    SourceASC->RemoveLooseGameplayTag(
        FGameplayTag::RequestGameplayTag("Combat.Combo.Window"));

    // 移除本段特定标签
    if (ComboWindowTag.IsValid())
    {
        SourceASC->RemoveLooseGameplayTag(ComboWindowTag);
    }

    UE_LOG(LogTemp, Log, TEXT("ComboAttack: Combo window EXPIRED — combo broken"));
}

void UGA_ComboAttack::OnDamageCheckEvent(FGameplayEventData Payload)
{
    // 在服务器端执行伤害检测
    if (GetWorld()->GetNetMode() == NM_Client)
    {
        return;  // 客户端不执行伤害逻辑（仅播放特效）
    }

    PerformDamageCheck();
}

void UGA_ComboAttack::PerformDamageCheck()
{
    AActor* SourceActor = GetSourceActor();
    if (!IsValid(SourceActor))
    {
        return;
    }

    // --- 计算实际技能倍率 ---
    float ActualMultiplier = BaseSkillMultiplier;

    // 如果在连击窗口内且是终结技，加上额外伤害加成
    UAbilitySystemComponent* SourceASC = GetSourceASC();
    if (bIsFinisher && IsValid(SourceASC))
    {
        if (SourceASC->HasMatchingGameplayTag(
            FGameplayTag::RequestGameplayTag("Combat.Combo.Window")))
        {
            // 终结技在连击窗口内：获得额外伤害！
            ActualMultiplier += FinisherDamageBonus;
            UE_LOG(LogTemp, Log, TEXT("ComboAttack: FINISHER BONUS! %.2f → %.2f"),
                BaseSkillMultiplier, ActualMultiplier);
        }
    }

    // --- 扇形检测并应用伤害 ---
    TArray<FOverlapResult> OverlapResults;
    TArray<TEnumAsByte<EObjectTypeQuery>> ObjectTypes;
    ObjectTypes.Add(UEngineTypes::ConvertToObjectType(ECC_Pawn));
    TArray<AActor*> Ignored;
    Ignored.Add(SourceActor);

    bool bHit = UKismetSystemLibrary::SphereOverlapActors(
        GetWorld(), SourceActor->GetActorLocation(), AttackRadius,
        ObjectTypes, APawn::StaticClass(), Ignored, OverlapResults);

    if (!bHit) return;

    const FVector Forward = SourceActor->GetActorForwardVector();
    const float CosHalfAngle = FMath::Cos(FMath::DegreesToRadians(AttackAngle * 0.5f));

    for (const FOverlapResult& Overlap : OverlapResults)
    {
        AActor* Target = Overlap.GetActor();
        if (!IsValid(Target)) continue;

        // 角度过滤
        FVector DirToTarget = (Target->GetActorLocation() -
            SourceActor->GetActorLocation()).GetSafeNormal();
        if (FVector::DotProduct(Forward, DirToTarget) <= CosHalfAngle) continue;

        // 获取目标ASC
        UAbilitySystemComponent* TargetASC =
            UAbilitySystemGlobals::GetAbilitySystemComponentFromActor(Target);
        if (!IsValid(TargetASC)) continue;

        // --- 创建伤害GE并应用 ---
        if (!DamageEffectClass) continue;

        FGameplayEffectContextHandle Context = SourceASC->MakeEffectContext();
        FGameplayEffectSpecHandle Spec = SourceASC->MakeOutgoingSpec(
            DamageEffectClass, GetAbilityLevel(), Context);

        if (Spec.IsValid())
        {
            // 设置技能倍率（命中连击时倍率更高）
            Spec.Data->SetSetByCallerMagnitude(
                FGameplayTag::RequestGameplayTag("Data.SkillMultiplier"),
                ActualMultiplier
            );

            // 如果是终结技且暴击 → 添加额外标签
            if (bIsFinisher)
            {
                Spec.Data->DynamicGrantedTags.AddTag(
                    FGameplayTag::RequestGameplayTag("Combat.Combo.Finisher"));
            }

            SourceASC->ApplyGameplayEffectSpecToTarget(*Spec.Data, TargetASC);
        }
    }

    // 播放命中GameplayCue
    if (OverlapResults.Num() > 0)
    {
        // 实际项目中通过GameplayCue播放
    }
}

void UGA_ComboAttack::OnMontageCompleted()
{
    EndAbility(CurrentSpecHandle, CurrentActorInfo, CurrentActivationInfo, true, false);
}

void UGA_ComboAttack::EndAbility(
    const FGameplayAbilitySpecHandle Handle,
    const FGameplayAbilityActorInfo* ActorInfo,
    const FGameplayAbilityActivationInfo ActivationInfo,
    bool bReplicateEndAbility,
    bool bWasCancelled)
{
    // 清理连击窗口计时器
    if (GetWorld())
    {
        GetWorld()->GetTimerManager().ClearTimer(ComboWindowTimerHandle);
    }

    // 如果技能被打断，也要关闭连击窗口
    if (bWasCancelled)
    {
        OnComboWindowExpired();
    }

    if (MontageTask && MontageTask->IsActive()) MontageTask->EndTask();
    if (WaitDamageEventTask && WaitDamageEventTask->IsActive()) WaitDamageEventTask->EndTask();

    Super::EndAbility(Handle, ActorInfo, ActivationInfo, bReplicateEndAbility, bWasCancelled);
}
```

```cpp
bool URPGBuffManager::ApplyBuff(TSubclassOf<UGameplayEffect> BuffEffectClass, float Level)
{
    if (!IsValid(ASC) || !BuffEffectClass)
    {
        return false;
    }

    // 创建GE上下文
    FGameplayEffectContextHandle Context = ASC->MakeEffectContext();
    Context.AddSourceObject(GetOwner());

    // 创建并应用GE到Owner（自身Buff）
    FGameplayEffectSpecHandle Spec = ASC->MakeOutgoingSpec(BuffEffectClass, Level, Context);
    if (!Spec.IsValid())
    {
        return false;
    }

    FActiveGameplayEffectHandle ActiveHandle = ASC->ApplyGameplayEffectSpecToSelf(*Spec.Data.Get());

    if (!ActiveHandle.IsValid())
    {
        return false;
    }

    // 获取Buff信息
    FRPGBuffInfo NewBuffInfo = GetBuffInfoFromActiveEffect(ActiveHandle);
    if (NewBuffInfo.BuffTag.IsValid())
    {
        // 检查是否已存在同Tag的Buff（可能堆叠）
        bool bFoundExisting = false;
        for (FRPGBuffInfo& ExistingBuff : ActiveBuffs)
        {
            if (ExistingBuff.BuffTag == NewBuffInfo.BuffTag)
            {
                // 更新堆叠层数
                ExistingBuff.StackCount++;
                ExistingBuff.ActiveHandle = ActiveHandle;
                ExistingBuff.RemainingTime = NewBuffInfo.RemainingTime;
                ExistingBuff.TotalDuration = NewBuffInfo.TotalDuration;
                OnBuffUpdated.Broadcast(ExistingBuff);
                bFoundExisting = true;
                break;
            }
        }

        if (!bFoundExisting)
        {
            NewBuffInfo.StackCount = 1;
            ActiveBuffs.Add(NewBuffInfo);
            OnBuffAdded.Broadcast(NewBuffInfo);
        }
    }

    return true;
}

bool URPGBuffManager::RemoveBuffByTag(const FGameplayTag& BuffTag)
{
    if (!IsValid(ASC) || !BuffTag.IsValid())
    {
        return false;
    }

    // 查找活跃Buff中匹配Tag的
    for (int32 i = ActiveBuffs.Num() - 1; i >= 0; --i)
    {
        if (ActiveBuffs[i].BuffTag == BuffTag)
        {
            // 移除活跃GE
            if (ActiveBuffs[i].ActiveHandle.IsValid())
            {
                ASC->RemoveActiveGameplayEffect(ActiveBuffs[i].ActiveHandle);
            }

            OnBuffRemoved.Broadcast(BuffTag);
            ActiveBuffs.RemoveAt(i);
            return true;
        }
    }

    return false;
}

void URPGBuffManager::RemoveAllBuffs()
{
    if (!IsValid(ASC)) return;

    for (const FRPGBuffInfo& Buff : ActiveBuffs)
    {
        if (Buff.ActiveHandle.IsValid())
        {
            ASC->RemoveActiveGameplayEffect(Buff.ActiveHandle);
        }
    }

    ActiveBuffs.Empty();
}

void URPGBuffManager::RemoveAllDebuffs()
{
    if (!IsValid(ASC)) return;

    for (int32 i = ActiveBuffs.Num() - 1; i >= 0; --i)
    {
        if (ActiveBuffs[i].bIsDebuff)
        {
            if (ActiveBuffs[i].ActiveHandle.IsValid())
            {
                ASC->RemoveActiveGameplayEffect(ActiveBuffs[i].ActiveHandle);
            }
            ActiveBuffs.RemoveAt(i);
        }
    }
}

bool URPGBuffManager::HasBuff(const FGameplayTag& BuffTag) const
{
    for (const FRPGBuffInfo& Buff : ActiveBuffs)
    {
        if (Buff.BuffTag == BuffTag)
        {
            return true;
        }
    }
    return false;
}

int32 URPGBuffManager::GetBuffStackCount(const FGameplayTag& BuffTag) const
{
    for (const FRPGBuffInfo& Buff : ActiveBuffs)
    {
        if (Buff.BuffTag == BuffTag)
        {
            return Buff.StackCount;
        }
    }
    return 0;
}

void URPGBuffManager::UpdateBuffStates()
{
    if (!IsValid(ASC)) return;

    // 移除已过期的Buff
    for (int32 i = ActiveBuffs.Num() - 1; i >= 0; --i)
    {
        FRPGBuffInfo& Buff = ActiveBuffs[i];

        if (Buff.ActiveHandle.IsValid())
        {
            float Remaining = 0.0f, Duration = 0.0f;
            if (ASC->GetGameplayEffectRemainingDuration(
                    Buff.ActiveHandle, Remaining, Duration))
            {
                Buff.RemainingTime = Remaining;
                Buff.TotalDuration = Duration;
                OnBuffUpdated.Broadcast(Buff);
            }
            else
            {
                // GE已过期但Handle还活跃 → 可能刚结束
                // 检查GE是否真的过期了
                if (Remaining <= 0.0f)
                {
                    OnBuffRemoved.Broadcast(Buff.BuffTag);
                    ActiveBuffs.RemoveAt(i);
                }
            }
        }
        else
        {
            // Handle无效 → Buff已被外部移除
            OnBuffRemoved.Broadcast(Buff.BuffTag);
            ActiveBuffs.RemoveAt(i);
        }
    }
}

    Info.TotalDuration = Duration;

    // 获取GE的标签（从GE的GrantedTags中提取第一个作为Buff标识标签）
    // 实际项目中在GE的GrantedTags中设置 Buff.xxx 标签
    // 这里简化处理

    return Info;
```

### 5.6.2 Boss 阶段系统实现

创建文件 `Source/YourProject/Private/GAS/Boss/BossPhaseSystem.cpp`：

```cpp
// ====================================================================
// 文件：Source/YourProject/Private/GAS/Boss/BossPhaseSystem.cpp
// 描述：Boss阶段系统实现
// ====================================================================

#include "GAS/Boss/BossPhaseSystem.h"
#include "AbilitySystemComponent.h"
#include "AbilitySystemInterface.h"
#include "GAS/AttributeSet/RPGAttributeSet.h"
#include "GameplayEffect.h"

UBossPhaseSystem::UBossPhaseSystem()
{
    PrimaryComponentTick.bCanEverTick = false;
}

void UBossPhaseSystem::BeginPlay()
{
    Super::BeginPlay();

    // 获取Boss的ASC
    if (IAbilitySystemInterface* ASI = Cast<IAbilitySystemInterface>(GetOwner()))
    {
        ASC = ASI->GetAbilitySystemComponent();
    }

    if (!IsValid(ASC))
    {
        UE_LOG(LogTemp, Error, TEXT("BossPhaseSystem: No ASC found on owner!"));
        return;
    }

    // ================================================================
    // 绑定生命值变化监听
    // ================================================================
    // 每当Boss血量变化，检查是否应该切换阶段

    ASC->GetGameplayAttributeValueChangeDelegate(
        URPGAttributeSet::GetHealthAttribute()
    ).AddUObject(this, &UBossPhaseSystem::OnBossHealthChanged);

    // ================================================================
    // 进入初始阶段（阶段1）
    // ================================================================
    if (Phases.Num() > 0)
    {
        SwitchToPhase(1);
    }
}

void UBossPhaseSystem::OnBossHealthChanged(const FOnAttributeChangeData& Data)
{
    // ================================================================
    // 获取当前生命值百分比
    // ================================================================
    // 从ASC获取AttributeSet → 获取MaxHealth → 计算百分比

    const URPGAttributeSet* AttrSet = ASC->GetSet<URPGAttributeSet>();
    if (!AttrSet) return;

    const float MaxHealth = AttrSet->GetMaxHealth();
    if (MaxHealth <= 0.0f) return;

    const float HealthPercent = Data.NewValue / MaxHealth;

    // ================================================================
    // 检查是否应该切换到新阶段
    // ================================================================
    int32 NewPhase = FindPhaseByHealthPercent(HealthPercent);

    if (NewPhase > 0 && NewPhase != CurrentPhaseNumber)
    {
        // 血量降到了新阶段的阈值 → 切换阶段！
        SwitchToPhase(NewPhase);
    }
}

int32 UBossPhaseSystem::FindPhaseByHealthPercent(float HealthPercent) const
{
    // ================================================================
    // 从高阶段到低阶段查找
    // 例如：Phases = [Phase1(1.0), Phase2(0.5), Phase3(0.25)]
    // HealthPercent = 0.3 → Phase3(0.25) 是满足条件的最高阶段 → 返回3
    // ================================================================

    int32 Result = 1;  // 默认至少是阶段1

    for (int32 i = Phases.Num() - 1; i >= 0; --i)
    {
        if (HealthPercent <= Phases[i].HealthThreshold)
        {
            Result = Phases[i].PhaseNumber;
            break;  // 找到第一个（最高阶段）满足条件的
        }
    }

    return Result;
}

void UBossPhaseSystem::ForceSwitchToPhase(int32 PhaseNumber)
{
    SwitchToPhase(PhaseNumber);
}

void UBossPhaseSystem::SwitchToPhase(int32 NewPhaseNumber)
{
    // ================================================================
    // 查找阶段数据
    // ================================================================
    const FBossPhase* NewPhaseData = nullptr;
    for (const FBossPhase& Phase : Phases)
    {
        if (Phase.PhaseNumber == NewPhaseNumber)
        {
            NewPhaseData = &Phase;
            break;
        }
    }

    if (!NewPhaseData)
    {
        UE_LOG(LogTemp, Error, TEXT("BossPhaseSystem: Phase %d not found in config!"),
            NewPhaseNumber);
        return;
    }

    // ================================================================
    // 广播阶段切换前事件
    // ================================================================
    OnPhaseChanging.Broadcast(CurrentPhaseNumber, NewPhaseNumber);

    UE_LOG(LogTemp, Log, TEXT("BossPhaseSystem: Switching from Phase %d to Phase %d — %s"),
        CurrentPhaseNumber, NewPhaseNumber, *NewPhaseData->PhaseName.ToString());

    // ================================================================
    // 清理旧阶段
    // ================================================================
    CleanupCurrentPhase();

    // ================================================================
    // 应用新阶段
    // ================================================================
    ApplyNewPhase(NewPhaseNumber);

    // 更新当前阶段
    CurrentPhaseNumber = NewPhaseNumber;

    // ================================================================
    // 广播阶段切换完成事件
    // ================================================================
    OnPhaseChanged.Broadcast(CurrentPhaseNumber);
}

void UBossPhaseSystem::CleanupCurrentPhase()
{
    if (CurrentPhaseNumber <= 0) return;

    // --- 查找旧阶段数据 ---
    const FBossPhase* OldPhase = nullptr;
    for (const FBossPhase& Phase : Phases)
    {
        if (Phase.PhaseNumber == CurrentPhaseNumber)
        {
            OldPhase = &Phase;
            break;
        }
    }

    if (!OldPhase || !IsValid(ASC)) return;

    // --- 移除旧阶段的技能 ---
    for (TSubclassOf<UGameplayAbility> AbilityClass : OldPhase->PhaseAbilities)
    {
        if (AbilityClass)
        {
            // 查找并移除该技能
            FGameplayAbilitySpec* Spec = ASC->FindAbilitySpecFromClass(AbilityClass);
            if (Spec)
            {
                ASC->ClearAbility(Spec->Handle);
            }
        }
    }

    // --- 移除旧阶段的持续效果 ---
    for (FActiveGameplayEffectHandle& Handle : ActivePhaseGEHandles)
    {
        if (Handle.IsValid())
        {
            ASC->RemoveActiveGameplayEffect(Handle);
        }
    }
    ActivePhaseGEHandles.Empty();

    // --- 移除旧阶段的标签 ---
    if (OldPhase->PhaseTag.IsValid())
    {
        ASC->RemoveLooseGameplayTag(OldPhase->PhaseTag);
    }
}

void UBossPhaseSystem::ApplyNewPhase(int32 PhaseNumber)
{
    if (!IsValid(ASC)) return;

    const FBossPhase* NewPhase = nullptr;
    for (const FBossPhase& Phase : Phases)
    {
        if (Phase.PhaseNumber == PhaseNumber)
        {
            NewPhase = &Phase;
            break;
        }
    }

    if (!NewPhase) return;

    // ================================================================
    // 临时无敌（阶段切换期间）
    // ================================================================
    if (NewPhase->TransitionInvincibleDuration > 0.0f)
    {
        // 添加无敌标签
        FGameplayTag InvincibleTag = FGameplayTag::RequestGameplayTag("State.Invincible");
        ASC->AddLooseGameplayTag(InvincibleTag);

        // 延迟后移除
        FTimerHandle InvincibleTimer;
        GetOwner()->GetWorldTimerManager().SetTimer(
            InvincibleTimer,
            [this, InvincibleTag]()
            {
                if (IsValid(ASC))
                {
                    ASC->RemoveLooseGameplayTag(InvincibleTag);
                }
            },
            NewPhase->TransitionInvincibleDuration,
            false
        );
    }

    // ================================================================
    // 授予新阶段技能
    // ================================================================
    for (TSubclassOf<UGameplayAbility> AbilityClass : NewPhase->PhaseAbilities)
    {
        if (AbilityClass)
        {
            ASC->GiveAbility(FGameplayAbilitySpec(AbilityClass, 1, -1, GetOwner()));
        }
    }

    // ================================================================
    // 应用新阶段的持续效果（如攻击力提升）
    // ================================================================
    for (TSubclassOf<UGameplayEffect> EffectClass : NewPhase->PhasePersistentEffects)
    {
        if (EffectClass)
        {
            FGameplayEffectContextHandle Context = ASC->MakeEffectContext();
            FGameplayEffectSpecHandle Spec = ASC->MakeOutgoingSpec(EffectClass, 1.0f, Context);
            if (Spec.IsValid())
            {
                FActiveGameplayEffectHandle GEHandle =
                    ASC->ApplyGameplayEffectSpecToSelf(*Spec.Data.Get());
                ActivePhaseGEHandles.Add(GEHandle);
            }
        }
    }

    // ================================================================
    // 添加阶段标签
    // ================================================================
    if (NewPhase->PhaseTag.IsValid())
    {
        ASC->AddLooseGameplayTag(NewPhase->PhaseTag);
    }

    UE_LOG(LogTemp, Log, TEXT("BossPhaseSystem: Phase %d applied — %d abilities, %d effects"),
        PhaseNumber, NewPhase->PhaseAbilities.Num(), NewPhase->PhasePersistentEffects.Num());
}

const FBossPhase* UBossPhaseSystem::GetCurrentPhaseData() const
{
    for (const FBossPhase& Phase : Phases)
    {
        if (Phase.PhaseNumber == CurrentPhaseNumber)
        {
            return &Phase;
        }
    }
    return nullptr;
}
```

### 5.7.2 伤害数字GameplayCue实现

```cpp
// ====================================================================
// 文件：Source/YourProject/Private/GAS/GameplayCue/GC_DamageNumber.cpp
// 描述：伤害数字GameplayCue实现
// ====================================================================

#include "GAS/GameplayCue/GC_DamageNumber.h"
#include "Blueprint/UserWidget.h"
#include "Components/TextBlock.h"
#include "Components/WidgetComponent.h"
#include "Kismet/GameplayStatics.h"

bool UGC_DamageNumber::OnExecute_Implementation(
    AActor* Target,
    const FGameplayCueParameters& Parameters) const
{
    if (!IsValid(Target) || !DamageNumberWidgetClass)
    {
        return false;
    }

    // ================================================================
    // 第一步：获取伤害数值
    // ================================================================
    // Parameters.NormalizedMagnitude 包含标准化后的伤害量

    float DamageAmount = Parameters.NormalizedMagnitude;
    if (DamageAmount <= 0.0f)
    {
        return false;  // 没有伤害数值，不显示
    }

    // ================================================================
    // 第二步：判断伤害类型（普通/暴击/治疗）
    // ================================================================
    // 从Parameters的标签中判断

    FLinearColor NumberColor = NormalDamageColor;
    FString Prefix = TEXT("");

    const FGameplayTagContainer& SourceTags = Parameters.AggregatedSourceTags;

    // 检查暴击标签
    if (SourceTags.HasTag(FGameplayTag::RequestGameplayTag("Combat.Result.Critical")))
    {
        NumberColor = CriticalDamageColor;
        Prefix = TEXT("CRIT! ");
    }
    // 检查是否格挡
    else if (SourceTags.HasTag(FGameplayTag::RequestGameplayTag("Combat.Result.Blocked")))
    {
        NumberColor = FLinearColor(0.5f, 0.5f, 0.5f);  // 灰色
        Prefix = TEXT("BLOCK! ");
    }
    // 检查是否治疗
    else if (DamageAmount > 0.0f &&
             SourceTags.HasTag(FGameplayTag::RequestGameplayTag("Combat.Result.Heal")))
    {
        NumberColor = HealColor;
        Prefix = TEXT("+");
    }

    // ================================================================
    // 第三步：在目标上方创建伤害数字Widget
    // ================================================================

    // 创建Widget实例
    UUserWidget* DamageWidget = CreateWidget<UUserWidget>(
        Target->GetWorld(), DamageNumberWidgetClass);

    if (!IsValid(DamageWidget))
    {
        return false;
    }

    // 计算显示位置
    FVector TargetLocation = Target->GetActorLocation() + OffsetFromTarget;

    // 将Widget投射到屏幕空间
    // 实际项目中，使用WidgetComponent或Screen Space + Projection

    // --- 简化处理：通过蓝图实现在Target上方绘制数字 ---
    // 在蓝图中使用Add to Viewport + 世界位置转屏幕坐标

    // 设置文本（如果Widget有TextBlock叫DamageText）
    if (UTextBlock* DamageText = Cast<UTextBlock>(
            DamageWidget->GetWidgetFromName(TEXT("DamageText"))))
    {
        DamageText->SetText(FText::FromString(
            FString::Printf(TEXT("%s%.0f"), *Prefix, DamageAmount)));
        DamageText->SetColorAndOpacity(FSlateColor(NumberColor));
    }

    // 将Widget添加到视口
    DamageWidget->AddToViewport();

    // 设置Widget的屏幕位置（世界坐标转屏幕坐标）
    APlayerController* PC = UGameplayStatics::GetPlayerController(Target->GetWorld(), 0);
    if (IsValid(PC))
    {
        FVector2D ScreenPos;
        if (PC->ProjectWorldLocationToScreen(TargetLocation, ScreenPos))
        {
            DamageWidget->SetPositionInViewport(ScreenPos - FVector2D(50, 0));
        }
    }

    // ================================================================
    // 第四步：设置弹出动画
    // ================================================================
    // 在蓝图中通过动画实现数字上升 + 透明度渐变

    // 在PopDuration秒后从视口移除Widget
    FTimerHandle RemoveTimerHandle;
    Target->GetWorldTimerManager().SetTimer(
        RemoveTimerHandle,
        [DamageWidget]()
        {
            if (IsValid(DamageWidget))
            {
                DamageWidget->RemoveFromParent();
            }
        },
        PopDuration,
        false
    );

    return true;
}
```

## 附录：常见问题排查

### Q1: 技能激活后立即结束，没有播放动画
**原因**: CommitAbility 返回了 false  
**排查**: 检查是否配置了有效的CooldownEffectClass（即使是空冷却），检查CostEffectClass是否正确

### Q2: 伤害应用了但目标血量没有减少
**原因**: IncomingDamage的PostGameplayEffectExecute没有正确实现  
**排查**: 确认 SetIncomingDamage 在函数末尾调用了Reset

### Q3: Buff移除后效果还在
**原因**: 直接移除了LooseTag但没有移除ActiveGE  
**排查**: 使用 ASC->RemoveActiveGameplayEffect 移除GE，GE的标签会自动清除

### Q4: 连击窗口内按技能没有反应
**原因**: 下一段技能的InputTag绑定不正确，或ActivationBlockedTags阻止了激活  
**排查**: 确认InputAction到GameplayTag的映射，确认技能没有在BlockedTags中

### Q5: 客户端和服务器伤害数值不一致
**原因**: 伤害计算在客户端也执行了（应该只在服务器执行）  
**排查**: 在伤害逻辑前添加 `if (GetWorld()->GetNetMode() == NM_Client) return;`
