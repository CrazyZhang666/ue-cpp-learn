# 第6章案例："C++角色 + 蓝图表现"混合交互系统

> **难度**：⭐⭐⭐（中级实战）
> **预计时间**：90分钟
> **目标**：综合运用本章所学的 BlueprintCallable、BlueprintNativeEvent、BlueprintImplementableEvent、动态多播委托，构建一个完整的C++角色系统，在蓝图中扩展表现层逻辑。

---

## 案例目标

创建一个 `AMixedCharacter` 角色类，实现：

1. **HP/MP属性系统**：完整的血量、法力管理
2. **委托通知系统**：血量变化、受伤害、死亡、法力变化的委托广播
3. **BlueprintNativeEvent**：`PlayHitReaction`（C++默认播放动画，蓝图可覆写加粒子特效）
4. **BlueprintImplementableEvent**：`OnLowHealth`（血量低于30%触发，蓝图中做UI警告）
5. **BlueprintCallable**：`GetHealthPercent`、`Heal`、`TakeDamage`、`RestoreMana` 等
6. **蓝图端的对应操作步骤**：从创建蓝图子类到覆写事件的全流程

AMixedCharacter::AMixedCharacter()
{
    // ----- 属性默认值 -----
    // 这些值在编辑器细节面板中可以修改，也允许蓝图子类覆盖

    // 血量系统

    CurrentHealth = MaxHealth;    // 初始血量：满血
    MaxMana = 50.0f;              // 最大法力：50点
    CurrentMana = MaxMana;        // 初始法力：满蓝
    Armor = 10.0f;                // 初始护甲：10点

    // 低血量阈值
    LowHealthThreshold = 0.3f;    // 血量低于30%时触发警告

    // 等级系统
    CurrentLevel = 1;             // 初始等级：1
    CurrentExperience = 0;        // 初始经验：0
    BaseExpForNextLevel = 100;    // 基础经验需求：100（1级升2级需要100经验）

    // 内部状态
    bIsDead = false;              // 初始：未死亡
    bWasLowHealth = false;        // 初始：未处于低血量状态

    // ----- 组件设置 -----
    // CharacterMovement是ACharacter内置的移动组件
    // GetCharacterMovement()返回UCharacterMovementComponent*
}

// ============================================================
// BlueprintCallable 实现
// ============================================================

void AMixedCharacter::TakeDamage(float DamageAmount, AActor* DamageCauser, bool bIsCritical)
{
    // ===== 步骤1：参数验证 =====
    if (DamageAmount <= 0.0f)
    {
        UE_LOG(LogTemp, Warning, TEXT("[%s] TakeDamage: 无效的伤害值 %.1f"), *GetName(), DamageAmount);
        return;  // 伤害值无效，直接返回
    }

    if (bIsDead)
    {
        UE_LOG(LogTemp, Warning, TEXT("[%s] TakeDamage: 角色已死亡，忽略伤害"), *GetName());
        return;  // 已经死亡，不再处理伤害
    }

    // ===== 步骤2：通过NativeEvent计算最终伤害（蓝图可覆写伤害公式）=====
    // 调用原始函数名，引擎自动选择执行C++默认版本还是蓝图覆写版本
    float FinalDamage = CalculateFinalDamage(DamageAmount, DamageCauser, bIsCritical);
    //                  ──────────────────
    // BlueprintNativeEvent：注意调用时用原始函数名，不是 _Implementation
    // 如果蓝图覆写了伤害公式（比如加上元素抗性计算），就执行蓝图版本
    // 如果蓝图没覆写，就执行C++的 CalculateFinalDamage_Implementation

    // ===== 步骤3：记录旧血量并计算新血量 =====
    float OldHealth = CurrentHealth;

    // 使用FMath::Clamp确保血量不超出[0, MaxHealth]范围
    CurrentHealth = FMath::Clamp(CurrentHealth - FinalDamage, 0.0f, MaxHealth);
    //                              ────────────────────  ───┬───  ────┬────
    //                              扣血后的值               下限0    上限MaxHealth
    // FMath::Clamp(值, 最小, 最大)：把值限制在[最小, 最大]内

    // 计算实际造成的伤害（可能因为血量不够而导致实际伤害小于FinalDamage）
    float ActualDamage = OldHealth - CurrentHealth;

    // ===== 步骤4：广播血量变化委托 =====
    // 所有监听者（UI、音效、AI）会同时收到通知
    OnHealthChanged.Broadcast(CurrentHealth, OldHealth, -ActualDamage);
    //           ───┬───
    // 广播：发送消息到所有绑定的监听者
    // -ActualDamage：负值表示血量减少

    // ===== 步骤5：广播受伤害委托 =====
    // 伤害数字UI、受击音效系统等会监听这个委托
    FVector AttackerLocation = DamageCauser
        ? DamageCauser->GetActorLocation()   // 有攻击者，用攻击者的位置
        : GetActorLocation();                 // 没有攻击者，用自己的位置

    OnDamaged.Broadcast(ActualDamage, AttackerLocation, DamageCauser, bIsCritical);
    //       ───┬───
    // 广播受伤害信息：谁打的、伤害多少、打到哪里、是否暴击

    // ===== 步骤6：播放受击反应（NativeEvent）=====
    // C++默认：播放受击动画
    // 蓝图可覆写：额外加粒子特效
    FVector HitDirection = DamageCauser
        ? (GetActorLocation() - AttackerLocation).GetSafeNormal()  // 攻击者指向我的方向
        : -GetActorForwardVector();  // 没有攻击者时默认从前方受击

    PlayHitReaction(HitDirection, ActualDamage);
    //             ────────┬────────
    // NativeEvent：调用原始函数名

    // ===== 步骤7：检查低血量状态 =====
    CheckLowHealthStatus();

    // ===== 步骤8：检查死亡 =====
    if (CurrentHealth <= 0.0f && !bIsDead)
    {
        bIsDead = true;

        // 先广播死亡委托（让监听者有时间反应）
        OnDeath.Broadcast(DamageCauser);
        //       ───┬───
        // 广播死亡信息，传递击杀者

        // 再执行死亡处理（NativeEvent）
        HandleDeath(DamageCauser);
        //         ──────┬─────
        // NativeEvent：调用原始函数名
    }

    // 调试日志
    UE_LOG(LogTemp, Log, TEXT("[%s] 受到%.1f点伤害 (原始%.1f)%s, 剩余血量%.1f/%.1f"),
           *GetName(), ActualDamage, DamageAmount,
           bIsCritical ? TEXT(" [暴击!]") : TEXT(""),
           CurrentHealth, MaxHealth);
}

void AMixedCharacter::Heal(float HealAmount, AActor* Healer)
{
    // ===== 参数验证 =====
    if (HealAmount <= 0.0f)
    {
        UE_LOG(LogTemp, Warning, TEXT("[%s] Heal: 无效的治愈量 %.1f"), *GetName(), HealAmount);
        return;
    }

    if (bIsDead)
    {
        UE_LOG(LogTemp, Warning, TEXT("[%s] Heal: 角色已死亡，不能治愈"), *GetName());
        return;  // 死人不能治愈
    }

    // ===== 计算新旧血量 =====
    float OldHealth = CurrentHealth;
    CurrentHealth = FMath::Clamp(CurrentHealth + HealAmount, 0.0f, MaxHealth);

    // 计算实际治愈量（可能因为满血而导致实际治愈小于HealAmount）
    float ActualHeal = CurrentHealth - OldHealth;

    if (ActualHeal <= 0.0f)
    {
        return;  // 血量已满，没有实际治愈
    }

    // ===== 广播血量变化委托 =====
    OnHealthChanged.Broadcast(CurrentHealth, OldHealth, ActualHeal);
    // ActualHeal：正值表示血量增加

    // ===== 检查低血量状态变化 =====
    CheckLowHealthStatus();

    UE_LOG(LogTemp, Log, TEXT("[%s] 被治愈%.1f点 (来自%s), 当前血量%.1f/%.1f"),
           *GetName(), ActualHeal, Healer ? *Healer->GetName() : TEXT("未知"),
           CurrentHealth, MaxHealth);
}

bool AMixedCharacter::ConsumeMana(float ManaCost)
{
    // ===== 参数验证 =====
    if (ManaCost <= 0.0f)
    {
        UE_LOG(LogTemp, Warning, TEXT("[%s] ConsumeMana: 无效的法力消耗 %.1f"), *GetName(), ManaCost);
        return true;  // 消耗0或负数视为成功
    }

    // ===== 检查法力是否足够 =====
    if (!CanAffordMana(ManaCost))
    {
        // 法力不足，触发蓝图表观事件
        OnManaInsufficient(ManaCost);
        //                ──────┬─────
        // ImplementableEvent：蓝图会显示"法力不足"提示、播放错误音效

        UE_LOG(LogTemp, Warning, TEXT("[%s] 法力不足: 需要%.1f, 当前%.1f"),
               *GetName(), ManaCost, CurrentMana);
        return false;
    }

    // ===== 扣除法力 =====
    float OldMana = CurrentMana;
    CurrentMana -= ManaCost;

    // ===== 广播法力变化委托 =====
    OnManaChanged.Broadcast(CurrentMana, OldMana);

    UE_LOG(LogTemp, Log, TEXT("[%s] 消耗法力%.1f, 剩余%.1f/%.1f"),
           *GetName(), ManaCost, CurrentMana, MaxMana);

    return true;
}

void AMixedCharacter::RestoreMana(float ManaAmount)
{
    // ===== 参数验证 =====
    if (ManaAmount <= 0.0f)
    {
        return;
    }

    // ===== 恢复法力 =====
    float OldMana = CurrentMana;
    CurrentMana = FMath::Clamp(CurrentMana + ManaAmount, 0.0f, MaxMana);

    float ActualRestore = CurrentMana - OldMana;

    if (ActualRestore > 0.0f)
    {
        OnManaChanged.Broadcast(CurrentMana, OldMana);

        UE_LOG(LogTemp, Log, TEXT("[%s] 恢复法力%.1f, 当前%.1f/%.1f"),
               *GetName(), ActualRestore, CurrentMana, MaxMana);
    }
}

bool AMixedCharacter::CanAffordMana(float ManaCost) const
{
    return CurrentMana >= ManaCost;
}

void AMixedCharacter::AddExperience(int32 ExperienceAmount)
{
    // ===== 参数验证 =====
    if (ExperienceAmount <= 0)
    {
        return;
    }

    // ===== 累加经验值 =====
    CurrentExperience += ExperienceAmount;

    // ===== 通知蓝图表观事件 =====
    OnExperienceGained(ExperienceAmount);
    //                 ───────┬──────
    // ImplementableEvent：蓝图中显示经验飘字、经验条增长动画

    // ===== 检查升级 =====
    int32 RequiredExp = GetExperienceForNextLevel();
    while (CurrentExperience >= RequiredExp)
    {
        // 升级：消耗经验值
        CurrentExperience -= RequiredExp;
        CurrentLevel++;

        // 根据新等级更新属性
        UpdateStatsForLevel(CurrentLevel);

        // 广播升级委托
        OnLevelUp.Broadcast(CurrentLevel);

        // 触发蓝图表观事件
        OnLevelUpVisual(CurrentLevel);
        //              ──────┬─────
        // ImplementableEvent：蓝图中播放升级特效

        // 重新计算下一级所需经验
        RequiredExp = GetExperienceForNextLevel();

        UE_LOG(LogTemp, Log, TEXT("[%s] 升级! 达到等级 %d"), *GetName(), CurrentLevel);
    }
}

float AMixedCharacter::GetHealthPercent() const
{
    // 防御性检查：防止除以零
    if (MaxHealth <= 0.0f)
    {
        return 0.0f;
    }
    return CurrentHealth / MaxHealth;
}

float AMixedCharacter::GetManaPercent() const
{
    if (MaxMana <= 0.0f)
    {
        return 0.0f;
    }
    return CurrentMana / MaxMana;
}

int32 AMixedCharacter::GetExperienceForNextLevel() const
{
    // 经验公式：基础经验 * 当前等级^2
    // 1级→2级：100 * 1^2 = 100
    // 2级→3级：100 * 2^2 = 400
    // 3级→4级：100 * 3^2 = 900
    return BaseExpForNextLevel * CurrentLevel * CurrentLevel;
}

float AMixedCharacter::GetExperiencePercent() const
{
    int32 Required = GetExperienceForNextLevel();
    if (Required <= 0)
    {
        return 0.0f;
    }
    return static_cast<float>(CurrentExperience) / static_cast<float>(Required);
}

void AMixedCharacter::HandleDeath_Implementation(AActor* Killer)
{
    // C++默认死亡处理：

    UE_LOG(LogTemp, Warning, TEXT("[%s] C++默认死亡处理: 被 [%s] 击杀"),
           *GetName(), Killer ? *Killer->GetName() : TEXT("未知"));

    // 1. 禁用移动输入
    if (APlayerController* PC = Cast<APlayerController>(GetController()))
    //  ──┬──                     ─────────┬─────────
    // Cast<子类*>(父类指针)：安全类型转换，失败返回nullptr
    {
        DisableInput(PC);  // 禁用所有输入
        // 注意：DisableInput是Actor的成员函数，不是完全禁用，详见第7章
    }

    // 2. 停止移动组件
    UCharacterMovementComponent* MoveComp = GetCharacterMovement();
    if (MoveComp)
    {
        MoveComp->StopMovementImmediately();  // 立即停止所有移动
        MoveComp->DisableMovement();           // 禁用移动组件
        MoveComp->SetComponentTickEnabled(false);  // 关闭每帧Tick
    }

    // 3. 关闭碰撞（让其他Actor可以穿过尸体）
    SetActorEnableCollision(false);

    // 4. 播放死亡动画
    if (DeathMontage)
    {
        PlayAnimMontage(DeathMontage, 1.0f);
    }

    // 蓝图可以覆写此函数做额外的事（如下）：
    //   ├── 播放死亡动画 + 渐变消失
    //   ├── 通知游戏模式检查胜负条件
    //   ├── 延时5秒后销毁Actor
    //   ├── 掉落物品/金币
    //   └── 播放死亡特效（碎裂、光点消散等）
}

float AMixedCharacter::CalculateFinalDamage_Implementation(float RawDamage, AActor* DamageCauser,
                                                            bool bIsCritical)
{
    // C++默认伤害计算公式：简单的护甲减伤

    // 护甲减伤公式（简单版本）
    // 每点护甲减少0.5%伤害，最多减少75%
    float ArmorReductionPercent = FMath::Min(Armor * 0.005f, 0.75f);
    //                              ──┬──
    // FMath::Min(a, b)：取两个值中较小的
    // 护甲 * 0.005 = 护甲值 * 0.5%
    // 10护甲 = 5%减伤，100护甲 = 50%减伤，150护甲 = 75%减伤（到上限）

    float DamageAfterArmor = RawDamage * (1.0f - ArmorReductionPercent);
    //                      原始伤害 * (1 - 减伤百分比) = 护甲减伤后的伤害

    // 暴击倍率：暴击时伤害翻倍
    if (bIsCritical)
    {
        DamageAfterArmor *= 2.0f;
    }

    // 确保最终伤害最小为1（即使是1点护甲对1点伤害）
    float FinalDamage = FMath::Max(DamageAfterArmor, 1.0f);
    //                  ──┬──
    // FMath::Max(a, b)：取两个值中较大的

    UE_LOG(LogTemp, Log, TEXT("[%s] C++伤害计算: 原始%.1f → 护甲减%.0f%% → 最终%.1f%s"),
           *GetName(), RawDamage, ArmorReductionPercent * 100.0f, FinalDamage,
           bIsCritical ? TEXT(" [暴击!]") : TEXT(""));

    return FinalDamage;

    // 蓝图可以覆写此函数做额外的事（如下）：
    //   ├── 读取Buff系统的影响（如"易伤"debuff增加受到伤害）
    //   ├── 元素抗性系统（火球打火元素=减伤）
    //   ├── 格挡/闪避判定
    //   └── 护盾吸收伤害
}

// ImplementableEvent：这些函数不需要C++实现！
// OnLowHealth()
// OnLevelUpVisual(int32 NewLevel)
// OnManaInsufficient(float RequiredMana)
// OnExperienceGained(int32 Amount)
// ============================================================

// ============================================================
// 内部辅助函数实现
// ============================================================

void AMixedCharacter::UpdateStatsForLevel(int32 NewLevel)
{
    // 根据等级更新属性的基础公式

    // 最大血量 = 基础100 + 每级20
    // 等级1：120，等级2：140，等级10：300 ...
    float NewMaxHealth = 100.0f + (NewLevel - 1) * HealthPerLevel;
    MaxHealth = NewMaxHealth;

    // 升级时回复全部血量
    CurrentHealth = MaxHealth;

    // 最大法力 = 基础50 + 每级10
    float NewMaxMana = 50.0f + (NewLevel - 1) * ManaPerLevel;
    MaxMana = NewMaxMana;

    // 升级时回复全部法力
    CurrentMana = MaxMana;

    // 护甲 = 基础10 + 每级2
    Armor = 10.0f + (NewLevel - 1) * ArmorPerLevel;

    UE_LOG(LogTemp, Log, TEXT("[%s] 属性更新至等级%d：HP=%.0f MP=%.0f 护甲=%.0f"),
           *GetName(), NewLevel, MaxHealth, MaxMana, Armor);
}

void AMixedCharacter::CheckLowHealthStatus()
{
    // 获取当前的低血量状态
    bool bCurrentlyLowHealth = IsLowHealth();

    // 只在状态发生"变化"时触发事件（避免每帧都触发）
    if (bCurrentlyLowHealth && !bWasLowHealth)
    {
        // 刚从健康变成低血量 → 触发警告
        OnLowHealth();
        //         ────────
        // ImplementableEvent：蓝图会显示低血量UI警告

        UE_LOG(LogTemp, Warning, TEXT("[%s] 进入低血量状态 (%.0f%%)"),
               *GetName(), GetHealthPercent() * 100.0f);
    }
    else if (!bCurrentlyLowHealth && bWasLowHealth)
    {
        // 刚从低血量恢复到健康 → 取消警告（蓝图自行处理）
        UE_LOG(LogTemp, Log, TEXT("[%s] 低血量状态解除 (%.0f%%)"),
               *GetName(), GetHealthPercent() * 100.0f);
    }

    // 更新状态记录
    bWasLowHealth = bCurrentlyLowHealth;
}

### 扩展1：添加法力护盾系统（NativeEvent）

```cpp
// 在 MixedCharacter.h 中添加：

/** 护盾吸收伤害（NativeEvent，蓝图可覆写吸收公式） */
UFUNCTION(BlueprintNativeEvent, Category = "MixedCharacter|Combat")
float AbsorbDamageWithShield(float IncomingDamage);

UCLASS()
class AMixedCharacter
{
    // ...

    /** 是否启用法力护盾 */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "MixedCharacter|Combat")
    bool bManaShieldActive = false;

    /** 每点法力吸收的伤害量 */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "MixedCharacter|Combat")
    float ManaShieldRatio = 2.0f;  // 1点法力吸收2点伤害
};

// 在 TakeDamage 中，第三步之前插入：

// 如果法力护盾激活且法力 > 0
if (bManaShieldActive && CurrentMana > 0.0f)
{
    // 调用NativeEvent：蓝图中可以覆写吸收公式
    float Absorbed = AbsorbDamageWithShield(FinalDamage);

    // 从法力中扣除等量法力
    float ManaCost = Absorbed / ManaShieldRatio;
    CurrentMana = FMath::Max(CurrentMana - ManaCost, 0.0f);

    // 减少实际伤害
    FinalDamage = FMath::Max(FinalDamage - Absorbed, 0.0f);

    OnManaChanged.Broadcast(CurrentMana, CurrentMana + ManaCost);
}
```
