# 第3章案例：可配置的拾取物系统

> **难度**：⭐⭐（进阶入门级）
> **预计时间**：45分钟
> **目标**：创建一个完整的拾取物Actor，综合练习UCLASS/USTRUCT/UENUM/UPROPERTY/UFUNCTION的全部用法。

---

## 案例目标

创建一个 `APickupItem` Actor，玩家靠近后自动拾取。包含：

- 不同类型的拾取物（枚举）
- 丰富的属性配置（UPROPERTY）
- 蓝图可调用的接口（UFUNCTION）
- 一个可蓝图继承的基类（UCLASS的各种参数）
- 拾取物的数据结构体（USTRUCT）

---

## 第一步：创建枚举和结构体

首先创建 `PickupTypes.h`：

```cpp
// PickupTypes.h
#pragma once

#include "CoreMinimal.h"
#include "Engine/DataTable.h"  // 支持DataTable的结构体需要这个
#include "PickupTypes.generated.h"  // 必须在最后！

// ===== 1. 拾取物类型枚举 =====
UENUM(BlueprintType)
enum class EPickupType : uint8
{
    Health      UMETA(DisplayName = "生命恢复"),
    Mana        UMETA(DisplayName = "法力恢复"),
    Ammo        UMETA(DisplayName = "弹药"),
    Weapon      UMETA(DisplayName = "武器"),
    Armor       UMETA(DisplayName = "护甲"),
    KeyItem     UMETA(DisplayName = "关键道具"),
    Gold        UMETA(DisplayName = "金币")
};

// ===== 2. 拾取物品质等级 =====
UENUM(BlueprintType)
enum class EPickupRarity : uint8
{
    Common      UMETA(DisplayName = "普通（白色）"),
    Uncommon    UMETA(DisplayName = "不凡（绿色）"),
    Rare        UMETA(DisplayName = "稀有（蓝色）"),
    Epic        UMETA(DisplayName = "史诗（紫色）"),
    Legendary   UMETA(DisplayName = "传说（橙色）")
};

// ===== 3. 拾取物数据行（可在DataTable中使用）=====
USTRUCT(BlueprintType)
struct FPickupData : public FTableRowBase  // 继承自FTableRowBase以支持DataTable
{
    GENERATED_BODY()

    // 拾取物名称
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "名称"))
    FText DisplayName;

    // 拾取物类型
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "类型"))
    EPickupType Type = EPickupType::Health;

    // 品质
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "品质"))
    EPickupRarity Rarity = EPickupRarity::Common;

    // 数值（生命恢复量、弹药数量等）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "数值", ClampMin = "0"))
    int32 Value = 10;

    // 自动拾取范围
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "拾取范围（厘米）", ClampMin = "0"))
    float PickupRadius = 150.0f;

    // 是否自动拾取（走过去就捡） vs 手动拾取（按E键捡）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "自动拾取"))
    bool bAutoPickup = true;

    // 拾取后是否销毁
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup",
              meta = (DisplayName = "拾取后销毁"))
    bool bDestroyOnPickup = true;
};
```

### 逐行理解

```cpp

//      DataTable = UE内置的表格数据系统（类似Excel读入UE）
```

## 第二步：创建拾取物Actor

创建 `PickupItem.h`：

```cpp
// PickupItem.h
#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "PickupTypes.h"              // 使用我们自己定义的枚举和结构体
#include "PickupItem.generated.h"

// 前置声明
class USphereComponent;
class UStaticMeshComponent;
class URotatingMovementComponent;

UCLASS(Blueprintable, BlueprintType)  // 蓝图可继承、可作为变量类型
class MYGAME_API APickupItem : public AActor
{
    GENERATED_BODY()

public:
    APickupItem();

    // ===== 组件 =====
    // 球形碰撞体：检测玩家进入范围
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Pickup | Components")
    USphereComponent* TriggerSphere;

    // 视觉模型
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Pickup | Components")
    UStaticMeshComponent* ItemMesh;

    // ===== 配置属性 =====
    // 拾取物数据（可以用DataTable赋值，也可以在编辑器中手动填）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Pickup | Config",
              meta = (DisplayName = "拾取物数据"))
    FPickupData PickupData;

    // 是否已被拾取（防止重复拾取）
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Pickup | State",
              meta = (DisplayName = "已拾取"))
    bool bIsPickedUp = false;

protected:
    virtual void BeginPlay() override;

    // ===== 碰撞回调 =====
    // 当有物体进入触发球时调用
    UFUNCTION()
    void OnTriggerBeginOverlap(
        UPrimitiveComponent* OverlappedComponent,  // 我们的碰撞体
        AActor* OtherActor,                         // 进入范围的Actor
        UPrimitiveComponent* OtherComp,             // OtherActor的碰撞体组件
        int32 OtherBodyIndex,                       // 碰撞体索引
        bool bFromSweep,                            // 是否来自扫描
        const FHitResult& SweepResult               // 碰撞结果详情
    );

    // ===== 拾取逻辑 =====
    // C++实现的核心拾取逻辑
    void OnPickedUp(class AMyCharacter* PickupCharacter);  // 拾取角色

    // 蓝图事件：拾取时播放特效（蓝图实现）
    UFUNCTION(BlueprintImplementableEvent, Category = "Pickup | Events")
    void OnPickupEffects();  // C++不实现，在蓝图中写特效逻辑

public:
    // 蓝图可调用的查询函数
    UFUNCTION(BlueprintPure, Category = "Pickup")
    FText GetDisplayName() const { return PickupData.DisplayName; }

    UFUNCTION(BlueprintPure, Category = "Pickup")
    EPickupType GetPickupType() const { return PickupData.Type; }

    UFUNCTION(BlueprintPure, Category = "Pickup")
    bool IsPickedUp() const { return bIsPickedUp; }
};
```

## 第三步：实现拾取物逻辑

创建 `PickupItem.cpp`：

```cpp
// PickupItem.cpp
#include "PickupItem.h"
#include "Components/SphereComponent.h"
#include "Components/StaticMeshComponent.h"
#include "MyCharacter.h"  // 你的玩家角色类

APickupItem::APickupItem()
{
    // 让这个Actor每帧Tick（只有需要每帧旋转或浮动时才开）
    PrimaryActorTick.bCanEverTick = false;  // 拾取物不需要每帧逻辑

    // ===== 创建根组件：触发球 =====
    TriggerSphere = CreateDefaultSubobject<USphereComponent>(TEXT("TriggerSphere"));
    RootComponent = TriggerSphere;  // 触发球作为根组件

    // 设置触发球属性
    TriggerSphere->SetSphereRadius(150.0f);  // 默认150cm半径
    TriggerSphere->SetCollisionEnabled(ECollisionEnabled::QueryOnly);  // 只检测不阻挡
    TriggerSphere->SetCollisionObjectType(ECC_WorldDynamic);  // 碰撞类型

    // 设置触发球只响应Pawn（角色）的进入
    TriggerSphere->SetCollisionResponseToAllChannels(ECR_Ignore);  // 先忽略所有
    TriggerSphere->SetCollisionResponseToChannel(ECC_Pawn, ECR_Overlap);  // 只响应Pawn

    // ===== 创建视觉模型 =====
    ItemMesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("ItemMesh"));
    ItemMesh->SetupAttachment(RootComponent);  // 挂在触发球下面

    // 设置模型不产生碰撞（只有触发球检测碰撞）
    ItemMesh->SetCollisionEnabled(ECollisionEnabled::NoCollision);

    // 加载默认模型（一个发光球体，引擎自带）
    static ConstructorHelpers::FObjectFinder<UStaticMesh> DefaultMesh(
        TEXT("/Engine/BasicShapes/Sphere")
    );
    if (DefaultMesh.Succeeded())
    {
        ItemMesh->SetStaticMesh(DefaultMesh.Object);
        ItemMesh->SetRelativeScale3D(FVector(0.5f));  // 缩小到50%
    }
}

void APickupItem::BeginPlay()
{
    Super::BeginPlay();

    // 绑定碰撞事件
    // AddDynamic：把OnTriggerBeginOverlap函数绑定到OnComponentBeginOverlap事件
    // 当有东西进入触发球时，自动调用OnTriggerBeginOverlap
    TriggerSphere->OnComponentBeginOverlap.AddDynamic(this, &APickupItem::OnTriggerBeginOverlap);

    // 根据拾取数据中的半径更新触发球大小
    TriggerSphere->SetSphereRadius(PickupData.PickupRadius);

    UE_LOG(LogTemp, Warning, TEXT("拾取物已生成: %s"), *PickupData.DisplayName.ToString());
}

void APickupItem::OnTriggerBeginOverlap(
    UPrimitiveComponent* OverlappedComponent,
    AActor* OtherActor,
    UPrimitiveComponent* OtherComp,
    int32 OtherBodyIndex,
    bool bFromSweep,
    const FHitResult& SweepResult)
{
    // 已经被捡过了，不重复处理
    if (bIsPickedUp)
    {
        return;
    }

    // 检查进入范围的Actor是否是玩家角色
    AMyCharacter* PlayerCharacter = Cast<AMyCharacter>(OtherActor);
    //  Cast<T> = UE安全的类型转换
    //  如果OtherActor是AMyCharacter(或其子类)，返回指针
    //  如果不是，返回nullptr

    if (PlayerCharacter == nullptr)
    {
        return;  // 不是玩家角色，不拾取
    }

    // 检查是否自动拾取
    if (!PickupData.bAutoPickup)
    {
        // 不是自动拾取，玩家需要按交互键——这里可以触发一个提示UI
        UE_LOG(LogTemp, Warning, TEXT("靠近 %s，按E拾取"), *PickupData.DisplayName.ToString());
        return;
    }

    // 执行拾取！
    OnPickedUp(PlayerCharacter);
}

void APickupItem::OnPickedUp(AMyCharacter* PickupCharacter)
{
    if (bIsPickedUp)
    {
        return;  // 双重检查，防止并发重复拾取
    }

    bIsPickedUp = true;

    // 根据拾取物类型执行不同的效果
    switch (PickupData.Type)
    {
        case EPickupType::Health:
            // 通知角色恢复血量
            PickupCharacter->Heal(PickupData.Value);
            UE_LOG(LogTemp, Warning, TEXT("拾取了生命恢复 +%d"), PickupData.Value);
            break;

        case EPickupType::Gold:
            // 增加金币
            PickupCharacter->AddGold(PickupData.Value);
            UE_LOG(LogTemp, Warning, TEXT("拾取了金币 +%d"), PickupData.Value);
            break;

        case EPickupType::Weapon:
            // 拾取武器（本节简化处理）
            UE_LOG(LogTemp, Warning, TEXT("拾取了武器: %s"), *PickupData.DisplayName.ToString());
            break;

        default:
            UE_LOG(LogTemp, Warning, TEXT("拾取了: %s x%d"),
                   *PickupData.DisplayName.ToString(), PickupData.Value);
            break;
    }

    // 通知蓝图播放拾取特效（BlueprintImplementableEvent）
    OnPickupEffects();

    // 是否拾取后销毁
    if (PickupData.bDestroyOnPickup)
    {
        Destroy();  // 从场景中移除此Actor
    }
}
```

## 实践扩展

### 扩展1：创建蓝图子类

1. 右键 `APickupItem` → Create Blueprint Class
2. 命名为 `BP_HealthPickup`
3. 打开蓝图 → Event Graph → 实现 `OnPickupEffects` 事件
4. 添加粒子特效和音效

### 扩展2：使用DataTable批量配置

1. 在Content中创建DataTable，行结构选 `FPickupData`
2. 像填Excel一样填充多行数据
3. 在代码中从DataTable读取数据来创建拾取物
