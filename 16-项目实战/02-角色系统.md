# 02 — 角色系统

> **章节目标**
> 实现游戏的核心——玩家角色 `ADungeonCharacter`。本章涵盖移动、属性、状态机、装备、升级和摄像机系统的完整实现。这是整个项目中代码量最大的一个类。

---

## 2.1 文件创建

### 2.1.1 创建 C++ 类

在UE编辑器中执行以下操作：

```
1. Tools → New C++ Class...
2. 父类选择 Character
3. 类名输入 DungeonCharacter
4. 路径选择 Source/DungeonCrawler/Character/
5. 点击 Create Class
6. 等待编译完成
```

生成两个文件：
- `Source/DungeonCrawler/Character/DungeonCharacter.h`
- `Source/DungeonCrawler/Character/DungeonCharacter.cpp`

## 2.2 DungeonCharacter.h — 头文件完整实现

```cpp
// ============================================================================
// DungeonCharacter.h — 玩家角色类头文件
// ============================================================================
// 这是整个游戏最重要的类。它定义了玩家角色的所有属性、状态和行为。
// 注释中的「属性」指 UPROPERTY 标记的成员变量
// 注释中的「方法」指 UFUNCTION 标记的成员函数
// ============================================================================

#pragma once

// ── 必要的头文件引用 ──
#include "CoreMinimal.h"                    // UE核心类型定义（FString, TArray等）
#include "GameFramework/Character.h"        // ACharacter 父类
#include "DungeonCharacter.generated.h"     // UHT（Unreal Header Tool）生成的头文件，必须最后include

// ── 前向声明（减少编译依赖，加快编译速度）──
// 前向声明的意思是："这些类存在，但我暂时不需要知道它们的完整定义"
// 在 .h 文件中能用前向声明就用前向声明，在 .cpp 中再 #include 完整头文件
class USpringArmComponent;                   // UE的弹簧臂组件，用于摄像机控制
class UCameraComponent;                      // UE的摄像机组件
class UDungeonEquipment;                     // 我们自己的装备类（第4章实现）
class UDungeonItemBase;                      // 物品基类
class UDungeonSkillBase;                     // 技能基类

// ── 枚举定义 ──
// UENUM(BlueprintType) 让枚举在蓝图中可用
// uint8 表示用8位无符号整数存储（可表示0-255，足够状态机使用）
UENUM(BlueprintType)
enum class ECharacterState : uint8
{
    Idle       UMETA(DisplayName = "待机"),    // 站在原地不动
    Moving     UMETA(DisplayName = "移动"),    // WASD移动中
    Attacking  UMETA(DisplayName = "攻击中"),  // 执行攻击动画，此时不可移动
    Blocking   UMETA(DisplayName = "格挡中"),  // 按住右键格挡
    Dodging    UMETA(DisplayName = "闪避中"),  // 闪避动画播放中（无敌帧）
    Stunned    UMETA(DisplayName = "硬直中"),  // 被敌人重击后短暂僵直
    Dead       UMETA(DisplayName = "死亡"),    // 生命值归零
};

// 装备槽位枚举
UENUM(BlueprintType)
enum class EEquipmentSlot : uint8
{
    None    UMETA(DisplayName = "无"),
    Weapon  UMETA(DisplayName = "武器"),      // 影响攻击力
    Helmet  UMETA(DisplayName = "头盔"),      // 影响防御力
    Armor   UMETA(DisplayName = "铠甲"),      // 影响防御力（比头盔加成更高）
    Boots   UMETA(DisplayName = "鞋子"),      // 影响移动速度
    Ring    UMETA(DisplayName = "戒指"),      // 特殊属性加成
};

// 可分配的属性类型
UENUM(BlueprintType)
enum class EAttributeType : uint8
{
    Strength     UMETA(DisplayName = "力量"),  // 每点+2攻击力
    Agility      UMETA(DisplayName = "敏捷"),  // 每点+1%暴击率
    Vitality     UMETA(DisplayName = "体质"),  // 每点+10最大生命值
    Intelligence UMETA(DisplayName = "智力"),  // 每点+5最大法力值
    Endurance    UMETA(DisplayName = "耐力"),  // 每点+5最大体力值
};

// ── 属性结构体 ──
// USTRUCT(BlueprintType) 让结构体在蓝图中可用
// 这个结构体保存角色的所有战斗属性
USTRUCT(BlueprintType)
struct FCharacterAttributes
{
    GENERATED_BODY()  // USTRUCT也必须有这个宏

    // 力量 — 主要影响物理攻击力
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|基础")
    float Strength = 10.0f;

    // 敏捷 — 影响暴击率和少量闪避
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|基础")
    float Agility = 10.0f;

    // 体质 — 主要影响生命值上限
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|基础")
    float Vitality = 10.0f;

    // 智力 — 主要影响法力值上限和技能伤害
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|基础")
    float Intelligence = 10.0f;

    // 耐力 — 主要影响体力值上限
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|基础")
    float Endurance = 10.0f;

    // ── 派生属性（由基础属性计算得出，不直接加点）──
    // 总攻击力 = 基础力量*2 + 装备加成
    float GetAttackPower() const { return Strength * 2.0f; }

    // 暴击率 = 敏捷 * 1% + 装备加成
    float GetCriticalChance() const { return Agility * 0.01f; }

    // 最大生命值 = 体质 * 10 + 基础值
    float GetMaxHealth() const { return 100.0f + Vitality * 10.0f; }

    // 最大法力值 = 智力 * 5 + 基础值
    float GetMaxMana() const { return 50.0f + Intelligence * 5.0f; }

    // 最大体力值 = 耐力 * 5 + 基础值
    float GetMaxStamina() const { return 50.0f + Endurance * 5.0f; }
};

// ============================================================================
// ADungeonCharacter 类声明
// ============================================================================
// class 后面跟冒号表示继承关系：ADungeonCharacter 继承自 ACharacter
// class 前面的 DUNGEONCRAWLER_API 是模块导出宏，确保类可以被外部模块使用
// ============================================================================
UCLASS()
class DUNGEONCRAWLER_API ADungeonCharacter : public ACharacter
{
    GENERATED_BODY()  // UCLASS必须配GENERATED_BODY()宏

public:
    // ========================================================================
    // 构造函数
    // ========================================================================
    ADungeonCharacter();

    // ========================================================================
    // 重写基类虚函数
    // ========================================================================
    // BeginPlay — 游戏开始时调用（类似Unity的Start）
    virtual void BeginPlay() override;

    // Tick — 每帧调用（类似Unity的Update）
    virtual void Tick(float DeltaTime) override;

    // SetupPlayerInputComponent — 绑定玩家输入（设置按键映射的地方）
    virtual void SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent) override;

protected:
    // ========================================================================
    // 摄像机组件
    // ========================================================================
    // VisibleAnywhere — 在蓝图细节面板可见（但不可编辑，因为是组件指针）
    // BlueprintReadOnly — 蓝图可读取
    // Category 用于在细节面板中分组织
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "摄像机")
    USpringArmComponent* CameraBoom;
    // 弹簧臂的作用：
    // 1. 在角色和摄像机之间保持固定距离
    // 2. 碰到墙壁时自动缩短距离（防止摄像机穿墙）
    // 3. 旋转时产生平滑的延迟跟随效果

    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "摄像机")
    UCameraComponent* FollowCamera;
    // 这是实际"拍摄画面"的摄像机组件
    // 它附着在弹簧臂的末端（Socket）

    // ── 摄像机参数 ──
    // 默认臂长（摄像机距离角色的默认距离）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "摄像机")
    float DefaultBoomLength = 350.0f;

    // 最小臂长（滚轮缩到最近的距离）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "摄像机")
    float MinBoomLength = 150.0f;

    // 最大臂长（滚轮拉到最远的距离）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "摄像机")
    float MaxBoomLength = 600.0f;

    // 瞄准时的臂长（通常比默认更近）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "摄像机")
    float AimBoomLength = 200.0f;

    // 滚轮缩放速度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "摄像机")
    float ZoomSpeed = 50.0f;

    // 当前目标臂长（用于平滑插值）
    float TargetBoomLength;

    // ========================================================================
    // 移动系统
    // ========================================================================
    // 基础移动速度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float WalkSpeed = 450.0f;

    // 冲刺速度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float SprintSpeed = 800.0f;

    // 跳跃高度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float JumpVelocity = 500.0f;

    // 闪避距离（向前冲的距离）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float DodgeDistance = 500.0f;

    // 闪避无敌帧持续时间（秒）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float DodgeInvincibleDuration = 0.2f;

    // 闪避体力消耗
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float DodgeStaminaCost = 20.0f;

    // 闪避冷却时间（防止无限闪避）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "移动")
    float DodgeCooldown = 0.5f;

    // 上次闪避的时间戳
    float LastDodgeTime = -999.0f;

    // ========================================================================
    // 属性系统
    // ========================================================================
    // 角色基础属性（力量、敏捷等）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性")
    FCharacterAttributes BaseAttributes;

    // 当前生命值（0 到 最大生命值）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|当前值")
    float CurrentHealth = 100.0f;

    // 当前法力值（消耗法力使用技能）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|当前值")
    float CurrentMana = 50.0f;

    // 当前体力值（消耗体力闪避/冲刺/格挡）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "属性|当前值")
    float CurrentStamina = 50.0f;

    // ========================================================================
    // 升级系统
    // ========================================================================
    // 当前等级
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "升级")
    int32 CharacterLevel = 1;

    // 当前经验值
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "升级")
    int32 CurrentExperience = 0;

    // 可用属性点
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "升级")
    int32 AvailableAttributePoints = 0;

    // ========================================================================
    // 装备系统
    // ========================================================================
    // TMap 是UE的"字典/映射"容器，用枚举作为Key，装备指针作为Value
    // 这样做的好处是——可以直接通过 EEquipmentSlot::Weapon 来获取武器
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "装备")
    TMap<EEquipmentSlot, UDungeonEquipment*> EquipmentSlots;

    // 背包（所有物品的数组）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "装备")
    TArray<UDungeonItemBase*> Inventory;

    // 背包最大容量
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "装备")
    int32 MaxInventorySize = 30;

    // ========================================================================
    // 战斗系统
    // ========================================================================
    // 当前是否在攻击中（用于防止攻击期间移动）
    UPROPERTY(BlueprintReadOnly, Category = "战斗")
    bool bIsAttacking = false;

    // 当前是否在格挡中
    UPROPERTY(BlueprintReadOnly, Category = "战斗")
    bool bIsBlocking = false;

    // 是否处于无敌状态
    UPROPERTY(BlueprintReadOnly, Category = "战斗")
    bool bIsInvincible = false;

    // 轻攻击伤害倍率
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "战斗")
    float LightAttackDamageMultiplier = 1.0f;

    // 重攻击伤害倍率
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "战斗")
    float HeavyAttackDamageMultiplier = 2.0f;

    // ========================================================================
    // 技能系统
    // ========================================================================
    // 4个技能槽（Q/E/R/F 分别对应索引 0/1/2/3）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "技能")
    TArray<UDungeonSkillBase*> Skills;

    // ========================================================================
    // 状态机
    // ========================================================================
    // 当前角色状态
    UPROPERTY(BlueprintReadOnly, Category = "状态")
    ECharacterState CurrentState = ECharacterState::Idle;

public:
    // ========================================================================
    // 公有的 Getter/Setter 方法
    // ========================================================================

    // ── 属性获取 ──
    // 获取总攻击力 = 基础攻击力 + 所有装备提供的攻击力加成
    UFUNCTION(BlueprintCallable, Category = "属性")
    float GetTotalAttack() const;

    // 获取总防御力 = 基础防御力 + 所有装备提供的防御力加成

    float GetTotalDefense() const;

    // 获取最大生命值
    UFUNCTION(BlueprintCallable, Category = "属性")
```

## 2.3 DungeonCharacter.cpp — 源文件完整实现

```cpp
// ============================================================================
// DungeonCharacter.cpp — 玩家角色类实现
// ============================================================================
// 每一行代码都有注释，帮助你理解"为什么这样写"而不只是"怎么写"
// ============================================================================

#include "DungeonCharacter.h"                // 自己的头文件
#include "GameFramework/SpringArmComponent.h" // 弹簧臂组件定义
#include "Camera/CameraComponent.h"           // 摄像机组件定义
#include "Components/CapsuleComponent.h"      // 胶囊碰撞体（ACharacter自带的）
#include "GameFramework/CharacterMovementComponent.h" // 角色移动组件
#include "Components/InputComponent.h"        // 输入组件（绑定按键用）
#include "Engine/World.h"                     // 获取世界上下文
#include "TimerManager.h"                     // 定时器管理（延迟执行）

// ── 构造函数 ──
// 构造函数在对象创建时调用（在CDO——类默认对象创建时执行）
// 这里设置所有组件的默认值和层级关系
ADungeonCharacter::ADungeonCharacter()
{
    // ===================================================================
    // Tick设置
    // ===================================================================
    // 设为true表示每帧调用此Actor的Tick函数（默认是true，这里显式确认）
    PrimaryActorTick.bCanEverTick = true;
    // 如果不需要Tick可以设为false，节省性能。但体力恢复等需要每帧更新，所以设为true

    // ===================================================================
    // 摄像机设置
    // ===================================================================
    // 创建弹簧臂组件 — 命名 "CameraBoom"
    CameraBoom = CreateDefaultSubobject<USpringArmComponent>(TEXT("CameraBoom"));
    // 把弹簧臂附着到根组件（ACharacter的CapsuleComponent是根组件）
    CameraBoom->SetupAttachment(RootComponent);
    // 弹簧臂长度 = 摄像机距离角色的距离（单位是厘米，UE5使用厘米作为单位）
    CameraBoom->TargetArmLength = DefaultBoomLength;  // 350.0f cm = 3.5米
    // 启用旋转延迟（Lag），让摄像机旋转时有平滑的跟随感，不会生硬地瞬间旋转
    CameraBoom->bEnableCameraRotationLag = true;
    CameraBoom->CameraRotationLagSpeed = 10.0f;  // 延迟速度，值越小延迟越大
    // 使用Pawn控制器的旋转（意味着摄像机跟随鼠标旋转）
    CameraBoom->bUsePawnControlRotation = true;
    // 弹簧臂的世界位置偏移（这里Z轴+50，让摄像机稍微高于角色）
    CameraBoom->SocketOffset = FVector(0.0f, 0.0f, 50.0f);
    // 碰撞检测 — 摄像机碰到墙壁时自动拉近（防止穿墙）
    CameraBoom->bDoCollisionTest = true;

    // 创建跟随摄像机 — 命名 "FollowCamera"
    FollowCamera = CreateDefaultSubobject<UCameraComponent>(TEXT("FollowCamera"));
    // 把摄像机附着到弹簧臂末端（弹簧臂的Socket叫 USpringArmComponent::SocketName）
    FollowCamera->SetupAttachment(CameraBoom, USpringArmComponent::SocketName);
    // 不直接使用Pawn控制器的旋转（让弹簧臂处理旋转，摄像机只负责拍摄）
    FollowCamera->bUsePawnControlRotation = false;

    // 初始化目标臂长（用于平滑缩放插值）
    TargetBoomLength = DefaultBoomLength;

    // ===================================================================
    // 角色移动设置
    // ===================================================================
    // bUseControllerRotationYaw 决定角色的朝向是否跟着控制器（鼠标）旋转
    // false = 角色不自动转向鼠标方向（常见于动作游戏）
    // true  = 角色始终面向鼠标方向（常见于射击游戏）
    bUseControllerRotationYaw = false;

    // 获取角色移动组件（ACharacter自带）并配置
    UCharacterMovementComponent* MovementComp = GetCharacterMovement();
    // 角色朝向会自动跟随加速度方向（即移动方向）
    // 这意味着按W键前进时角色会自动面向前方
    MovementComp->bOrientRotationToMovement = true;
    // 旋转速度 — 值越大，角色转身越快
    MovementComp->RotationRate = FRotator(0.0f, 540.0f, 0.0f); // Yaw轴每秒540度
    // 默认步行速度
    MovementComp->MaxWalkSpeed = WalkSpeed;
    // 跳跃Z轴初速度
    MovementComp->JumpZVelocity = JumpVelocity;

    // ===================================================================
    // 初始化装备槽（确保所有槽位都有默认值 nullptr）
    // ===================================================================
    // 把所有槽位初始化为 nullptr（空装备）
    // FindOrAdd 如果Key不存在则添加并返回引用
    EquipmentSlots.FindOrAdd(EEquipmentSlot::Weapon) = nullptr;
    EquipmentSlots.FindOrAdd(EEquipmentSlot::Helmet) = nullptr;
    EquipmentSlots.FindOrAdd(EEquipmentSlot::Armor)  = nullptr;
    EquipmentSlots.FindOrAdd(EEquipmentSlot::Boots)  = nullptr;

    // ===================================================================

// ── Tick ──
// 每帧调用一次
// DeltaTime = 上一帧到本帧的时间间隔（秒），例如60FPS时约为0.0167秒
void ADungeonCharacter::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);

    // ── 体力自动恢复 ──
    // 体力会随着时间缓慢恢复（但不是战斗中）
    UpdateStamina(DeltaTime);

    // ── 摄像机平滑缩放 ──
    // FMath::FInterpTo 是UE的"平滑插值到目标值"函数
    // 参数：当前值, 目标值, DeltaTime, 插值速度
    // 这样滚轮缩放不会"跳变"，而是平滑过渡
    float NewLength = FMath::FInterpTo(
        CameraBoom->TargetArmLength,  // 当前长度
        TargetBoomLength,             // 目标长度
        DeltaTime,                    // 帧间隔
        8.0f                          // 插值速度（值越大过渡越快）
    );
    CameraBoom->TargetArmLength = NewLength;
}

// ── SetupPlayerInputComponent ──
// 这个函数在游戏开始时被调用，用于绑定按键输入
// 所有的输入绑定都在这里完成
void ADungeonCharacter::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent)
{
    Super::SetupPlayerInputComponent(PlayerInputComponent);

    // ===================================================================
    // 移动绑定
    // ===================================================================
    // 绑定轴向输入（Axis）— 用于持续的输入值（如按住W键持续前进）
    // BindAxis 参数1: 输入名称（在Project Settings → Input中定义）
    // BindAxis 参数2: 回调对象（this = ADungeonCharacter自己）
    // BindAxis 参数3: 回调函数指针
    PlayerInputComponent->BindAxis("MoveForward", this, &ADungeonCharacter::MoveForward);
    PlayerInputComponent->BindAxis("MoveRight",   this, &ADungeonCharacter::MoveRight);
    // 鼠标水平移动 → 旋转Yaw（左右看）
    PlayerInputComponent->BindAxis("LookRight",   this, &ADungeonCharacter::AddControllerYawInput);
    // 鼠标垂直移动 → 旋转Pitch（上下看）
    PlayerInputComponent->BindAxis("LookUp",      this, &ADungeonCharacter::AddControllerPitchInput);
    // 鼠标滚轮 → 摄像机缩放
    PlayerInputComponent->BindAxis("ZoomCamera",  this, &ADungeonCharacter::ZoomCamera);

    // ===================================================================
    // 动作绑定（Action）— 用于按下/释放的离散事件
    // ===================================================================
    // 跳跃 — 空格键按下和释放
    // IE_Pressed  = 按下时调用
    // IE_Released = 释放时调用
    PlayerInputComponent->BindAction("Jump", IE_Pressed,  this, &ADungeonCharacter::Jump);
    PlayerInputComponent->BindAction("Jump", IE_Released, this, &ACharacter::StopJumping);

    // 闪避 — Shift键按下
    PlayerInputComponent->BindAction("Dodge", IE_Pressed, this, &ADungeonCharacter::PerformDodge);

    // 轻攻击 — 鼠标左键按下
    PlayerInputComponent->BindAction("LightAttack", IE_Pressed, this, &ADungeonCharacter::PerformLightAttack);

    // 重攻击 — 鼠标右键按下（点击）
    PlayerInputComponent->BindAction("HeavyAttack", IE_Pressed, this, &ADungeonCharacter::PerformHeavyAttack);

    // 格挡 — 鼠标右键按下/释放（注意和重攻击的区分：按住=格挡，点击=重攻击）
    // 实际实现中需要区分"按下持续"和"快速点击"。这里先在代码层面分别绑定
    // 我们通过计时器来判断："如果右键被按住超过0.2秒，则进入格挡状态"
    PlayerInputComponent->BindAction("Block", IE_Pressed,  this, &ADungeonCharacter::StartBlocking);
    PlayerInputComponent->BindAction("Block", IE_Released, this, &ADungeonCharacter::StopBlocking);

    // 技能 — Q/E/R/F 键
    PlayerInputComponent->BindAction("Skill1", IE_Pressed, this, &ADungeonCharacter::UseSkill, 0);
    // 注意：BindAction 不支持直接传参数的方式。上面的写法在旧版本中有效。
    // UE5中推荐使用 EnhancedInput，这里为教学需要展示传统输入方式
    // 实际项目中建议使用 Enhanced Input Actions（详见第7章输入系统）

    // 冲刺 — 左Shift（区分：按住Shift=冲刺，单次按Shift=闪避）
    PlayerInputComponent->BindAction("Sprint", IE_Pressed,  this, &ADungeonCharacter::StartSprint);
    PlayerInputComponent->BindAction("Sprint", IE_Released, this, &ADungeonCharacter::StopSprint);
}

// ── MoveForward ──
// 处理前后移动输入
// Value: 正数=前进(W), 负数=后退(S), 0=无输入
void ADungeonCharacter::MoveForward(float Value)
{
    // 如果当前正在攻击或处于无法移动的状态，忽略移动输入
    if (CurrentState == ECharacterState::Attacking ||
        CurrentState == ECharacterState::Stunned ||
        CurrentState == ECharacterState::Dead)
    {
        return;  // 提前返回，不处理移动
    }

    // FRotationMatrix 将旋转信息转换为旋转矩阵
    // GetControlRotation() 获取控制器的旋转（即摄像机/鼠标的朝向）
    // GetUnitAxis(EAxis::X) 取旋转矩阵的X轴单位向量（即"前方"方向）
    // 这样计算出的移动方向是"基于摄像机朝向的前方"，而不是"角色的前方"
    const FRotator YawRotation(0.0f, GetControlRotation().Yaw, 0.0f);
    const FVector Direction = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);

    // AddMovementInput 让角色沿指定方向移动
    // 参数1: 世界空间中的方向向量
    // 参数2: 缩放值（1.0 = 全速，0.5 = 半速）
    AddMovementInput(Direction, Value);
}

// ── MoveRight ──
// 处理左右移动输入
// Value: 正数=右移(D), 负数=左移(A), 0=无输入
void ADungeonCharacter::MoveRight(float Value)
{
    // 同样检查是否可以移动
    if (CurrentState == ECharacterState::Attacking ||
        CurrentState == ECharacterState::Stunned ||
        CurrentState == ECharacterState::Dead)
    {
        return;
    }

    // 取控制器的Y轴（右方向），然后沿该方向移动
    const FRotator YawRotation(0.0f, GetControlRotation().Yaw, 0.0f);
    const FVector Direction = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
    AddMovementInput(Direction, Value);
}

// ── PerformDodge ──
// 执行闪避动作
void ADungeonCharacter::PerformDodge()
{
    // ── 条件检查 ──
    // 1. 不能在攻击、死亡、硬直状态闪避
    if (CurrentState == ECharacterState::Attacking ||
        CurrentState == ECharacterState::Dead ||
        CurrentState == ECharacterState::Stunned)
    {
        return;
    }

    // 2. 检查体力是否足够
    if (CurrentStamina < DodgeStaminaCost)
    {
        // 体力不足，无法闪避（可以在这里播放提示音）
        UE_LOG(LogTemp, Warning, TEXT("体力不足，无法闪避！"));
        return;
    }

    // 3. 检查冷却时间
    float CurrentTime = GetWorld()->GetTimeSeconds();
    if (CurrentTime - LastDodgeTime < DodgeCooldown)
    {
        return;  // 冷却中，不能闪避
    }

    // ── 执行闪避 ──
    // 消耗体力
    CurrentStamina = FMath::Clamp(CurrentStamina - DodgeStaminaCost, 0.0f, GetMaxStamina());

    // 记录闪避时间（用于冷却判定）
    LastDodgeTime = CurrentTime;

    // 切换状态到闪避
    SetCharacterState(ECharacterState::Dodging);

    // 进入无敌状态
    bIsInvincible = true;

    // 设置定时器，在无敌帧结束后关闭无敌
    // FTimerHandle 是一个"定时器句柄"，可以用来取消还未执行的定时器
    FTimerHandle InvincibilityTimerHandle;
    GetWorld()->GetTimerManager().SetTimer(
        InvincibilityTimerHandle,             // 定时器句柄
        this,                                  // 回调对象
        &ADungeonCharacter::DisableInvincibility, // 回调函数（成员函数指针）
        DodgeInvincibleDuration,              // 延迟时间（秒）
        false                                  // 是否循环（false=只执行一次）
    );

    // 让角色向前冲刺
    // 获取角色当前面朝方向
    FVector DodgeDirection = GetActorForwardVector();
    // LaunchCharacter 让角色沿指定方向"弹射"
    // 参数1: 速度向量（世界空间）
    // 参数2: 是否覆盖XY速度（true=完全覆盖）
    // 参数3: 是否覆盖Z速度（false=保留原有的垂直速度，即跳跃不受影响）
    LaunchCharacter(DodgeDirection * DodgeDistance, true, false);
}

    SetCharacterState(ECharacterState::Idle);

// 鼠标滚轮缩放摄像机

void ADungeonCharacter::ZoomCamera(float Value)
{
    // 忽略微小的输入（滚轮在某些鼠标上可能轻微抖动）
    if (FMath::Abs(Value) < 0.01f)
    {
        return;
    }

    // 计算新的目标臂长
    // Value为正时，TargetBoomLength减小（拉近）
    // Value为负时，TargetBoomLength增大（拉远）
    TargetBoomLength = FMath::Clamp(
        TargetBoomLength - Value * ZoomSpeed,  // 新长度
        MinBoomLength,                          // 最小值约束
        MaxBoomLength                           // 最大值约束
    );
}

// ── ApplyDamage ──
// 角色受到伤害的处理
// RawDamage: 未经过防御力削减的原始伤害
// DamageCauser: 造成伤害的来源（用于事件广播）
void ADungeonCharacter::ApplyDamage(float RawDamage, AActor* DamageCauser)
{
    // 如果已经死亡，不再处理伤害
    if (CurrentState == ECharacterState::Dead)
    {
        return;
    }

    // 如果处于无敌状态（闪避中），不受伤害
    if (bIsInvincible)
    {
        return;
    }

    // ── 格挡减伤 ──
    float ActualDamage = RawDamage;
    if (bIsBlocking)
    {
        // 格挡时减少70%的伤害
        ActualDamage *= 0.3f;
        // 格挡消耗体力（与伤害量成正比，但有最小值）
        float StaminaCost = FMath::Max(ActualDamage * 0.5f, 5.0f);
        CurrentStamina = FMath::Clamp(CurrentStamina - StaminaCost, 0.0f, GetMaxStamina());

        UE_LOG(LogTemp, Log, TEXT("格挡成功！原伤害: %f → 实际伤害: %f"), RawDamage, ActualDamage);
    }

    // ── 防御力减伤 ──
    // 伤害公式：实际伤害 = 原始伤害 - 防御力 * 0.5
    // 但最少造成1点伤害（保底，防止0伤）
    float Defense = GetTotalDefense();
    ActualDamage = FMath::Max(ActualDamage - Defense * 0.5f, 1.0f);

    // ── 扣血 ──
    CurrentHealth = FMath::Clamp(CurrentHealth - ActualDamage, 0.0f, GetMaxHealth());

    UE_LOG(LogTemp, Log, TEXT("%s 受到 %f 点伤害，剩余生命: %f"),
        *GetName(), ActualDamage, CurrentHealth);

    // ── 检查死亡 ──
    if (CurrentHealth <= 0.0f)
    {
        Die();
    }
    else
    {
        // 进入短暂硬直（被击中后的后摇）
        SetCharacterState(ECharacterState::Stunned);
        // 0.3秒后自动恢复到Idle
        FTimerHandle StunTimerHandle;
        GetWorld()->GetTimerManager().SetTimer(
            StunTimerHandle,
            this,
            &ADungeonCharacter::DieInternal, // 这里应该是一个恢复状态的函数
            0.3f,
            false
        );
    }
}

// ── GetExperienceToNextLevel ──
// 计算升级所需经验值
// 公式：100 * 等级^2 * 0.5，使得越往后升级越难
int32 ADungeonCharacter::GetExperienceToNextLevel() const
{
    // FMath::Square 计算平方
    return FMath::FloorToInt(100.0f * FMath::Square(CharacterLevel) * 0.5f);
}

// ── AddExperience ──
// 添加经验值，并检查是否可以升级
void ADungeonCharacter::AddExperience(int32 Amount)
{
    if (CharacterLevel >= 99)
    {
        return;  // 满级（如果设了上限）不再获得经验
    }

    CurrentExperience += Amount;

    UE_LOG(LogTemp, Log, TEXT("获得 %d 经验值！当前经验: %d / %d"),
        Amount, CurrentExperience, GetExperienceToNextLevel());

    // 检查是否可以升级（可能连升多级）
    while (CurrentExperience >= GetExperienceToNextLevel())
    {
        LevelUp();
    }
}

// ── LevelUp ──
// 升级处理
void ADungeonCharacter::LevelUp()
{
    // 扣除升级所需经验（而不是清空，这样可以保留溢出经验）
    CurrentExperience -= GetExperienceToNextLevel();

    // 等级+1
    CharacterLevel++;

    // 每次升级获得3个属性点
    AvailableAttributePoints += 3;

    // 升级时回满生命和法力（常见设计）
    CurrentHealth = GetMaxHealth();
    CurrentMana = GetMaxMana();

    UE_LOG(LogTemp, Log, TEXT("🎉 升级了！当前等级: %d，可用属性点: %d"),
        CharacterLevel, AvailableAttributePoints);

    // ── 广播升级事件（让UI系统更新界面）──
    // 获取GameMode并广播事件

// ── AllocateAttributePoint ──
// 分配属性点
void ADungeonCharacter::AllocateAttributePoint(EAttributeType AttributeType)
{
    // 检查是否有可用属性点
    if (AvailableAttributePoints <= 0)
    {
        UE_LOG(LogTemp, Warning, TEXT("没有可用的属性点！"));
        return;
    }

    // 消耗1个属性点
    AvailableAttributePoints--;

    // 根据类型增加对应属性
    switch (AttributeType)
    {
    case EAttributeType::Strength:
        BaseAttributes.Strength += 1.0f;     // 力量+1
        break;
    case EAttributeType::Agility:
        BaseAttributes.Agility += 1.0f;      // 敏捷+1
        break;
    case EAttributeType::Vitality:
        BaseAttributes.Vitality += 1.0f;     // 体质+1
        // 体质变化会影响最大生命，需要按比例调整当前生命
        // 例如：原来满血100，最大生命变成110，当前生命也变成110
        {
            float OldMaxHealth = GetMaxHealth();
            // 体质已经+1了，需要重新计算最大生命然后同步
            // 简化为：体质变化时当前生命按比例调整
        }
        break;
    case EAttributeType::Intelligence:
        BaseAttributes.Intelligence += 1.0f;  // 智力+1
        break;
    case EAttributeType::Endurance:
        BaseAttributes.Endurance += 1.0f;     // 耐力+1
        break;
    }

    UE_LOG(LogTemp, Log, TEXT("分配了1点属性到[%d]，剩余属性点: %d"),
        static_cast<int32>(AttributeType), AvailableAttributePoints);
}

// ── EquipItem ──
// 装备一件物品到对应的装备槽
bool ADungeonCharacter::EquipItem(UDungeonEquipment* Equipment)
{
    if (Equipment == nullptr)
    {
        UE_LOG(LogTemp, Warning, TEXT("尝试装备空的装备！"));
        return false;
    }

    // 获取装备应该放在哪个槽位
    EEquipmentSlot Slot = Equipment->GetSlotType();

    // 如果该槽位已有装备，先卸下旧装备放回背包
    if (EquipmentSlots[Slot] != nullptr)
    {
        UDungeonEquipment* OldEquipment = EquipmentSlots[Slot];
        AddToInventory(OldEquipment);  // 旧装备放回背包
    }

    // 从背包中移除新装备
    RemoveFromInventory(Equipment);

    // 把新装备放入装备槽
    EquipmentSlots[Slot] = Equipment;

    UE_LOG(LogTemp, Log, TEXT("装备了 %s 到 %d 槽位"),
        *Equipment->GetName(), static_cast<int32>(Slot));

    return true;
}

// ── UnequipItem ──
// 卸下指定槽位的装备
bool ADungeonCharacter::UnequipItem(EEquipmentSlot Slot)
{
    if (EquipmentSlots[Slot] == nullptr)
    {
        UE_LOG(LogTemp, Warning, TEXT("该槽位没有装备！"));
        return false;
    }

    UDungeonEquipment* Equipment = EquipmentSlots[Slot];

    // 检查背包是否已满
    if (Inventory.Num() >= MaxInventorySize)
    {
        UE_LOG(LogTemp, Warning, TEXT("背包已满，无法卸下装备！"));
        return false;
    }

    // 放回背包
    AddToInventory(Equipment);

    // 清空装备槽
    EquipmentSlots[Slot] = nullptr;

    UE_LOG(LogTemp, Log, TEXT("卸下了 %d 槽位的装备"), static_cast<int32>(Slot));
    return true;
}

// ── AddToInventory ──
// 将物品添加到背包
bool ADungeonCharacter::AddToInventory(UDungeonItemBase* Item)
{
    if (Item == nullptr)
    {
        return false;
    }

    // 检查背包是否已满
    if (Inventory.Num() >= MaxInventorySize)
    {
        UE_LOG(LogTemp, Warning, TEXT("背包已满！无法拾取 %s"), *Item->GetName());
        return false;
    }

    // 添加到数组
    Inventory.Add(Item);

    UE_LOG(LogTemp, Log, TEXT("拾取了 %s，背包空间: %d/%d"),
        *Item->GetName(), Inventory.Num(), MaxInventorySize);

    return true;
}

// ── RemoveFromInventory ──
// 从背包中移除物品
bool ADungeonCharacter::RemoveFromInventory(UDungeonItemBase* Item)
{
    if (Item == nullptr)
    {
        return false;
    }

    // Remove 返回移除的元素数量（0或1，因为背包中不会重复存放同一个对象）
    int32 RemovedCount = Inventory.Remove(Item);
    return RemovedCount > 0;
}

// ── SetCharacterState ──
// 切换角色状态
void ADungeonCharacter::SetCharacterState(ECharacterState NewState)
{
    // 如果状态没有变化，不做任何事
    if (CurrentState == NewState)
    {
        return;
    }

    // ── 退出旧状态的处理 ──
    switch (CurrentState)
    {
    case ECharacterState::Blocking:
        bIsBlocking = false;  // 退出格挡状态
        break;
    case ECharacterState::Attacking:
        bIsAttacking = false; // 退出攻击状态
        break;
    default:
        break;
    }

    // ── 记录旧状态用于日志 ──
    ECharacterState OldState = CurrentState;

    // ── 设置新状态 ──
    CurrentState = NewState;

    // ── 进入新状态的处理 ──
    switch (NewState)
    {
    case ECharacterState::Idle:
        // 空闲状态：允许移动、允许攻击
        break;
    case ECharacterState::Moving:
        // 移动状态
        break;
    case ECharacterState::Attacking:
        bIsAttacking = true;
        // 攻击期间降低移动速度
        GetCharacterMovement()->MaxWalkSpeed = WalkSpeed * 0.3f;  // 降低到30%
        break;
    case ECharacterState::Blocking:
        bIsBlocking = true;
        // 格挡期间降低移动速度
        GetCharacterMovement()->MaxWalkSpeed = WalkSpeed * 0.5f;
        break;
    case ECharacterState::Dodging:
        // 闪避中（移动由 LaunchCharacter 控制，不修改 MaxWalkSpeed）
        break;
    case ECharacterState::Stunned:
        // 硬直期间不能移动
        GetCharacterMovement()->MaxWalkSpeed = 0.0f;
        break;
    case ECharacterState::Dead:
        // 死亡处理
        GetCharacterMovement()->MaxWalkSpeed = 0.0f;
        GetCharacterMovement()->DisableMovement();  // 彻底禁用移动
        break;
    }

    // 日志输出状态切换
    UE_LOG(LogTemp, Verbose, TEXT("状态切换: %d → %d"),
        static_cast<int32>(OldState), static_cast<int32>(NewState));
}

// ── UpdateStamina ──
// 每帧更新体力（自动恢复）
void ADungeonCharacter::UpdateStamina(float DeltaTime)
{
    // 体力在非闪避、非冲刺状态下自动恢复
    // 每秒恢复最大体力的5%
    float StaminaRecoveryRate = GetMaxStamina() * 0.05f;  // 每秒恢复量

    // 如果正在格挡，体力恢复速度减半
    if (bIsBlocking)
    {
        StaminaRecoveryRate *= 0.5f;
    }

    // 累加恢复量（乘以DeltaTime是因为这是每帧更新，需要按时间缩放）
    CurrentStamina = FMath::Clamp(
        CurrentStamina + StaminaRecoveryRate * DeltaTime,  // 新值
        0.0f,                                               // 最小值
        GetMaxStamina()                                     // 最大值
    );
}

// ── Die ──
// 玩家角色死亡
void ADungeonCharacter::Die()
{
    if (CurrentState == ECharacterState::Dead)
    {
        return;  // 防止重复调用
    }

    // 切换到死亡状态
    SetCharacterState(ECharacterState::Dead);

    // 禁用输入（玩家不能再操控）
    // DisableInput 需要传入 PlayerController
    if (APlayerController* PC = Cast<APlayerController>(GetController()))
    {
        DisableInput(PC);
    }

    // 禁用碰撞（角色倒地后不应再阻挡其他角色）
    GetCapsuleComponent()->SetCollisionEnabled(ECollisionEnabled::NoCollision);

    // 播放死亡动画
    // PlayAnimMontage(DeathMontage);

    // 通知 GameMode 玩家死亡
    // 实际项目中需要通过委托通知
    UE_LOG(LogTemp, Log, TEXT("玩家角色已死亡！"));

    // ── 实际项目中的死亡流程 ──
    // 1. 播放死亡动画（~2秒）
    // 2. 显示"你死了"界面
    // 3. 提供选项：从存档加载 / 返回主菜单
}

// ── DieInternal ──
// 死亡内部处理（委托回调、清理资源等）
void ADungeonCharacter::DieInternal()
{
    // 清理所有定时器
    GetWorld()->GetTimerManager().ClearAllTimersForObject(this);

    // 禁用Tick（死人不更新）
    SetActorTickEnabled(false);
}
```

## 2.7 常见问题排查

### 问题1：角色无法移动
**原因**：输入映射未配置或名称不匹配
**解决**：检查 Project Settings → Input 中的 Axis/Action 映射名称是否与代码中的一致

### 问题2：摄像机看不到角色
**原因**：弹簧臂的 SocketOffset 或 TargetArmLength 设置不当
**解决**：调整 CameraBoom 参数，确保摄像机在角色身后上方

### 问题3：角色不跟随鼠标旋转
**原因**：`bUseControllerRotationYaw` 设置问题
**解决**：这个值应该是 `false`（动作游戏），旋转由 `bOrientRotationToMovement` 控制

### 问题4：攻击时角色还能移动
**原因**：移动函数没有正确检查状态
**解决**：在 `MoveForward` 和 `MoveRight` 中添加状态检查

### 问题5：编译报错找不到 Module
**原因**：Build.cs 中缺少模块依赖
**解决**：检查 `DungeonCrawler.Build.cs` 中的模块列表是否完整
