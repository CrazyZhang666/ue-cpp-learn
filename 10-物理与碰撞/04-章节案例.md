## AProjectile 弹丸基类

`AProjectile` 是所有弹射武器的基类。它使用 `UProjectileMovementComponent` 控制弹道，提供完整的碰撞检测和生命周期管理。

### 头文件：Projectile.h

```cpp
// ==================== Projectile.h ====================
// 弹丸基类 —— 所有弹射武器（手榴弹、火箭弹、箭矢等）的基类
// 路径：Source/MyGame/Public/Weapons/Projectile.h

#pragma once  // 防止头文件被重复包含

#include "CoreMinimal.h"                       // UE核心类型的头文件（FVector, FString等）
#include "GameFramework/Actor.h"              // AActor基类
#include "Projectile.generated.h"             // 由UHT自动生成的反射系统头文件

// 前向声明（不需要完整引入头文件，减少编译依赖）
class UProjectileMovementComponent;           // 弹丸运动组件
class USphereComponent;                       // 球体碰撞组件
class UParticleSystem;                        // 粒子系统（爆炸特效等）
class USoundBase;                            // 音效资源
class UDamageType;                           // 伤害类型

/**
 * AProjectile —— 弹丸基类
 *
 * 职责：
 * - 使用 UProjectileMovementComponent 控制弹道运动
 * - 使用 SphereComponent + OnComponentHit 检测碰撞
 * - 碰撞时触发爆炸逻辑（由子类重写）
 * - 自动在若干秒后销毁以清理世界
 */
UCLASS()  // 声明这是一个UE反射类
class MYGAME_API AProjectile : public AActor   // MYGAME_API 是模块导出宏
{
    GENERATED_BODY()  // 必须的UE反射系统宏，为类生成反射代码

public:
    // ==================== 构造函数 ====================
    // 头文件中只声明，实现在cpp中
    AProjectile();

protected:
    // ==================== 组件 ====================

    // 球体碰撞组件 —— 用于碰撞检测
    // VisibleAnywhere = 在编辑器中可见
    // BlueprintReadOnly = 蓝图只读（不能修改，但可以读取）
    // Category = 在编辑器的分类标签
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Components")
    USphereComponent* CollisionComponent;
    // 为什么用球体？
    // 1. 性能最优（球体碰撞检测最简单）
    // 2. 形状均匀（弹丸从哪个方向碰撞都一样）
    // 3. 适合大多数弹丸（子弹、手榴弹、火箭弹近似球形）


    // 弹丸运动组件 —— 控制弹道飞行
    // UProjectileMovementComponent 是UE内置的弹道控制组件
    // 它自动处理：初始速度、重力影响、弹跳逻辑、子步精度
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Components")
    UProjectileMovementComponent* ProjectileMovement;

    // 静态网格体（弹丸的视觉外观）
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Components")
    UStaticMeshComponent* ProjectileMesh;


    // ==================== 弹道参数 ====================

    // 初始速度（弹丸刚发射时的速度，厘米/秒）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory")
    float InitialSpeed = 3000.0f;
    // 默认3000 cm/s = 30 m/s —— 一个比较慢的弹丸速度
    // 实际子弹速度可能到 50000~100000 cm/s

    // 最大速度（弹丸能达到的最高速度，0 = 无限制）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory")
    float MaxSpeed = 0.0f;

    // 是否受重力影响
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory")
    bool bShouldBounce = false;
    // false → 抛物线弹道（手榴弹、迫击炮弹）
    // true  → 直线弹道（子弹、激光）——这里重力关闭

    // 重力缩放（0=无重力, 1=正常重力, 2=双倍重力）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory",
        meta = (EditCondition = "bShouldBounce==false"))
        // EditCondition 条件编辑：只在 bShouldBounce==false 时才能在编辑器中编辑此属性
    float GravityScale = 1.0f;
    // 默认地球重力 = 980 cm/s² (Z轴向下)

    // 弹跳次数（0=不弹跳, -1=无限弹跳）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory")
    int32 BounceCount = 0;

    // 弹跳速度损失（每次弹跳后剩余速度的比例）
    // 0.6 = 每次弹跳保留60%的速度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Trajectory",
        meta = (EditCondition = "BounceCount!=0", ClampMin = "0.0", ClampMax = "1.0"))
        // ClampMin / ClampMax 确保值在 0~1 之间
    float BounceVelocityLoss = 0.6f;


    // ==================== 碰撞检测参数 ====================

    // 碰撞检测的扫描半径（用于 SphereSweepMulti）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Collision")
    float CollisionCheckRadius = 10.0f;

    // 碰撞时是否触发爆炸（由子类实现具体爆炸逻辑）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Collision")
    bool bExplodeOnHit = true;


    // ==================== 生命周期 ====================

    // 弹丸的最大存活时间（秒），到期后自动销毁
    // 防止弹丸永远飞行占用资源
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Lifecycle")
    float LifeSpan = 10.0f;

    // 伤害数值（基类中定义，子类可以覆盖）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Damage")
    float BaseDamage = 50.0f;


    // ==================== 特效资源配置 ====================

    // 飞行时的尾迹粒子（如烟雾拖尾）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Effects")
    UParticleSystem* TrailParticle;

    // 爆炸粒子效果
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Effects")
    UParticleSystem* ExplosionParticle;

    // 爆炸音效
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Projectile|Effects")
    USoundBase* ExplosionSound;


    // ==================== 虚函数（子类可重写） ====================

    // 发射弹丸 —— 设置初始速度并将弹丸投入世界
    // BlueprintCallable = 可在蓝图中调用此函数
    // virtual = 子类可以重写以自定义发射行为
    UFUNCTION(BlueprintCallable, Category = "Projectile")
    virtual void Launch(const FVector& LaunchVelocity);

    // 弹丸击中某物时调用
    // 参数：Hit —— 碰撞的详细信息（命中点、法线、命中Actor等）
    UFUNCTION()
    virtual void OnProjectileHit(
        UPrimitiveComponent* HitComponent,   // 碰撞体组件
        AActor* OtherActor,                  // 碰撞的对方Actor
        UPrimitiveComponent* OtherComp,      // 对方的具体组件
        FVector NormalImpulse,               // 碰撞冲量方向
        const FHitResult& Hit               // 碰撞命中结果
    );

    // 执行爆炸逻辑 —— 子类必须重写
    // BlueprintImplementableEvent = 可以在蓝图中重写（纯蓝图事件）
    UFUNCTION(BlueprintImplementableEvent, Category = "Projectile")
    void Explode(const FHitResult& Hit);

    // C++版本的爆炸逻辑 —— 也可以用 virtual 让C++子类重写
    virtual void Explode_CPP(const FHitResult& Hit);

    // 预测弹丸的飞行轨迹
    // 返回路径上的一系列点（用于UI预览弹道线）
    UFUNCTION(BlueprintCallable, Category = "Projectile")
    TArray<FVector> PredictTrajectory(
        const FVector& StartLocation,   // 模拟的起始位置
        const FVector& StartVelocity,   // 模拟的初始速度
        float TimeStep,                 // 每步的时间间隔（秒）
        int32 MaxSteps                  // 最大模拟步数
    ) const;
};
```

### 实现文件：Projectile.cpp

```cpp
// ==================== Projectile.cpp ====================
// 弹丸基类的实现
// 路径：Source/MyGame/Private/Weapons/Projectile.cpp

#include "Projectile.h"                    // 自己的头文件
#include "Components/SphereComponent.h"    // USphereComponent 类定义
#include "Components/StaticMeshComponent.h" // UStaticMeshComponent 类定义
#include "GameFramework/ProjectileMovementComponent.h"  // UProjectileMovementComponent 类定义
#include "Kismet/GameplayStatics.h"        // UGameplayStatics 工具函数库
#include "Particles/ParticleSystem.h"      // UParticleSystem 类定义
#include "Sound/SoundBase.h"              // USoundBase 类定义
#include "Engine/World.h"                  // UWorld 类定义（包含射线检测等世界API）
#include "DrawDebugHelpers.h"             // DrawDebugLine/Sphere 等调试可视化

// ==================== 构造函数 ====================
AProjectile::AProjectile()
{
    // 步骤1：设置基本属性
    // 允许每帧Tick（ProjectileMovementComponent需要Tick来更新位置）
    PrimaryActorTick.bCanEverTick = true;
    // 注释掉可以略微提升性能（ProjectileMovement会自行处理更新）
    // 但保留它可以让你在Tick中添加自定义逻辑

    // 设置网络同步（多人游戏）
    // bReplicates = true;  // 需要网络支持时取消注释


    // 步骤2：创建并配置球体碰撞组件
    // 球体碰撞组件作为根组件——弹丸的位置由球体决定
    CollisionComponent = CreateDefaultSubobject<USphereComponent>(TEXT("CollisionComponent"));
    RootComponent = CollisionComponent;  // 设为根组件
    // CreateDefaultSubobject 只应在构造函数中调用
    // 它向UE的对象系统注册这个子对象

    CollisionComponent->SetSphereRadius(15.0f);  // 半径15cm —— 一个小球大小
    // 碰撞模式：查询和物理
    // 选择 QueryAndPhysics 而不是 QueryOnly 因为：
    // - 需要物理反弹（如果开启了弹跳）
    // - 需要碰撞事件（OnComponentHit）
    CollisionComponent->SetCollisionEnabled(ECollisionEnabled::QueryAndPhysics);

    // 设置碰撞预设为 "BlockAllDynamic"
    // 这意味着：阻挡所有动态物体（Pawn, PhysicsBody, Vehicle等）
    // 但 WorldStatic 的响应取决于具体预设
    CollisionComponent->SetCollisionProfileName(FName(TEXT("BlockAllDynamic")));
    // BlockAllDynamic预设的特点：
    // - ObjectType = WorldStatic（自身标记）
    // - Block响应：WorldDynamic, Pawn, PhysicsBody, Vehicle, Destructible
    // - Ignore响应：通常不Ignore任何东西（除了Visibility/Camera）
    //
    // ⚠️ 如果你想让弹丸也阻挡WorldStatic（墙壁等），使用 "BlockAll"
    // CollisionComponent->SetCollisionProfileName(FName(TEXT("BlockAll")));


    // 🔑 至关重要：开启碰撞事件通知
    // 如果不设置这个，即使碰撞发生了，OnComponentHit也不会被触发！
    CollisionComponent->SetNotifyRigidBodyCollision(true);
    // SetNotifyRigidBodyCollision(true) 告诉引擎：
    // "每当这个组件发生物理碰撞时，请触发OnComponentHit委托"


    // 步骤3：创建弹丸运动组件
    ProjectileMovement = CreateDefaultSubobject<UProjectileMovementComponent>(TEXT("ProjectileMovement"));
    // UProjectileMovementComponent 是UE专门为弹丸运动设计的内置组件
    // 它处理：
    // - 每帧更新位置（基于当前速度 × DeltaTime）
    // - 重力加速度（如果启用了重力）
    // - 弹跳逻辑（碰到障碍物后反弹）
    // - 物理子步（高速运动不穿墙）

    // 运动组件需要引用碰撞组件来确定"什么是我的碰撞体"
    // 设置 UpdatedComponent 告诉运动组件它将更新谁的变换
    ProjectileMovement->UpdatedComponent = CollisionComponent;
    // 运动组件会自动读取CollisionComponent的位置，并在每帧写入新位置


    // 应用默认的弹道参数到ProjectileMovement组件
    ProjectileMovement->InitialSpeed = InitialSpeed;      // 初始速度 3000cm/s
    ProjectileMovement->MaxSpeed = MaxSpeed;              // 最大速度（0=无限制）
    ProjectileMovement->bShouldBounce = bShouldBounce;    // 是否弹跳
    ProjectileMovement->ProjectileGravityScale = GravityScale; // 重力缩放
    // ProjectileGravityScale = 0  → 无重力（直线飞行）
    // ProjectileGravityScale = 1  → 标准地球重力
    // ProjectileGravityScale = 2  → 双倍重力

    ProjectileMovement->Bounciness = BounceVelocityLoss;  // 弹跳速度保留比例
    // Bounciness = 1.0 → 完全弹性碰撞（无能量损失，弹到同样的高度）
    // Bounciness = 0.6 → 每次弹跳保留60%的速度
    // Bounciness = 0.0 → 完全不弹


    // 步骤4：创建弹丸的视觉网格体
    ProjectileMesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("ProjectileMesh"));
    // 将网格体挂载在球体碰撞下面
    ProjectileMesh->SetupAttachment(CollisionComponent);
    // SetupAttachment：将ProjectileMesh作为CollisionComponent的子节点
    // 这样当CollisionComponent移动时，ProjectileMesh会自动跟随

    // 视觉网格体不需要有碰撞（碰撞由CollisionComponent负责）
    // 这是"碰撞与视觉分离"的设计模式——性能好、灵活性高
    ProjectileMesh->SetCollisionEnabled(ECollisionEnabled::NoCollision);
    // 如果启用碰撞会产生"双碰撞"——ProjectileMesh和CollisionComponent各自检测碰撞，浪费双倍性能


    // 步骤5：设置生命周期
    // 弹丸发出后10秒还没碰到任何东西，就自动销毁
    // 防止弹丸飞出世界边界后永远占用资源
    InitialLifeSpan = LifeSpan;
    // AActor::InitialLifeSpan 是引擎内置属性
    // 设置为正数后，Actor会在指定的秒数后自动调用 Destroy()
}


// ==================== 发射弹丸 ====================
void AProjectile::Launch(const FVector& LaunchVelocity)
{
    UE_LOG(LogTemp, Log, TEXT("发射弹丸！速度: %s (大小: %.1f cm/s)"),
        *LaunchVelocity.ToString(),        // 速度向量
        LaunchVelocity.Size()              // 速度大小（速率）
    );

    // 步骤1：激活弹丸运动组件
    // 设置初始速度向量
    ProjectileMovement->Velocity = LaunchVelocity;
    // 运动组件会从这一帧开始使用此速度更新位置
    // Velocity是一个FVector，代表每秒移动的厘米数

    // 步骤2：标记弹丸为"飞行中"
    // 这个标记可用于其他系统判断弹丸状态
    // 例如：伤害系统检查"这个弹丸是否在飞行"来决定是否造成伤害

    // 步骤3：生成飞行尾迹粒子效果（如果有设置的话）
    // 粒子效应会跟随弹丸，模拟烟雾/火焰拖尾
    if (TrailParticle)
    {
        UGameplayStatics::SpawnEmitterAttached(
            TrailParticle,                       // 粒子系统资源
            CollisionComponent,                  // 挂载到哪个组件（跟随弹丸）
            NAME_None,                           // Socket名（不需要特殊Socket）
            FVector::ZeroVector,                 // 相对位置（组件原点）
            FRotator::ZeroRotator,               // 相对旋转（不旋转）
            EAttachLocation::KeepRelativeOffset  // 保持相对偏移
        );
    }
}


// ==================== 碰撞回调 ====================
void AProjectile::OnProjectileHit(
    UPrimitiveComponent* HitComponent,
    AActor* OtherActor,
    UPrimitiveComponent* OtherComp,
    FVector NormalImpulse,
    const FHitResult& Hit)
{
    // ===== 安全检查 =====
    // 防止弹丸命中自己（虽然不太可能，但安全起见）
    if (OtherActor == this)
    {
        return;  // 忽略自碰撞
    }

    // 防止弹丸命中发射者
    // 在发射弹丸时，可以在弹丸上设置一个Instigator
    // if (OtherActor == GetInstigator()) return;

    // ===== 日志输出 =====
    UE_LOG(LogTemp, Warning,
        TEXT("弹丸命中！目标: %s | 命中位置: %s | 冲量: %.1f"),
        OtherActor ? *OtherActor->GetName() : TEXT("World"),  // 可能直接命中世界几何体
        *Hit.Location.ToString(),     // 命中点的世界坐标
        NormalImpulse.Size()          // 碰撞冲量大小
    );

    // ===== 处理命中效果 =====

    // 步骤1：生成爆炸粒子效果
    if (ExplosionParticle)
    {
        // 在命中点生成爆炸粒子
        UGameplayStatics::SpawnEmitterAtLocation(
            GetWorld(),                       // 当前世界
            ExplosionParticle,                // 粒子系统资源
            Hit.Location,                     // 生成位置（命中点）
            Hit.ImpactNormal.Rotation(),      // 旋转（让粒子"正面"朝向表面法线）
            // FVector::Rotation() 返回将X轴转到该向量方向的旋转角
            FVector(1.0f),                    // 缩放（1倍）
            true                              // 自动销毁（粒子播完后自动清理）
        );
    }

    // 步骤2：播放爆炸音效
    if (ExplosionSound)
    {
        UGameplayStatics::PlaySoundAtLocation(
            GetWorld(),
            ExplosionSound,
            Hit.Location
        );
    }

    // 步骤3：调试可视化 —— 在命中点画红球（开发期间用）
    DrawDebugSphere(
        GetWorld(),
        Hit.Location,     // 球心位置
        20.0f,            // 半径20cm
        16,               // 分段数
        FColor::Red,      // 红色
        false,            // 不永久
        3.0f              // 显示3秒
    );

    // 步骤4：执行爆炸逻辑
    if (bExplodeOnHit)
    {
        Explode_CPP(Hit);  // 调用C++版本的爆炸逻辑
    }

    // 步骤5：无论是否爆炸，弹丸都应该消失
    // 子弹/手榴弹碰到东西后就不应该继续飞行了
    Destroy();
    // AActor::Destroy() 会将Actor从世界中移除
    // 注意：这会同时移除所有子组件和挂载的对象
}


// ==================== C++爆炸逻辑 ====================
void AProjectile::Explode_CPP(const FHitResult& Hit)
{
    // 基类的爆炸逻辑是空的
    // 这是一个"钩子"——让子类可以添加自己的爆炸行为
    // 子类应该调用 super::Explode_CPP(Hit) 或自己完全实现

    UE_LOG(LogTemp, Log, TEXT("基类 Explode_CPP 调用 —— 子类应重写此方法实现具体爆炸逻辑"));
    // 在基类中只是输出一条日志
    // 具体爆炸逻辑由 AGrenade 等子类实现
}


// ==================== 弹道预测 ====================
TArray<FVector> AProjectile::PredictTrajectory(
    const FVector& StartLocation,
    const FVector& StartVelocity,
    float TimeStep,
    int32 MaxSteps) const
{
    // 创建一个空的路径点数组
    TArray<FVector> PathPoints;
    // 预分配内存：提前分配空间，避免后面每次Add都重新分配
    PathPoints.Reserve(MaxSteps + 1);
    // Reserve不会改变数组长度，只是预留容量


    // ---- 初始化模拟状态 ----
    FVector CurrentLocation = StartLocation;   // 当前位置
    FVector CurrentVelocity = StartVelocity;   // 当前速度
    // 重力向量（假设使用标准重力）
    float Gravity = -980.0f * GravityScale;    // UE5标准重力加速度
    // CM为单位的标准重力 ≈ -980 cm/s²
    FVector GravityVector(0.0f, 0.0f, Gravity);
    // 只在Z轴上有重力（向下）


    // ---- 第一步：加入起始点 ----
    PathPoints.Add(StartLocation);


    // ---- 逐步模拟 ----
    for (int32 Step = 0; Step < MaxSteps; ++Step)
    {
        // 计算这一步需要的时间
        float CurrentTimeStep = TimeStep;
        // 如果 TimeStep = 0.05s，则在60FPS下差不多每3帧模拟一次


        // 半隐式欧拉积分 —— 这是物理模拟中常用的位置更新方法
        //
        // 步骤A：更新速度（先于位置）
        // 速度受重力影响：v(t+dt) = v(t) + a * dt
        CurrentVelocity += GravityVector * CurrentTimeStep;
        // 如果GravityScale=1.0，CurrentVelocity.Z 每步减少 980 * 0.05 = 49 cm/s

        // 步骤B：更新位置（使用新的速度）
        // 位置 = 位置 + 速度 × 时间
        CurrentLocation += CurrentVelocity * CurrentTimeStep;
        // 如果速度是3000cm/s，每步移动 3000 * 0.05 = 150cm


        // 将该位置加入路径点数组
        PathPoints.Add(CurrentLocation);


        // 可选：做碰撞检测，如果路径会碰到障碍物就提前终止模拟
        // 这样做可以让弹道预览线在实际碰撞点处停止
        // （省略了碰撞检测部分——完整实现会在实际项目中添加）
    }

    // 返回完整的路径点数组
    // 调用者可以遍历这些点，用DrawDebugLine连接它们来画弹道预览
    return PathPoints;
}
```

## AGrenade 手榴弹

`AGrenade` 继承自 `AProjectile`，添加了抛物线弹道、延迟爆炸和范围伤害。

```cpp
 * 行为流程：
 * 1. 发射 → 初始化抛物线弹道

 * 3. 落地 → 开始倒计时（或直接计时引爆）

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Grenade|Explosion")
    float MinExplosionDamage = 20.0f;
    // 在爆炸边缘（5米处）至少受到20点伤害

    // ==================== 冲击波参数 ====================

    // 爆炸冲击波的力度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Grenade|Explosion")
    float ExplosionImpulse = 10000.0f;
    // 冲量大小：用于推动爆炸范围内的物理物体

    // 是否对物理物体施加爆炸冲击力
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Grenade|Explosion")
    bool bApplyExplosionImpulse = true;


    // ==================== 调试参数 ====================

    // 是否在编辑器中可视化爆炸范围（仅开发用）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Grenade|Debug")

    // 爆炸可视化持续时间（秒）

    float DebugDrawDuration = 5.0f;

    bool bFuseActivated = false;
```

```cpp
// ==================== Grenade.cpp ====================
// 手榴弹类实现
// 路径：Source/MyGame/Private/Weapons/Grenade.cpp

#include "Grenade.h"
#include "Components/SphereComponent.h"
#include "GameFramework/ProjectileMovementComponent.h"
#include "Kismet/GameplayStatics.h"
#include "Engine/World.h"

#include "Engine/Engine.h"        // GEngine 全局引擎指针


// ==================== 构造函数 ====================
AGrenade::AGrenade()
{
    // 步骤1：覆盖基类的弹道参数 —— 手榴弹的弹道特征

    // 初始速度设为中等（手榴弹一般不是高速弹丸）
    InitialSpeed = 1500.0f;
    // 1500 cm/s = 15 m/s —— 手榴弹抛出的合理速度

    // 手榴弹不弹跳（碰到地面就停）
    BounceCount = 0;

    // 开启重力 —— 抛物线的关键！
    bShouldBounce = false;    // 关闭弹跳模式
    GravityScale = 1.0f;      // 标准重力

    // 弹丸碰到物体后消失（不用反弹）
    BounceVelocityLoss = 0.0f;  // 碰撞后剩余速度0% —— 即：碰到就停


    // 步骤2：更新ProjectileMovement组件为手榴弹的配置
    // 构造函数中 ProjectileMovement 已经被 AProjectile 创建好了
    // 这里覆盖它的参数（而不是重新创建一个）
    if (ProjectileMovement)
    {
        ProjectileMovement->InitialSpeed = InitialSpeed;
        ProjectileMovement->bShouldBounce = false;        // 不弹跳
        ProjectileMovement->ProjectileGravityScale = 1.0f; // 标准重力
        ProjectileMovement->Bounciness = 0.0f;            // 完全不弹
        // 手榴弹的物理表现：一个受重力影响的抛物线运动物体
        // 碰到任何物体后停下（不反弹），然后开始引爆倒计时
    }

    // 步骤3：覆盖碰撞设置（如果需要与基类不同的碰撞行为）
    if (CollisionComponent)
    {
        // 确保手榴弹会与WorldStatic发生碰撞（会碰到地面/墙壁）
        // BlockAllDynamic 默认已经阻挡 WorldStatic
        // 但明确设置一下更安全：
        CollisionComponent->SetCollisionResponseToChannel(ECC_WorldStatic, ECR_Block);
        // ECR_Block = 阻挡，这意味着手榴弹会被地面和墙壁挡住
    }


    // 步骤4：覆盖伤害数值
    BaseDamage = MaxExplosionDamage;  // 基类的伤害数值设为最大爆炸伤害
}


// ==================== 重写：发射 ====================
void AGrenade::Launch(const FVector& LaunchVelocity)
{
    // 步骤1：调用基类的Launch（设置Velocity等基本操作）
    Super::Launch(LaunchVelocity);
    // Super是C++关键字，指向父类AProjectile
    // Super::Launch(LaunchVelocity) 调用 AProjectile::Launch()


    // 步骤2：根据配置决定何时开始引信倒计时
    if (!bTriggerFuseOnBounce)
    {
        // 如果不需要等落地才计时 → 发射后立即开始倒计时
        ActivateFuse();
    }
    // 如果需要等落地才计时 → 等到 OnProjectileHit 被调用时再激活


    // 步骤3：记录发射日志
    UE_LOG(LogTemp, Log,
        TEXT("手榴弹已发射！速度: %.1f cm/s | 引信时间: %.1f秒 | 落地激活: %s"),
        LaunchVelocity.Size(),
        FuseTime,
        bTriggerFuseOnBounce ? TEXT("是") : TEXT("否")
    );
}


// ==================== 重写：碰撞回调 ====================
void AGrenade::OnProjectileHit(
    UPrimitiveComponent* HitComponent,
    AActor* OtherActor,
    UPrimitiveComponent* OtherComp,
    FVector NormalImpulse,
    const FHitResult& Hit)
{

    if (ProjectileMovement)

        ProjectileMovement->StopMovementImmediately();

        // 这样手榴弹会停在碰撞点，而不是继续穿过去
    }

    // 步骤2：如果配置为"落地激活"，现在激活引信
    if (bTriggerFuseOnBounce && !bFuseActivated)
    {
        UE_LOG(LogTemp, Log, TEXT("手榴弹落地，激活引信（%.1f秒后爆炸）"), FuseTime);
        ActivateFuse();
    }

    // 步骤3：可选：根据物理材质播放不同的落地音效
    // 例如：手榴弹落在金属表面和落在草地上发出不同的声音
    if (Hit.PhysMaterial.IsValid())
    {
        // 获取物理材质做判断
        // UPhysicalMaterial* PhysMat = Hit.PhysMaterial.Get();
        // PlayBounceSound(PhysMat->SurfaceType);
    }

    // ⚠️ 注意：不要在这里调用 Super::OnProjectileHit()
    // 因为基类的 OnProjectileHit 会调用 Explode 和 Destroy()
    // 手榴弹需要等引信倒计时结束后才爆炸
    // 所以我们跳过基类的碰撞处理，在这里只做落地逻辑
}


// ==================== 激活引信 ====================
void AGrenade::ActivateFuse()
{
    // 防止重复激活（如果已经被激活了就不要再次激活）
    if (bFuseActivated)
    {
        UE_LOG(LogTemp, Warning, TEXT("引信已经激活，忽略重复激活"));
        return;  // 提前返回，不重复设置计时器
    }

    // 标记引信已激活
    bFuseActivated = true;

    // 使用GetWorldTimerManager设置一个定时器
    // 在 FuseTime 秒后自动调用 OnFuseComplete 函数
    GetWorldTimerManager().SetTimer(
        FuseTimerHandle,          // 输出：计时器的句柄（用于后续管理）
        this,                     // 回调对象（this = 当前AGrenade实例）
        &AGrenade::OnFuseComplete, // 回调函数指针
        FuseTime,                 // 延迟时间（秒）
        false                     // 是否循环（false = 只执行一次）
    );
    // SetTimer 参数完整解释：
    // 参数1 TimerHandle  → 句柄，用于取消/暂停计时器
    // 参数2 Object        → 调用回调的对象
    // 参数3 Callback       → 到时间后调用的函数
    // 参数4 Time           → 延迟时间（秒）
    // 参数5 bLoop          → true=重复执行, false=只执行一次
    // 参数6 InitialStartDelay → 第一次执行前的额外延迟（默认0）
    //
    // ⚠️ 注意：计时器依赖于Tick()来更新
    // 如果World暂停(Tick被禁用)，计时器也会停止
}


// ==================== 引信完成：执行爆炸 ====================
void AGrenade::OnFuseComplete()
{
    UE_LOG(LogTemp, Warning, TEXT("💥 手榴弹爆炸！位置: %s"), *GetActorLocation().ToString());

    // 步骤1：构造一个"虚拟的"碰撞结果 —— 因为引信引爆不是由物理碰撞触发的
    FHitResult ExplosionHit;
    ExplosionHit.Location = GetActorLocation();          // 以手榴弹当前位置为爆炸中心
    ExplosionHit.ImpactNormal = FVector::UpVector;       // 爆炸"向上"（虽然爆炸是全方向的）
    ExplosionHit.Actor = this;                           // Actor是手榴弹自己
    // FHitResult 通常由射线/碰撞检测返回
    // 但这里我们手动构造一个，传给Explode_CPP


    // 步骤2：先应用爆炸冲击力（让物理物体飞起来）
    if (bApplyExplosionImpulse)
    {
        ApplyExplosionImpulse();
    }

    // 步骤3：再应用爆炸伤害（让范围内的Pawn受伤）
    ApplyExplosionDamage();

    // 步骤4：生成爆炸视觉特效和音效（复用基类的特效生成逻辑）
    // 通过调用基类的 OnProjectileHit 间接生成（但跳过Destroy逻辑）
    // 更好的做法是直接调用特效生成函数
    if (ExplosionParticle)
    {
        UGameplayStatics::SpawnEmitterAtLocation(
            GetWorld(),
            ExplosionParticle,
            ExplosionHit.Location,
            FRotator::ZeroRotator,  // 爆炸粒子不需要特殊朝向
            FVector(1.0f),
            true
        );
    }
    if (ExplosionSound)
    {
        UGameplayStatics::PlaySoundAtLocation(
            GetWorld(),
            ExplosionSound,
            ExplosionHit.Location
        );
    }


    // 步骤5：调试可视化 —— 显示爆炸范围球体
    if (bDebugExplosion)
    {
        DrawDebugSphere(
            GetWorld(),
            ExplosionHit.Location,  // 球心
            ExplosionRadius,        // 半径
            32,                      // 分段数（更多分段=更圆的球）
            FColor::Orange,          // 橙色
            false,                   // 不永久
            DebugDrawDuration        // 指定持续时间
        );
    }


    // 步骤6：延迟销毁手榴弹Actor
    // 延迟销毁是为了让爆炸粒子有时间播放完毕
    SetLifeSpan(0.5f);
    // 0.5秒后自动调用Destroy()
    // 为什么不在爆炸后立即Destroy？
    // 因为子组件（如粒子系统组件）需要时间来渲染效果
}


// ==================== 爆炸范围检测与伤害 ====================
void AGrenade::ApplyExplosionDamage()
{
    // ---- 步骤1：使用球体扫描检测爆炸范围内的所有Pawn ----

    FVector ExplosionCenter = GetActorLocation();  // 爆炸中心 = 手榴弹当前位置

    // 配置碰撞查询参数
    FCollisionQueryParams QueryParams;
    QueryParams.AddIgnoredActor(this);       // 忽略手榴弹自己
    // 如果有发射者，也忽略发射者（防止炸到自己）
    // if (GetInstigator()) QueryParams.AddIgnoredActor(GetInstigator());
    QueryParams.bTraceComplex = false;       // 用简单碰撞 —— 爆炸不需要精确命中点


    // 执行球体扫描（多物体版本）
    TArray<FHitResult> HitResults;  // 接收所有命中结果
    bool bHit = GetWorld()->SweepMultiByChannel(
        HitResults,                                  // 输出：所有命中物体
        ExplosionCenter,                             // 扫描起点
        ExplosionCenter + FVector(0, 0, 1),         // 扫描终点（几乎不变）
        // 这里使用几乎不变的终点，让扫瞄变成"原地球体检测"
        // 这是实现"球形范围检测"的常用技巧
        FQuat::Identity,                             // 不旋转球体
        ECC_Pawn,                                    // 只检测Pawn通道（敌人/玩家）
        FCollisionShape::MakeSphere(ExplosionRadius), // 球体形状（半径=爆炸半径）
        QueryParams                                  // 查询参数
    );


    // ---- 步骤2：如果没有命中任何东西，记录日志并返回 ----
    if (!bHit || HitResults.Num() == 0)
    {
        UE_LOG(LogTemp, Log, TEXT("爆炸范围内没有检测到任何Pawn"));
        return;
    }

    UE_LOG(LogTemp, Warning, TEXT("爆炸范围内检测到 %d 个Pawn"), HitResults.Num());


    // ---- 步骤3：对每个命中的Pawn计算并应用衰减伤害 ----
    int32 ActorsDamaged = 0;  // 计数器：受到了伤害的Actor数量

    for (const FHitResult& Hit : HitResults)
    {
        // 3.1 获取命中的Actor
        AActor* HitActor = Hit.GetActor();
        if (!HitActor)
        {
            continue;  // 空Actor，跳过
        }

        // 3.2 计算命中点到爆炸中心的距离
        float Distance = FVector::Dist(ExplosionCenter, Hit.Location);
        // Dist 函数返回两点之间的欧几里得距离（厘米）


        // 3.3 🔑 核心逻辑：根据距离计算衰减伤害
        // 使用线性插值（Lerp）计算伤害

        // 首先将距离映射到 0~1 的范围
        float DistanceRatio = FMath::Clamp(Distance / ExplosionRadius, 0.0f, 1.0f);
        // DistanceRatio = 0.0 → 在爆炸中心（距离=0）
        // DistanceRatio = 0.5 → 在爆炸半径的一半处
        // DistanceRatio = 1.0 → 在爆炸半径边缘（距离=ExplosionRadius）
        // DistanceRatio > 1.0 → 不在爆炸范围内（Clamp后=1.0）

        // 用Lerp计算最终伤害：从MaxDamage线性过渡到MinDamage
        float FinalDamage = FMath::Lerp(MaxExplosionDamage, MinExplosionDamage, DistanceRatio);
        // FMath::Lerp(A, B, Alpha) = A + (B - A) * Alpha
        // Alpha=0.0 → 返回A（MaxDamage，中心伤害）
        // Alpha=0.5 → 返回(A+B)/2（中点伤害）
        // Alpha=1.0 → 返回B（MinDamage，边缘伤害）

        // 也可以使用非线性衰减，让伤害衰减更"自然"：
        // float FinalDamage = MaxExplosionDamage * FMath::Square(1.0f - DistanceRatio);
        // 平方衰减：距离一半时只有25%伤害（而不是线性的50%）


        // 3.4 记录日志
        UE_LOG(LogTemp, Log,
            TEXT("  爆炸命中: %s | 距离: %.1f cm | 伤害: %.1f (衰减%.0f%%)"),
            *HitActor->GetName(),
            Distance,
            FinalDamage,
            DistanceRatio * 100.0f  // 衰减百分比
        );


        // 3.5 应用伤害到目标Actor
        // 实际项目中使用伤害系统，这里用日志代替
        // UGameplayStatics::ApplyDamage(
        //     HitActor,
        //     FinalDamage,
        //     GetInstigatorController(),
        //     this,
        //     UDamageType::StaticClass()
        // );

        ActorsDamaged++;  // 计数+1
    }

    UE_LOG(LogTemp, Warning, TEXT("爆炸共对 %d 个Actor造成伤害"), ActorsDamaged);


    // ---- 步骤4：调试可视化——对每个命中的Actor画线 ----
    if (bDebugExplosion)
    {
        for (const FHitResult& Hit : HitResults)
        {
            // 从爆炸中心到命中位置画红色虚线
            DrawDebugLine(
                GetWorld(),
                ExplosionCenter,
                Hit.Location,
                FColor::Red,
                false,
                DebugDrawDuration,
                0,
                1.0f  // 线宽1像素
            );
        }
    }
}


// ==================== 爆炸冲击波 ====================
void AGrenade::ApplyExplosionImpulse()
{
    FVector ExplosionCenter = GetActorLocation();

    // ---- 步骤1：检测爆炸范围内的所有物理物体 ----
    // 这次检测 PhysicsBody 通道，而不是 Pawn
    // 目的是找到所有受物理模拟的物体并推动它们

    TArray<FHitResult> HitResults;
    FCollisionQueryParams QueryParams;
    QueryParams.AddIgnoredActor(this);
    QueryParams.bTraceComplex = false;

    GetWorld()->SweepMultiByChannel(
        HitResults,
        ExplosionCenter,
        ExplosionCenter + FVector(0, 0, 1),
        FQuat::Identity,
        ECC_PhysicsBody,                             // 只检测物理物体通道
        FCollisionShape::MakeSphere(ExplosionRadius),
        QueryParams
    );


    // ---- 步骤2：对每个物理物体施加放射状冲量 ----
    for (const FHitResult& Hit : HitResults)
    {
        UPrimitiveComponent* HitComponent = Hit.GetComponent();
        if (!HitComponent)
        {
            continue;  // 没有组件，跳过
        }

        // 检查组件是否在物理模拟中
        if (!HitComponent->IsSimulatingPhysics())
        {
            continue;  // 不在物理模拟中，跳过
        }

        // 计算距离衰减
        float Distance = FVector::Dist(ExplosionCenter, Hit.Location);
        float Falloff = FMath::Clamp(1.0f - (Distance / ExplosionRadius), 0.0f, 1.0f);

        // 计算实际冲量大小
        float ActualImpulse = ExplosionImpulse * Falloff;

        // 施加放射状冲量（从爆炸中心向外推开物体）
        HitComponent->AddRadialImpulse(
            ExplosionCenter,                        // 爆炸中心
            ExplosionRadius,                        // 爆炸半径
            ActualImpulse,                          // 冲量强度
            ERadialImpulseFalloff::RIF_Linear,      // 线性衰减
            true                                    // bVelocityChange（直接修改速度）
        );

        UE_LOG(LogTemp, Log,
            TEXT("  冲击波推动: %s | 力度: %.1f (距离: %.1f cm)"),
            *HitComponent->GetName(),
            ActualImpulse,
            Distance
        );
    }
}
```

## AHitscanWeapon 即时命中武器

`AHitscanWeapon` 是一个独立的武器类（不继承自AProjectile）。它使用LineTrace实现"即时命中"（Hitscan）——按下扳机的瞬间就判定是否击中。

### 头文件：HitscanWeapon.h

```cpp
// ==================== HitscanWeapon.h ====================
// 即时命中武器 —— LineTrace射击 + 子弹散布 + 穿透检测
// 路径：Source/MyGame/Public/Weapons/HitscanWeapon.h

#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "HitscanWeapon.generated.h"

/**
 * AHitscanWeapon —— 即时命中武器
 *
 * 工作流程：
 * 1. 开火 → 计算子弹散布（随机偏移）
 * 2. LineTraceMultiByChannel → 检测弹道上的所有目标
 * 3. 对每个命中目标应用伤害
 * 4. 如果开启穿透 → 继续处理后续目标（伤害递减）
 * 5. 画出弹道可视化线（Debug）
 */
UCLASS()
class MYGAME_API AHitscanWeapon : public AActor
{
    GENERATED_BODY()

public:
    AHitscanWeapon();

    // 开火函数 —— 由玩家输入或AI调用
    UFUNCTION(BlueprintCallable, Category = "Weapon")
    virtual void Fire();

protected:
    // ==================== 组件 ====================

    // 武器网格体（枪的外观）
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Components")
    USkeletalMeshComponent* WeaponMesh;
    // 使用USkeletalMeshComponent而不是UStaticMeshComponent
    // 因为枪械通常有动画（换弹、后坐力）


    // ==================== 射击参数 ====================

    // 最大射击距离（厘米）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Range")
    float MaxRange = 50000.0f;
    // 50000cm = 500m —— 一个标准步枪的有效射程

    // 基础伤害
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Damage")
    float BaseDamage = 35.0f;

    // 爆头伤害倍率
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Damage")
    float HeadshotMultiplier = 2.0f;
    // 爆头伤害 = BaseDamage × HeadshotMultiplier


    // ==================== 子弹散布参数 ====================

    // 最大散布角度（度）
    // 子弹实际方向 = 瞄准方向 + 在这个锥形范围内的随机偏移
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Spread")
    float MaxSpreadAngle = 2.0f;
    // 2度的散布 —— 在100米处散布约3.5米的圆（近似）
    // 散布~0.5° = 狙击枪精度
    // 散布~5°   = 霰弹枪精度
    // 散布~10°  = 腰射精度

    // 连续射击时散布增加量（每发）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Spread")
    float SpreadIncreasePerShot = 0.3f;

    // 散布恢复速度（度/秒，不开火时散布逐渐缩小）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Spread")
    float SpreadRecoveryRate = 5.0f;

    // 当前散布角度（运行时变化）
    float CurrentSpreadAngle = 0.5f;
    // 初始散布很小（第一发很准）
    // 连续开火会增加散布

    // 最小散布角度
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Spread")
    float MinSpreadAngle = 0.1f;

    // 最大散布角度（散布不会超过这个值）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Spread")
    float MaxCumulativeSpread = 8.0f;


    // ==================== 穿透参数 ====================

    // 是否开启子弹穿透
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Penetration")
    bool bEnablePenetration = false;

    // 最大穿透次数（子弹能穿过几个敌人）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Penetration",
        meta = (EditCondition = "bEnablePenetration"))
        // EditCondition确保只在开启穿透时才能编辑此属性
    int32 MaxPenetrations = 2;
    // 2 = 子弹最多能穿过2个敌人，命中第3个时停止

    // 每次穿透后的伤害倍率
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Penetration",
        meta = (EditCondition = "bEnablePenetration", ClampMin = "0.0", ClampMax = "1.0"))
    float PenetrationDamageMultiplier = 0.7f;
    // 0.7 = 每穿透一个目标，伤害变为原来的70%
    // 第1个目标：100%伤害
    // 第2个目标：70%伤害
    // 第3个目标：49%伤害


    // ==================== 弹道可视化参数 ====================

    // 是否画出弹道线（仅开发阶段）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Debug")
    bool bDrawBulletTrace = true;

    // 弹道线颜色
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Debug")
    FColor BulletTraceColor = FColor::Red;

    // 弹道线持续时间（秒）
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Debug")
    float TraceDrawDuration = 2.0f;

    // 命中点标记颜色
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Debug")
    FColor HitMarkerColor = FColor::Green;


    // ==================== 碰撞通道设置 ====================

    // 射击检测使用的碰撞通道
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Weapon|Collision")
    TEnumAsByte<ECollisionChannel> WeaponTraceChannel = ECC_Visibility;
    // 也可以使用自定义通道（如在Project Settings中创建的"Weapon"通道）
    // 使用自定义通道可以精细控制哪些物体被命中、哪些物体被忽略


    // ==================== 辅助方法 ====================

    // 获取射击起点（通常是从枪口Socket位置）
    UFUNCTION(BlueprintCallable, Category = "Weapon")
    virtual FVector GetMuzzleLocation() const;

    // 获取瞄准方向（应用了散布之后的最终方向）
    UFUNCTION(BlueprintCallable, Category = "Weapon")
    virtual FVector GetAimDirection() const;

    // 计算子弹散布偏移
    // 返回一个在散布锥内的随机方向
    FVector ApplySpread(const FVector& AimDirection) const;

    // 更新散布（Tick中调用——自动恢复）
    void UpdateSpread(float DeltaTime);

    // 处理单个命中的目标
    void ProcessHit(const FHitResult& Hit, float Damage);
};
```

### 实现文件：HitscanWeapon.cpp

```cpp
// ==================== HitscanWeapon.cpp ====================
// 即时命中武器实现
// 路径：Source/MyGame/Private/Weapons/HitscanWeapon.cpp

#include "HitscanWeapon.h"
#include "Components/SkeletalMeshComponent.h"
#include "Engine/World.h"
#include "DrawDebugHelpers.h"
#include "Kismet/GameplayStatics.h"
#include "GameFramework/Character.h"
#include "Engine/EngineTypes.h"


// ==================== 构造函数 ====================
AHitscanWeapon::AHitscanWeapon()
{
    // 步骤1：设置基本属性
    PrimaryActorTick.bCanEverTick = true;  // 需要Tick来恢复散布
    // Tick频率默认与帧率相同

    // 步骤2：创建武器网格体
    WeaponMesh = CreateDefaultSubobject<USkeletalMeshComponent>(TEXT("WeaponMesh"));
    RootComponent = WeaponMesh;
    // SkeletalMeshComponent用于枪械模型（支持换弹动画等）


    // 步骤3：MuzzleSocket名称
    // 在骨骼网格体的骨骼树上设置一个Socket作为枪口位置
    // 注意：这个字符串需要在骨骼网格体编辑器中设置匹配的Socket名称
    // static const FName MuzzleSocketName = FName(TEXT("MuzzleFlashSocket"));
}


// ==================== 开火主函数 ====================
void AHitscanWeapon::Fire()
{
    // ===== 第1步：安全检查 =====
    if (!GetWorld())
    {
        UE_LOG(LogTemp, Error, TEXT("Fire()失败: GetWorld() 返回 nullptr"));
        return;  // 世界不存在，不能开火
    }


    // ===== 第2步：获取射击起点 =====
    FVector MuzzleLocation = GetMuzzleLocation();
    // MuzzleLocation = 枪口Socket的世界坐标位置


    // ===== 第3步：获取瞄准方向并应用散布 =====
    FVector AimDirection = GetAimDirection();
    // AimDirection = 从摄像机/角色朝向获取的瞄准单位向量

    FVector FinalDirection = ApplySpread(AimDirection);
    // FinalDirection = 在AimDirection基础上加了随机散布的最终方向


    // ===== 第4步：计算射击终点 =====
    FVector TraceEnd = MuzzleLocation + (FinalDirection * MaxRange);
    // 终点 = 枪口位置 + 最终方向 × 最大射程
    // 这条射线最长500米（MaxRange=50000cm）
    // 实际命中距离通常远小于这个值


    // ===== 第5步：配置射线检测参数 =====
    FCollisionQueryParams QueryParams;
    QueryParams.bTraceComplex = false;  // 使用简单碰撞体，性能更好
    QueryParams.AddIgnoredActor(this);  // 忽略武器自己
    // 如果有持有者（角色），也忽略持有者
    if (AActor* OwnerActor = GetOwner())
    {
        QueryParams.AddIgnoredActor(OwnerActor);
        // 防止射线命中自己手中的枪或自己的身体
    }


    // ===== 第6步：执行射线检测 =====

    if (bEnablePenetration)
    {
        // ---- 穿透模式：使用 LineTraceMultiByChannel ----
        // 射线穿过多个目标，返回所有命中结果

        TArray<FHitResult> HitResults;
        bool bHit = GetWorld()->LineTraceMultiByChannel(
            HitResults,                 // 输出：所有命中结果（按距离排序）
            MuzzleLocation,             // 从枪口
            TraceEnd,                   // 到最大射程处
            WeaponTraceChannel,         // 碰撞通道（ECC_Visibility 或自定义通道）
            QueryParams                 // 查询参数
        );

        if (bHit)
        {
            // 遍历所有命中，最多处理 MaxPenetrations+1 个目标
            int32 TargetsHit = 0;
            float CurrentDamage = BaseDamage;

            for (const FHitResult& Hit : HitResults)
            {
                // 超出穿透次数上限，停止
                if (TargetsHit > MaxPenetrations)
                {
                    break;
                }

                // 处理当前目标
                ProcessHit(Hit, CurrentDamage);

                // 每穿透一次，伤害衰减
                CurrentDamage *= PenetrationDamageMultiplier;
                // 第1个目标：BaseDamage
                // 第2个目标：BaseDamage × 0.7
                // 第3个目标：BaseDamage × 0.7 × 0.7

                TargetsHit++;
            }

            UE_LOG(LogTemp, Log, TEXT("穿透射击命中了 %d 个目标"), TargetsHit);
        }
    }
    else
    {
        // ---- 非穿透模式：使用 LineTraceSingleByChannel ----
        // 射线命中第一个阻挡物体就停止

        FHitResult HitResult;
        bool bHit = GetWorld()->LineTraceSingleByChannel(
            HitResult,                  // 输出：命中结果
            MuzzleLocation,             // 从枪口
            TraceEnd,                   // 到最大射程处
            WeaponTraceChannel,         // 碰撞通道
            QueryParams                 // 查询参数
        );

        if (bHit)
        {
            // 处理唯一的命中目标
            ProcessHit(HitResult, BaseDamage);
        }
        else
        {
            UE_LOG(LogTemp, Log, TEXT("射击未命中任何目标"));
        }
    }


    // ===== 第7步：弹道可视化 =====
    if (bDrawBulletTrace)
    {
        // 如果没有命中，画出完整弹道线
        // 如果命中了，画出从枪口到命中点的线

        // 这里简化处理：画一条完整的红色弹道线
        DrawDebugLine(
            GetWorld(),
            MuzzleLocation,           // 从枪口
            TraceEnd,                 // 到最大射程处
            BulletTraceColor,         // 红色
            false,                    // 不永久
            TraceDrawDuration,        // 持续指定秒数
            0,                        // 默认深度
            1.5f                      // 1.5像素宽
        );

        // 在命中点画出绿球（如果命中了的话）
        // 注意：这里是简化版，在处理穿透模式时只会画第一个命中的位置
        // 完整版应该把可视化放在ProcessHit中
    }


    // ===== 第8步：更新散布 =====
    // 每次开火都会让散布变大（模拟连续射击的后坐力）
    CurrentSpreadAngle = FMath::Min(
        CurrentSpreadAngle + SpreadIncreasePerShot,  // 当前散布 + 增量
        MaxCumulativeSpread                          // 不超过最大散布
    );
    // 例如：CurrentSpreadAngle从0.5 → 0.8 → 1.1 → 1.4 ...
    // 但不会超过8.0度


    // ===== 第9步：触发开火音效和特效 =====
    // （省略音效和特效的触发——实际项目中会播放枪声、枪口火焰等）
}


// ==================== 获取枪口位置 ====================
FVector AHitscanWeapon::GetMuzzleLocation() const
{

    if (WeaponMesh)
    {
        // 获取名为 "MuzzleFlash" 的Socket的世界坐标位置
        FVector MuzzleSocketLocation = WeaponMesh->GetSocketLocation(FName(TEXT("MuzzleFlash")));
        // 如果Socket不存在，GetSocketLocation会返回组件的位置

        return MuzzleSocketLocation;
    }

    // 方法2：回退方案 —— 直接返回Actor位置
    // 当没有武器网格体时使用Actor自己的位置
    return GetActorLocation();

    // ⚠️ 注意：Socket需要在骨骼网格体编辑器（或Static Mesh编辑器）中手动创建
    // 常见的Socket命名：MuzzleFlash, BarrelEnd, EjectPort 等
}


// ==================== 获取瞄准方向 ====================
FVector AHitscanWeapon::GetAimDirection() const
{
    // 方法1：获取持有者的控制器视角方向（最常用）
    AActor* OwnerActor = GetOwner();
    // GetOwner返回这个武器的拥有者（通常是Character）

    if (OwnerActor)
    {
        // 尝试从Pawn/Character获取控制器
        APawn* OwnerPawn = Cast<APawn>(OwnerActor);
        // Cast 是UE的运行时类型转换，安全（失败返回nullptr）
        // 等价于 C++ 的动态类型转换

        if (OwnerPawn && OwnerPawn->GetController())
        {
            // 通过控制器获取视角方向
            FRotator ControlRotation = OwnerPawn->GetControlRotation();
            // GetControlRotation 返回控制器的旋转（玩家的视角旋转）

            FVector AimDir = ControlRotation.Vector();
            // FRotator::Vector() 返回该旋转对应的"前方"单位向量
            // 等价于：FRotationMatrix(ControlRotation).GetUnitAxis(EAxis::X)

            return AimDir;
        }
    }


    // 方法2：回退方案 —— 使用武器自己的前方方向
    return GetActorForwardVector();
    // 当无法获取Owner/Controller时使用
}


// ==================== 子弹散布 ====================
FVector AHitscanWeapon::ApplySpread(const FVector& AimDirection) const
{
    // ---- 如果散布角度为0，直接返回瞄准方向（无散布）----
    if (CurrentSpreadAngle <= 0.0f)
    {
        return AimDirection;
        // 无散布，子弹精确命中瞄准点
    }


    // ---- 生成随机散布偏移 ----
    // 散布是一个圆锥，其半角 = CurrentSpreadAngle（度）

    // 步骤1：将散布角度转换为弧度
    float HalfAngleRad = FMath::DegreesToRadians(CurrentSpreadAngle * 0.5f);
    // 半角 = 散布角度的一半
    // 例如：散布2° → 半角1°

    // 步骤2：在散布锥内生成随机方向
    // 方法：在球面上均匀采样，然后将采样限制在锥形范围内

    // 生成一个在散布锥内的随机单位向量
    // 使用FMath的随机函数生成随机球面坐标
    float Theta = FMath::FRandRange(0.0f, 2.0f * PI);
    // 方位角：0到360度的随机值
    // PI ≈ 3.14159

    float Phi = FMath::FRandRange(0.0f, HalfAngleRad);
    // 极角：0到半角的随机值
    // 这创建了一个在锥形范围内的随机方向

    // 将球面坐标转换为局部偏移方向
    FVector SpreadOffset(
        FMath::Sin(Phi) * FMath::Cos(Theta),  // X分量
        FMath::Sin(Phi) * FMath::Sin(Theta),  // Y分量
        FMath::Cos(Phi)                        // Z分量（接近1时在锥中心）
    );
    // 这个向量是在"本地空间"中的——Z轴指向锥形的中心轴方向


    // 步骤3：将局部偏移转换到世界空间
    // 创建一个坐标系，其中Z轴 = 瞄准方向
    FVector WorldZ = AimDirection.GetSafeNormal();  // 锥中心轴
    // GetSafeNormal 确保是单位向量（防止除以零）

    // 创建两个与瞄准方向正交的基向量（X轴和Y轴）
    FVector WorldX, WorldY;
    WorldZ.FindBestAxisVectors(WorldX, WorldY);
    // FindBestAxisVectors 找到与WorldZ正交的两个互相正交的向量
    // WorldX 和 WorldY 分别是锥形截面的水平方向和垂直方向

    // 将局部偏移转换到世界空间
    FVector FinalDirection =
        SpreadOffset.X * WorldX +  // X方向的分量
        SpreadOffset.Y * WorldY +  // Y方向的分量
        SpreadOffset.Z * WorldZ;   // Z方向的分量（沿着瞄准方向）

    // 确保结果是单位向量
    FinalDirection.Normalize();
    // Normalize 将向量缩放为单位长度


    // ---- 调试：比较原始方向和散布后方向 ----
    // float DeviationAngle = FMath::RadiansToDegrees(
    //     FMath::Acos(FVector::DotProduct(AimDirection, FinalDirection))
    // );
    // UE_LOG(LogTemp, Log, TEXT("散布偏移角度: %.2f 度"), DeviationAngle);

    return FinalDirection;
}


// ==================== 更新散布（恢复） ====================
void AHitscanWeapon::UpdateSpread(float DeltaTime)
{
    // 如果不处于最小散布，就逐渐恢复（减小）散布
    if (CurrentSpreadAngle > MinSpreadAngle)
    {
        // 每帧减小散布：散布 = 散布 - 恢复速率 × DeltaTime
        CurrentSpreadAngle -= SpreadRecoveryRate * DeltaTime;
        // 例如：SpreadRecoveryRate=5.0, DeltaTime=0.016s
        // 每帧减小 5.0 × 0.016 = 0.08度
        // 从最大散布8°恢复到最小0.1°需要 (8-0.1)/5 ≈ 1.6秒

        // 将散布夹在最小值和最大值之间
        CurrentSpreadAngle = FMath::Clamp(CurrentSpreadAngle, MinSpreadAngle, MaxCumulativeSpread);
        // 确保散布不会低于最小散布，也不会高于最大散布
    }


    // ---- 额外优化：只在散布值变化时才打印日志 ----
    // if (GEngine)
    // {
    //     GEngine->AddOnScreenDebugMessage(
    //         -1,                             // Key（-1=每次显示新消息）
    //         0.0f,                           // 持续时间（0=仅显示一帧）
    //         FColor::Yellow,                 // 颜色
    //         FString::Printf(TEXT("散布: %.2f°"), CurrentSpreadAngle)

}


// ==================== 处理命中目标 ====================
void AHitscanWeapon::ProcessHit(const FHitResult& Hit, float Damage)
{
    // ===== 步骤1：安全检查 =====
    AActor* HitActor = Hit.GetActor();
    if (!HitActor)
    {
        UE_LOG(LogTemp, Warning, TEXT("ProcessHit: HitActor 无效"));
        return;
    }


    // ===== 步骤2：检查是否命中角色（可以对其造成伤害）=====
    ACharacter* HitCharacter = Cast<ACharacter>(HitActor);
    // Cast<ACharacter> 判断HitActor是不是Character类型
    // 如果是，返回Character指针；如果不是，返回nullptr

    float FinalDamage = Damage;  // 初始伤害 = 传入的伤害值

    if (HitCharacter)
    {
        // ---- 爆头检测 ----
        // 通过检查命中的骨骼名称是否为"head"来判断是否爆头
        // 这要求骨骼网格体的头部骨骼命名为 "head"（或类似名称）
        if (Hit.BoneName == FName(TEXT("head")))
        // BoneName 只在命中SkeletalMeshComponent时有效
        // 需要开启 bTraceComplex=true 才能获得精确的骨骼命中信息
        {
            FinalDamage *= HeadshotMultiplier;  // 爆头伤害 ×2

            UE_LOG(LogTemp, Warning,
                TEXT("🎯 爆头！目标: %s | 伤害: %.1f (基础%.1f × %.1f倍爆头)"),
                *HitCharacter->GetName(),
                FinalDamage,
                Damage,
                HeadshotMultiplier
            );
        }
        else
        {
            UE_LOG(LogTemp, Log,
                TEXT("命中角色: %s | 部位: %s | 伤害: %.1f"),
                *HitCharacter->GetName(),
                *Hit.BoneName.ToString(),   // 命中的骨骼名称
                FinalDamage
            );
        }


        // ---- 应用伤害 ----
        // 实际项目中在这里调用伤害系统：
        // UGameplayStatics::ApplyPointDamage(
        //     HitCharacter,              // 目标
        //     FinalDamage,               // 伤害值
        //     Hit.ImpactNormal,          // 命中方向（用于击退方向）
        //     Hit,                       // 命中结果
        //     GetInstigatorController(), // 造成伤害的控制器
        //     this,                      // 造成伤害的Actor（本武器）
        //     UDamageType::StaticClass() // 伤害类型
        // );
    }
    else
    {
        // 命中环境物体（不是角色）—— 只记录日志
        UE_LOG(LogTemp, Log,
            TEXT("命中环境: %s | 位置: %s"),
            *HitActor->GetName(),
            *Hit.Location.ToString()
        );
    }


    // ===== 步骤3：生成命中特效 =====

    // 3.1 在命中点生成弹孔/粒子效果
    // SpawnBulletImpactEffect(Hit.Location, Hit.ImpactNormal);

    // 3.2 播放命中音效（根据物理材质选择不同音效）
    if (Hit.PhysMaterial.IsValid())
    {
        // 根据表面类型播放不同的击中音效
        // PlayImpactSound(Hit.PhysMaterial->SurfaceType, Hit.Location);
    }


    // ===== 步骤4：调试可视化 =====

    // 在命中点画绿球
    DrawDebugSphere(
        GetWorld(),
        Hit.Location,           // 命中点
        10.0f,                  // 半径10cm
        12,                     // 分段数
        HitMarkerColor,         // 绿色（或其他配置的颜色）
        false,                  // 不永久
        TraceDrawDuration       // 持续时间
    );

    // 在命中表面画法线方向（黄色短线）
    DrawDebugLine(
        GetWorld(),
        Hit.Location,                              // 从命中点
        Hit.Location + Hit.ImpactNormal * 50.0f,   // 到法线方向50cm处
        FColor::Yellow,                            // 黄色
        false,
        TraceDrawDuration,
        0,
        1.0f
    );


    // ===== 步骤5：输出调试信息到屏幕 =====
    if (GEngine)  // GEngine是全局引擎指针，确保它存在
    {
        // AddOnScreenDebugMessage 在屏幕上显示调试文字
        GEngine->AddOnScreenDebugMessage(
            -1,                              // Key（-1 = 每次都创建新消息，不覆盖旧消息）
            3.0f,                            // 显示持续时间（秒）
            FColor::White,                   // 文字颜色
            FString::Printf(                 // 格式化字符串
                TEXT("命中: %s | 距离: %.1fm | 伤害: %.1f"),
                *HitActor->GetName(),        // 命中目标名称
                Hit.Distance / 100.0f,       // 转换cm为m
                FinalDamage                  // 最终伤害
            )
        );
    }
}
```

## 完整的武器使用Actor（AWeaponCharacter）

要测试上述武器系统，你需要一个能持有和使用武器的角色。以下是简化版的使用示例：

```cpp
// ==================== WeaponCharacter.h ====================
// 简化的武器使用角色
// 路径：Source/MyGame/Public/WeaponCharacter.h

#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Character.h"
#include "WeaponCharacter.generated.h"

class AHitscanWeapon;
class AGrenade;

UCLASS()
class MYGAME_API AWeaponCharacter : public ACharacter
{
    GENERATED_BODY()

public:
    AWeaponCharacter();

protected:
    virtual void BeginPlay() override;
    virtual void Tick(float DeltaTime) override;
    virtual void SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent) override;

    // ==================== 武器相关 ====================

    // 当前装备的武器类（在编辑器中指定）
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category = "Weapon")
    TSubclassOf<AHitscanWeapon> WeaponClass;
    // TSubclassOf 限制只能选择指定类型的子类

    // 当前装备的武器实例（运行时生成）
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Weapon")
    AHitscanWeapon* CurrentWeapon;

    // 手榴弹类
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category = "Weapon")
    TSubclassOf<AGrenade> GrenadeClass;

    // ==================== 输入动作 ====================

    void OnFirePressed();       // 按下开火键
    void OnThrowGrenade();      // 按下投掷手榴弹键
};

// ==================== WeaponCharacter.cpp ====================
// 路径：Source/MyGame/Private/WeaponCharacter.cpp

#include "WeaponCharacter.h"
#include "HitscanWeapon.h"
#include "Grenade.h"
#include "Engine/World.h"

AWeaponCharacter::AWeaponCharacter()
{
    PrimaryActorTick.bCanEverTick = true;
}

void AWeaponCharacter::BeginPlay()
{
    Super::BeginPlay();

    // ---- 生成武器并装备 ----
    if (WeaponClass && GetWorld())
    {
        // SpawnActor 在当前位置生成一个Actor
        CurrentWeapon = GetWorld()->SpawnActor<AHitscanWeapon>(
            WeaponClass,                  // 要生成的类
            GetActorLocation(),           // 生成位置
            GetActorRotation()            // 生成旋转
        );

        if (CurrentWeapon)
        {
            // 将武器挂载到角色的Mesh上
            CurrentWeapon->AttachToComponent(
                GetMesh(),                          // 挂载到角色的骨骼网格体
                FAttachmentTransformRules::SnapToTargetIncludingScale, // 吸附到目标（包含缩放）
                FName(TEXT("WeaponSocket"))         // 挂载的Socket名称（需要在骨骼网格体编辑器中创建）
            );

            // 设置武器的拥有者
            CurrentWeapon->SetOwner(this);
        }
    }
}

void AWeaponCharacter::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);

    // 更新当前武器的散布恢复
    if (CurrentWeapon)
    {
        CurrentWeapon->UpdateSpread(DeltaTime);
    }
}

void AWeaponCharacter::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent)
{
    Super::SetupPlayerInputComponent(PlayerInputComponent);

    // 绑定开火按键（鼠标左键）
    PlayerInputComponent->BindAction(
        FName(TEXT("Fire")),       // 动作名称（在Project Settings → Input中定义）
        IE_Pressed,                 // 按下时触发
        this,                       // 回调对象
        &AWeaponCharacter::OnFirePressed  // 回调函数
    );

    // 绑定投掷手榴弹按键（G键）
    PlayerInputComponent->BindAction(
        FName(TEXT("ThrowGrenade")),
        IE_Pressed,
        this,
        &AWeaponCharacter::OnThrowGrenade
    );
}

void AWeaponCharacter::OnFirePressed()
{
    if (CurrentWeapon)
    {
        CurrentWeapon->Fire();
    }
}

void AWeaponCharacter::OnThrowGrenade()
{
    if (GrenadeClass && GetWorld())
    {
        // ---- 生成手榴弹 ----
        // 在角色前方生成手榴弹
        FVector SpawnLocation = GetActorLocation() + GetActorForwardVector() * 100.0f + FVector(0, 0, 100.0f);
        // 在角色前方100cm、上方100cm处生成

        AGrenade* Grenade = GetWorld()->SpawnActor<AGrenade>(
            GrenadeClass,
            SpawnLocation,
            GetActorRotation()
        );

        if (Grenade)
        {
            // ---- 发射手榴弹 ----
            // 计算投掷速度（向前上方45度）
            FVector ThrowDirection = GetActorForwardVector() + FVector(0, 0, 1.0f);
            // 向量(1,0,1) → 前方45度角（向前和向上等量）

            ThrowDirection.Normalize();       // 归一化为单位向量
            ThrowDirection *= 1500.0f;        // 乘以期望的速度大小

            Grenade->Launch(ThrowDirection);  // 调用Launch发射

            // 设置发射者（让手榴弹不会炸到自己）
            Grenade->SetOwner(this);

            UE_LOG(LogTemp, Log, TEXT("投掷手榴弹！方向: %s"), *ThrowDirection.ToString());
        }
    }
}
```
